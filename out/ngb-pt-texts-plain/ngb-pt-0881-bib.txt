༄༅། །རྒྱ་གར་སྐད་དུ། །ཨུ་ཏྲི་ན་བི་མན་ཏ་ཕྲོ་ཤ་པུ་ཤ་ཏན་ཏྲ། བོད་སྐད་དུ། དྲག་པོ་ངན་སྔགས་བསྐལ་པའི་དུམ་བུའི་རྒྱུད། འཇམ་དཔལ་གཤིན་རྗེ་གཤེད་ལ་ཕྱག་འཚལ་ལོ། །ཡང་དག་ངེས་པར་བཤད་པར་བྱ། །ལྔ་བརྒྱ་དུས་ཀྱི་སྙིགས་མ་ལ། །སྔགས་པ་ནང་རྒྱས་བྱེད་པའི་དུས། །ལྷ་སྲིན་མཐུ་འགྲན་བྱེད་པའི་དུས། །སྔགས་པ་བྱིན་རླབས་བྱེད་པའི་དུས། །མ་ཕུར་གཤིན་རྗེས་ག་ས་ཙམ་ན། །སྔ་བརྒྱ་དུས་ཀྱི་སྙིགས་མར་སླེབས་ཙམ་ན། །གྲོང་བསྙེན་ནང་ཁུང་མཐུ་འགྲན་དུས། །དེ་ཚེ་ཕྱག་ཆ་བསྐོར་བར་བྱ། །བགེགས་རིགས་སྟོངས་ཕྲག་བརྒྱད་ཅུ་ཡང་། །དེད་དཔོན་དྲེགས་ཆེན་བརྒྱད་ཅུར་འདུས།

ཡ་མ་ཡ་ཨག་ནེ་༄༅། ཁྲ་ཏྲུ༷་ཛ་ལ་ར༷། ཡ་མ་ཡ་ཁས་སྨྱོ། སྲོག་མྱོགས་རྦད། ཡ་མ་ཡ་ཕྲོ་ཕྲོ་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ། ཨ་ཡི་འོག་ཏུ་གང་དམིགས་གཞུག །དམིགས་པ་ཕོ་མོ་རང་གནད་འཇོམས། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། གཤིན་རྗེ་གཤེད་ཀྱི་ལེའུ་སྟེ་གསུམ་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །ལྷ་ཆེན་མཧྭའི་ཕྱག་ཆ་ནི། །ལྷ་ནི་མ་ཧྭ་དེ་བ་སྒོམ། །དྲག་པོ་ངན་སྔགས་འདི་ཡིས༷་བ༷སྒྲུབ། །དེ་བ་ཧཱུཾ༷་ཏྲི་མ་ཧྭ་དི་ཧྲེ་ལྷི་ནི་ཧཱུཾ༷་ཕཊ། དེ་བ་ཨེ་ཧུར་ཐུམས་ཛ། དེ་བ་ཏྲིག་ནན། དེ་བ་མུར་ཡ་རྦད། དེ་བ་ཨ་སྟ༷་པ་ཡ་ནན། །དེ་བ་ཨ་ཨག་ནེ་ཤ་ཏྲུ༷་ཛ་ལ་ར༷། དེ་བ་ཨ་ཁེ་ས་སྦྱོ། སྲོག་མྱོགས་རྦད། དེ་བ་ཨ་ཕྲོ་ཕྲོ་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ།ཨ་ཡི་འོག་ཏུ་གང་དམིགས་གཞུག །དམིགས་པ་ཕོ་མོས་རང་གནད་འཇོམས། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། མ་ཧྭ་དེ་བའི་ལེའུ་སྟེ་བཞི་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །མུ་སྟེགས་བཀའ་བཟློག་ཕྱག་ཆ་ནི། །ལྷ་ནི་ཀྭ་ལ་ཡཀྟ་བསྒོམ། །བཀའ་ཟློག་རང་སྔགས་བསྒྲུབ་པའི་རྫས། །བིམ་ཛོ་ནི་ནག་པོའི་ཙིསྟ་ཕྲ་མ། །ཡཀྟ་ཀུ་རུ་རྦད་མྱོགས་རྦད། མྱགས་རྦད། བིམ་ཛོ་ནི་སོ་ཧུར་ཐུམས་ཛ། །བིམ་ཛོ་ནི་སྤོ་ཏིག་སོག་ནན། བིམ་ཇོ་ནི་སྤོ་མུ་ར་ཡ་རྦད། བིམ་ཛོ་ནི་སྭ་སྟ༷་པ་ཡ་ནན། བིམ་ཛོ་ནི་སྲོ་ཨག་ནེ་ཤ་ཏྲུ༷་ཛ་ལ་ར༷། བིམ་ཛོ་ནི་སྦོ་ཁོས་སྨྱོ། དམྱག་རྦད། བིམ་ཛོ་ནི་སྤོ་སྒྲོ་ཕྲོ་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ། 
སྤོ་ཡི་འོག་ཏུ་གང་དམིགས་གཞུག །ལྷ་ཡི་དགོངས་པ་རང་གནད་འཇོམས། །དྲག་པོ་ལས་བདུན་འདིར་བསྟན་པས། །མུ་སྟེགས་བཀའ་བཟློག་རྣམ་པར་ཞབྲུགས། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། མུ་སྟེགས་ཀྭ་ལ་ཡཀྟའི་ལེའུ་སྟེ་ལྔ་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །ལས་ཀྱི་མ་མོའི་ཕྱག་ཆ་ནི། །ལྷ་ནི་ལས་ཀྱི་མ་མོ་བསྒོམ་། །སྔགས་ནི་བསྒྲུབ་སྔགས་མཁའ་འགྲོའི་སྙིང་། །མ་མ་་ག་རག་ཤ་མུ་ཏྲ་མ་མ་སྨྱོ། རུ་ལུ་རུ་ལུ་སྨྱོ། ཁ་མུ་ཏྲ་སྨྱོ། མ་༷མ་༷རུ་ལུ་མ་ཧུར་ཐུམས་ཛ། །མ་མ་རུ་ལུ་མིག་ནན། མ་མ་རུ་ལུ་མྱ་ར་ཡ་རྦད། མ་མ་རུ་ལུ་མ་མ་སྟ༷་པ་ཡ་ནན། མ་མ་རུ་ལུ་མ་ཨག་ནེ་ཤ་ཏྲུ༷་ཛ་ལ་རམ། མ་མ་རུ་ལུ་མ་ཁེ་ས་སྨྱོ། སྲོག་མྱོགས་རྦད། མ་མ་རུ་ལུ་རུ་ལུ་མ། ཕྲོ་ཕྲོ་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ། མ་ཡི་འོག་ཏུ་གང་དམིགས་གཞུག །དམིགས་པ་ཕོ་མོ་རང་གནད་འཇོམས། །མ་མོ་ལས་བདུད་རྣལ་འབྱོར་སོགས། །དྲག་པོ་ངན་སྔགས་བསྐལ་བ་དུམ་བུའི་རྒྱུད་ལས། ལས་ཀྱི་མ་མོ་མཁའ་འགྲོ་མའི་ལེའུ་སྟེ་དྲུག་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །བདུད་པོ་ལེགས་ལྡན་ཕྱག་ཆ་ནི། །རང་ཉིད་བྱེ་བ་གོང་རིར་བསྒོམ། །བསྒྲུབ་སྔགས་དྲག་པོ་འདི་བཟླས་སོ། །ད་ཤ་བག་ནག་ཚོགས་ཀྱང་ཚོགས་གེ་ས་མ་ཡ་ཕཊ། ད་ཤ་དུ་ཧུར་ཐུམས་ཛ། ད་ཤ་དུ་ཏིག་ནན། ད་ཤ་དུ་ནཱ་ཤཱ་ར་ཡ་ཕཊ། ད་ཤ་དུ་སྟ༷་པ་ཡ་ནན། ད་ཤ་དུ་ཨག་ནེ་ཤ་ཏྲུ༷་ཛ་ལ་ར༷། ད་ཤ་དུ་ཁེས་སྨྱོ། སྲོག་མྱོགས་རྦད། 
ད་ཤ་༄༅། །དུ་ཕྲོ་ཕྲོ་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ། དུ་ཡི་འོག་ཏུ་གང་དམིགས་གཞུག །དམིགས་པ་བཟླས་པས་ཕོ་མོ་འཇོམས། །བདུད་ཀྱི་ལས་བདུན་འཕྲུལ་གྱི་མིང་། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། དྲག་པོ་བདུན་གྱི་ལེའུ་སྟེ་བདུན་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །མ་བདུད་ནག་པོའི་ཕྱག་ཆ་བསྒོམ། །ལྷ་ནི་ཕོ་ལས་མོ་རུ་བསྒྱུར། །བསྒྲུབ་སྔགས་བདུད་མོའི་སྲོག་སྔགས་ནི། །ཡ་བདུད་༷ཧྭ་ཡ་སྒྲི་བ་དུ་ནག་མོ་ཙན་ཏྲི་ཕྲ་མ་དེ་སྦྱོ། སྒྲི་བ་དུ་ཧུར་ཐུམས་ཛ། སྒྲི་བ་དུག་ཏྲིག་ནན། སྒྲི་བ་དུ་མུ་ར་ཡ་ཕཊ། སྒྲི་བ་དུ་སྟ༷་པ་ཡ་ནན། སྒྲི་བ་དུ་ཨག་ནེ་ཤ་ཏྲུ༷་ཛ་ལ་ར༷། སྒྲི་བ་དུ་ཁོས་སྨྱོ། སྲོག་མྱོགས་རྦད། སྒྲི་བ་དུ་ཕྲོ་ཕྲོ་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ། དུ་ཡི་འོག་ཏུ་གང་དམིགས་གཞུག །བདུད་མོ་དམིགས་པས་བདུད་མོ་འཇོམས། །བདུད་མོ་ལས་བདུད་མོ་ཡི་གཤེད། །རང་དབང་མེད་པས་དྲང་ལས་བྱེད། དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། མ་བདུད་ནག་མོའི་ལེའུ་སྟེ་བརྒྱད་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །སྲིན་པོའི་ཕྱག་ཆ་རང་བསྐོར་ན། །རང་ཉིད་ལས་ཀྱི་སྲིན་པོར་བསྒོམ། །སྲིན་པོའི་སྲོག་སྔགས་བསྒྲུབ་པ་ནི། །རག་ཤ་ཕཊ་དུ་ནེ་དུ་ནེ། ཨག་ཨག་ཨག་མོ་སན་ཛ་ཡ། བྱི་མ་ར་ཧ་ཧ་ཧི་ར༷་སྙིང་ཁ་རག །ཨེ་ཏོ་མ་མོ་ལི་བྱིན་ཙ་ཡ་འུར་ལྭ རག་ཤ་ར་ཧུར་ཐུམས་ཛ། རག་ཤ་ར་མ་ར་ཡ་ཕཊ། རག་ཤ་ར་སྟ༷་པ་ཡ་ནན། 
རག་ཤ་ར་ཨག་ནེ་ཤ་ཏྲུ༷་ཛ་ལ་ར༷། རག་ཤ་ར་ཁེས་སྨྱོ། སྲོག་མྱོགས་རྦད། རག་ཤ་ར་ཕྲོ་ཕྲོ༷་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ། ར་ཡི་འོག་ཏུ་གང་དམིགས་གཞུག །སྲིན་པོའི་དམིགས་པས་སྲིན་པོ་འཇོམས། །སྲིན་པོའི་ལས་བདུན་སྲིན་པོའི་གཤེད། །རང་དབང་མེད་པར་རང་ལས་བྱེད། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། རག་ཤ་སྲིན་པོའི་ལེའུ་སྟེ་དགུ་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །རག་མོ་སྲིན་མོའི་ཕྱག་ཆ་བསྐོར། །ལྷ་ནི་ཕོ་ལ་སྲིན་མོར་བསྒོམ། །སྲིན་མོ་བསྒྲུབ་སྔགས་བཟླས་པས་གཏེམ། །རག་མོ༷་ཏེ༷་ཀེ་ཛ་ཛ་རུ་ལུ་རུ་ལུ་ར་ཏྲི་ཀྲབས་ཀྲབས་རག་མོ་ར་ཧུར་ཐུམས་ཛ། རག་མོ་ཏྲིག་ནན་རག་མོ་མྱ་ར་ཡ་ཕཊ། རག་མོ་ར་སྟ༷་པ་ཡ་ནན།མོ་ར་ཨག་ནེ་ཤ་ཏྲུ༷་ཛ་ལ་ར༷། རག་མོ་ཁེས་སྨྱོ། སྲོག་མྱོགས་རྦད། །རག་མོ་ར་ཕྲོ་ཕྲོ་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ། ར་ཡི་འོག་ཏུ་གང་དམིགས་གཞུག་པར་བྱ། །སྲིན་མོའི་སྟོབས་ཀྱིས་སྲིན་མོ་འཇོམས། །སྲིན་མོའི་ལས་བདུན་སྲིན་མོ་གཤེད། །རང་དབང་མེད་པར་རང་ལས་བྱེད། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། སྲིན་མོའི་ལེའུ་སྟེ་བཅུ་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །ཡཀྟ་གནོད་སྤྱིན་ཕྱག་ཆ་ནི། །བསྒོམ་པའི་ལྷ་སྟོབས་ཀུ་བེ་ར། །བཟླས་པའི་བསྒྲུབ་སྔགས་དྲང་བ་ནི། །ཡཀྟ་རྫ་པ་ལ་ཛ་ལེན་ས་མ་ཡ། མེ་ཀུ་བེ་ར་ཧཱུཾ༷་ཕཊ། ཡཀྟ་ཀྟ་ཧུར་ཐུམས་ཛ། ཡཀྟཀྟཏྲིགནན། ཡཀྟཡཀྟ་མྱ་ར་ཡ་ཕཊ། 
ཡཀྟ་ཀྟ་སྟ༷་པ་ཡ་༄༅། །ནན། ཡཀྟ་ཀྟ་ཨག་ནེ་ཤ་ཧྲུ༷་ཛ་ལ་ར༷། ཡཀྟ་ཀྟ་ཁེས་སྨྱོ། སྲོག་མྱོགས་རྦད། ཡཀྟ་ཀྟ་སྒྲོ་ཕྲོ༷་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ། ཀྟ་ཡི་འོག་ཏུ་གང་དམིགས་གཞུག་བྱས་ལ། །གནོད་སྤྱིན་སྟོབས་ཀྱིས་གནོད་སྤྱིན་འཇོམས། །གནོད་སྦྱིན་ལས་བདུན་གནོད་སྤྱིན་གཤེད། །རང་དབང་མེད་པར་རང་ལས་བྱེད། དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། གནོད་སྤྱིན་གྱི་ལེའུ་སྟེ་བཅུ་གཅིག་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །ཀྟ་ཀྟ་གདོན་མོའི་ཕྱག་ཆ་ནི། །སྒོམ་པའི་ལྷ་ནི་གནོད་སྤྱིན་མོ། །བཟླས་པའི་བསྒྲུབ་སྔགས་རྒྱུན་དུ་དྲངས། །ཡ༷ག་༷མོ་གོང་མོ་སྙིང་ཁ་རག་མོ་སྨྱོ། ཡག་མོ་ཀྟ་ཧུར་ཐུམས་ཛ། ཡཀྟ་མོ་ཀྟ་ཏྲིག་ནན། ཡག་མོ་ཀྟ་སྨྱུ་ར་ཡ་ཕཊ། ཡག་མོ་ཀྟ་སྟ༷་པ་ཡ་ནན། ཡག་མོ་ཀྟ་ཨག་ནེ་ཤ་ཏྲུ༷་ཛ་ལ་ར༷། །ཡཀྟ་ཁེས་སྨྱོ། སྲོག་མྱོགས་རྦད། ཡག་མོ་ཀྟ་ཕྲོ་ཕྲོ་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ། ཀྟ་ཡི་འོག་ཏུ་གང་དམིགས་གཞུག །ཡག་མོའི་སྟོབས་ཀྱིས་ཡག་མོ་འཇོམས། །ཡག་མོའི་ལས་བདུན་ཡག་མོའི་གཤེད། །རང་དབང་མེད་པར་རང་ལས་བྱེད། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། གནོད་སྤྱིན་ནག་མོའི་ལེའུ་སྟེ་བཅུ་གཉིས་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །ཀླུ་བདུད་ཕྱག་ཆ་བསྐོར་བ་ནི། །བསྒོམ་པའི་ལྷ་ནི་བྱ་ཁྱུང་མ། །ཀླུ་བདུད་མགོ་བོ་བྱ་ཁྱུང་བསྒོམ། ཁུན་དུ་ལས་སྔགས་འདི་བརྗོད་དོ། །ནཱ་ག་ལེ། དུ༷་པ་ལེ། ཤུན་ཤུ་ཐུ། 
བ་ཡུ་ཁ་བི་ཏི་ཏི་སྟན་པ། ཁྲམ་ཧུ་ཕུ་ཧཱུཾ༷་ཕཊ། ན་ག་ཕུ་ཧུར་ཐུམས༷ ་ཛ༷་ན་ག་ཕུ་ཏྲིག་ནན། ནཱ་ག་ཕུ་མྱ་ར་ཡ་ཕཊ། ནཱ་ག་ཕུ་སྟ༷་པ་ཡ་ནན། ནཱ་ག་ཕུ་ཏྲིག་ནན། ནཱ་ག་ཕུ་མྱ་ར་ཡ་ཕཊ། ནཱ་ག་ཕུ་སྟ༷་པ་ཡ་ནན། ནཱ་ག་ཕུ་ཨག་ནེ་ཤ་ཏྲུ༷་ཛ་ལ་རེ། ནཱ་ག་ཕུ་གེས་སྨྱོ། སྲོག་མྱོགས་རྦད། ནཱ་ག་ཕུ་ཕྲོ་ཕྲོ་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ། ཕུ་ཡི་འོག་ཏུ་གང་དམིགས་གཞུག །ནཱ་ག་སྟོབས་ཀྱིས་ནཱ་ག་འཇོམས། །ནཱ་གའི་ལས་བདུན་ནཱ་གའི་གཤེད། །རང་དབང་མེད་པར་རང་ལས་བྱེད། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། ཀླུ་བདུད་ཀྱི་ལེའུ་སྟེ་བཅུ་གསུམ་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །བཙན་གྱཕྱག་ཆ་བསྐོར་བྱ་ན། །བསྒོམ་པའི་ལྷ་ནི་ལྷ་བཙན་ཡ་ཤུད་བསྒོམ། །དྲག་པོ་བསྒྲུབ་སྔགས་རྒྱུན་དུ་བཟླས། །ཙན་ཏྲི་ཧྲི་ཏྲི་ར༷་ཁ་རག་ཏྲ་དུ་ཟི་རྦད། མ་མ་སྒག་ཤ་ཟ་བཞི་མཚོ་ལ་བཛྲ། ཙན་ཏྲི་ཙ་ཧཱུཾ༷་ཐུམས་ཛ། ཙན་ཏྲིག་ཙ་ཏྲིག་ནན། ཙན་ཏྲི་ཙ་བཱ་ར་ཡ་ཕཊ། ཙན་ཏྲི་ཙ་སྟ༷་པ་ཡ་ཕཊ། ཙན་ཏྲི་ཙ་ཨག་ནེ་ཤ་ཏྲུ༷་ཛ་ལ་ར༷། ཙན་ཏྲི་ཙ་ཁེས་སྨྱོ། སྲོགས་མྱོགས་རྦད། ཙན་ཏྲི་ཙ་སྐྲོ་ཕྲོ༷་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ། ཙ་ཡི་འོག་ཏུ་གང་དམིགས་གཞུག །བཙན་གྱི་སྟོབས་ཀྱིས་བཙན་རྣམས་འཇོམས བཙན་གྱི་ལས་བདུན་བཙན་གྱི་གཤེད། །རང་དབང་མེད་པར་རང་ལས་བྱེད། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། དྲག་པོ་བཙན་གྱི་ལེའུ་སྟེ་བཅུ་བཞི་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ།

དུ་ཡི་འོག་ཏུ་ཐེ་བྲང་༄༅། །གཞུག །གིང་ཆེན་སྟོབས་དྲག་ཤར་དེ་འཇོམས། ཁྲང་གིང་ལས་བདུན་ཐེ་བྲང་གཤེད། །རང་དབང་མེད་པར་རང་ལས་བྱེད། །གིང་ཆེན་ཕྱག་ཆ་རང་ལ་བསྐོར། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། དགྲ་ལྷ་རྡོ་རྗེ་ལེགས་པའི་ལེའུ་སྟེ་བཙྭ་བརྒྱད་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །དམ་སྲིའི་ཕྱག་ཆ་བསྐོར་བར་བྱ། དམ་སྲིའི་གཉེན་པོར་མ་མོ་བསྒོམ། །དམ་སྲི་སྲོག་སྙིང་བསྒྲུབ་འཛབ་དྲང་། །རོ་ཏྲི་རོ་ལ་ཡ་མ་པ་ཡ༷་ཏྲི་ཡ་སྲི་ཛ་རྦད། རུ་ཏྲི་སྲི་ཧུར་ཐུམས་ཛ། རུ་ཏྲི་སྲི་ཧྲིག་ནན། རུ་ཏྲི་སྲི་ས་མ་ར་ཡ་རྦད། རུ་ཏྲི་སྲི་སྟ༷་པ་ཡ་ནན། རུ་ཏྲི་སྲི་ཨག་ནེ་ཤ་ཏྲུ༷་ཛ་ལ་ར༷། རུ་ཏྲི་སྲི་ཁས་སྨྱོ། སྲོག་མྱོགས་བོད། རུ་ཏྲི་སྲི་ཕྲོ༷་ཕྲོ༷་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ། །སྲི་ཡི་འོག་ཏུ་དམ་སྲི་གཞུག །དམ་སྲི་སྟོབས་དྲག་བལ་མོ་འཇོམས། །དམ་སྲིའི་ལས་བདུན་ཕུང་འདྲེའི་གཤེད། །རང་དབང་མེད་པར་རང་ལས་བྱེད། །དམ་སྲིའི་ཕྱག་ཆ་རང་ལ་བསྐོར། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། དམ་སྲི་འདྲེ་མིག་གི་ལེའུ་སྟེ་བཅུ་དགུ་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །འགོང་པོ་ཕྱག་ཆ་བསྐོར་བར་བྱ། །འགོང་པོའི་གཉེན་པོར་སྲོག་བདག་བསྒོམ། །འགོང་པོ་སྲོག་སྙིང་བསྒྲུབ་འཛབ་དྲང་། །ཧ༷་ཤ༷་ཧ༷་ཤ༷་ཏྲི། ཨོ་སེ་རྒྱལ་བ་ཤ་ནི་ཏྲི་ཏྲི་ཐང་ལང་དུ་ན་ཀ༷་ཀ༷་ཡ༷་ར་ཏྲི་རེ། །ཧ༷་ཤ༷་ཏྲི་ཏྲི་ཧུར་ཐུམས་ཛ། ཧ༷ཤ༷ཏྲིཧྲིགནན། ཧ༷ཤ༷ཏྲིཏྲི་སྟམ་པ་ཡ་ནན། 
ཧ༷་ཤ༷་ཏྲི་ཏྲི་ཨག་༄༅ ལ་ཐལ་བར་བསམ། །དྲུག་པ་འབྱུང་བ་རླུང་གཏོར་བས། །དུག་མཚོ་ཁོལ་མས་ཁྱེར་བསམ་ལ། །སྲིད་པ་ཆགས་འཇིག་འབྱུང་བ་དཀྲུག །ལྷ་མེད་འདྲེ་མེད་ཕན་གནོད་མེད། །ཆོས་ཉིད་རྟོགས་མེད་མཉམ་པའི་ངང་། །བསམ་གཏན་དེ་དྲུག་སྤྱི་རུ་གཅེས། །དགུག་གཟེར་ལ་སོགས་ལས་ཐབས་བདུན། །བདུན་བསྲུང་སྤྱི་དང་མཐུན་པར་སྦྱར། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། དྲག་པོ་བསམ་གཏན་གྱི་ལེའུ་སྟེ་ཉེར་གཅིག་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །ལོག་ནོན་ཕྱག་ཆ་ཁྱབ་པར་ཡིན། །ལྷ་སྲིན་གཏེམ་ཐབས་ཁྱད་པར་གསུམ། །བཛྲ་སརྦ་དུ་། ཨེ་ཨེ། ཧུར་ཧུར་ཐུམས་ཐུམས་ཛ་ཛ། ཨེ་ཨེ། ཏྲིག་ཏྲིག་ནན་ནན། ཨེ་ཨེ། མ་མ་ར་ར་ཡ་ཡ་རྦད་རྦད། ཨེ་ཨེ། སྟ༷་པ་ཡ་སྟ༷་པ་ཡ་ནན་ནན། ཨེ་ཨེ་ཨག་ཨག་ནེ་ནེ་ཤ་ཤ་ཏྲུ༷་ཏྲུ༷་ཛ་ཛ་ལ་ལ་ར༷་རེ་ཨེ་ཨེ་ཁེས་ཁེས་སྨྱོ་སྨྱོ། སྲོག་སྲོག་མྱོགས་མྱོགས་རྦད་རྦད། །ཨེ་ཨེ་ཕྲོ༷་ཕྲོ༷་ཕྲོ༷་ཕྲོ༷་ཧྲིལ་ཧྲིལ་ཧྲིལ་ཧྲིལ་ཤརཤར་ཤར་ཤར་ཕོབ་ཕོ༷བ། འདི་བཞིན་ཀུན་ལ་སྦྱར་བར་བྱ། །སྙིང་པོ་འབྲུ་གཅིག་གཏེམ་པ་ནི། །ཨེ་ཧུར་ཐུམས་ཛ། ཨ་ཏྲིག་ནན། ཨེ་བཱ་ར་ཡ་རྦད། ཨེ་ཏ༷་པ་ཡ་ནན། ཨེ་ཨ་ནེ་ཤ་ཏྲུ༷་ཛ་ལ་ར༷། ཨེ་ཁེས་སྨྱོ། སྲོག་མྱོགས་བཱ་ར་ཡ་རྦད། ཨེ་ཕྲོ་ཕྲོ་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ། འདིབཞིནགཞནལའང་སྦྱར་བར་བྱ། །སྒོམ་གཏེམའབྲུགཅིགགཏེམབྱས་ན། ཡ་ཧུར་ཐུམསཛཡཏྲིགནན། 
ཡམྱར་ཡ་རྦད། ཡ་སྟ༷་པ་ཡ་ནན། ཡ་ཨག་ནེ་ཤ་ཏྲུ༷་ཛ་ལ་ར༷། ཡ་ཁས་སྨྱོ། སྲོག་མྱོགས་རྦད། ཡ་ཕྲོ་ཕྲོ༷་ཧྲིལ་ཧྲིལ་ཤར་ཤར་ཕོབ། འདི་བཞིན་འོག་ནས་ཀུན་ལ་སྦྱར། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། ལོག་ནོན་གྱི་ལེའུ་སྟེ་ཉེར་གཉིས་པའོ།། །།དེ་ནས་ཡང་དང་བཤད་བྱ་བ། །རོ་ལངས་རྨེ་འཐབ་ནང་སྟོབས་ཕྱག་ཆ་འགྲན་པའི་ཚེ། ཁོའི་བསྒྲུབ་སྔགས་ལོག་གིས་བཟློག་ལ་གཏམ་པར་བྱ། །རང་གིས་བསྒྲུབ་སྔགས་རྣལ་མར་བྱ། ཁོ་ཡི་བསྒྲུབ་སྔགས་བཟློག་པ་ལ། །གཟིར་དང་སྲོག་ཕྲོག་གསད་པར་བྱ། །ཕཊ་ཧཱུཾ༷་ཡ་ནང་ཤ་རོ་ནི་པ་བཛྲ། ཨེ་ལྟོ། ཡ་མ་ས་ཛ་ཛ་སྣ་རྒྱ་ཨ། སྨྱོ་བདུན་བཛྲ་ཨོ། ཛ་ཐུམས་ཧུར་ཨེ་ཏན་དུ་སརྦ་བཛྲ། ནན༷་ཏྲི༷ག་ཨེ་སྟན་དུ་རྦད་སྭ་བཛྲ། རྦད་ཡ་ར་མ་ཨེ། སྟན་དུ་རྦ་ས་བཛྲ། ནན་ཡ་པ་སྟ༷། ཨེ་སྟན་དུ་རྦ་ས་བཛྲ་ཏྲུ༷་ཤ་ནི་ཨག་ཨེ་སྟན་དུ་རྦ་ཛ། རྦད་མྱོགས་སྲོག་སྦྱོ་ཁེས། ཨེ་བསྟན་དུ་རྦ་ས་བཛྲ་ཕོབ་ཤར་ཤར་ཧྲིལ་ཧྲིལ་ཕྲོ་ཕྲོ། ཨེ་སྟན་དུ་རྦ་ས་བཛྲ། ཨེ་ཡི་འོག་ཏུ་དགྲ་བགེགས་གཞུག །དམིགས་པ་ཕོ་མོ་ཡུལ་ནས་འཛོམ། །ཡ་མ་ར་ནི་ཙ་ཡསྟེ་ཡ། ཡསྟེ་ཡ་ར་ནི་ཡོ་ཏ་ཡ། ཡ་ཏ་ཡོ་ནི་རུ་ཏུ་མེ་ཡ། ཡ་མེ༷ ་ཏོ༷ ་ས་ཙ་ར་མ་ཡ། ཛ་ཐུམས་ཧུར་ཡ་མ་ཡ་ནནཏྲིགཡམཡ། རྦདཡ་ར་མ་ཡ་ཡ་མ། ནན་ཡ་པ་སྟ༷་པ་ཡ་མ་ཡ། ར༷་ལ་ཛ་ཏྲུ༷་ཤ་ནི། ཨག་ཡ་མ་ཡ། རྦད་མྱོགས་སྲོགས་སྨྱོ། ཡ་ཁེས་ཡ་མ་ཡ། ཕོབ་ཤར་ཤར་ཧྲིལ་ཧྲིལ་ཕྲོ་ཕྲོ། ཡ་མ་ཡ། 
ཨ་ཡི་སྟེང་དུ་གང་དམིགས་༄༅། །གཞུག །དམིགས་པ་རང་གནས་ཕོ་མོ་འཇོམས། དེ་བཞིན་ཀུན་ལ་སྦྱར་བར་བྱ། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། རོ་ལངས་རྨེ་འཐབ་ནང་སྟོབས་ཕྱག་ཆ་འགྲན་པའི་ལེའུ་སྟེ་ཉེར་གསུམ་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །ཐུན་མོང་སྤྱི་ཡི་ཕྱག་ཆ་ནི། །དཔལ་ཆེན་ཧེ་རུར་གསལ་བསྒོམ་ལ། །འཕྲིན་ལས་གཞུང་བསྲང་རང་སྔགས་བཟླས། །ཨོ་བཛྲ་ཀྲོ་དྷ་མ་མ་ཤི་ཧེ་རུ་ཀ །བཛྲ་ཀྲོ་དྷ་ཤ་རི་སྟ༷། ཨོ༷་རུ་ལུ་རུ་ལུ། ཧཱུཾ༷་སྨྱོ་ཧཱུཾ༷། འདི་མཇུག་ཕྱག་ཆའི་ལས་སྦྱར་བས། ཁྲོ་བོ་ཁྲག་འཐུང་གཙོ་འཁོར་བཅས་པའི་ཕྱག་ཆར་འགྲོ། །ཕྱག་ན་རྡོ་རྗེ་གསལ་སྒོམས་ལ། །འཕྲིན་ལས་གཞུང་བསྲངས་དྲག་སྔགས་བཟླས། །ཨོ་ནི་ལ༷་བ་ར་དྷ་ར་བཛྲ་ཕུ་ནི་ཧཱུཾ༷་ཕཊ། ཏབྱ་ཐ་སྤྱོ་སྨྱོ་བན། འདི་འཇུག་ཕྱག་ཆའི་ལས་སྦྱར་བས། །མཐུ་སྟོབས་བདག་པོའི་ཕྱག་ཆར་འགྲོ། །པདྨ་དབང་ཆེན་གསལ་བསྒོམས་ལ། །འཕྲིན་ལས་ག༷་བསྲང་བསྒྲུབ་སྔགས་བཟླས། །ཨོ་ཧྲི་པདྨ་ཏ་ཀྲིད་བཛྲ་ཀྲོ་དྷ་ཧ་ཡ་སྙི་བ་ཧུ་ལུ་ཧུ་ལུ་ཧཱུཾ༷་ཕཊ། འདི་གཞུག་ཕྱག་ཆའི་ལས་སྦྱར་བས། །པདྨ་དབང་ཆེན་ཕྱག་ཆར་འགྲོ། །རྡོ་རྗེ་གཞོན་ནུ་གསལ་བསྒོམས་ལས། །འཕྲན་ལས་གཞུང་བསྲངས་བསྒྲུབ་སྔགས་བཟླས། །ཨོ་བཛྲ་ཀི་ལི་ཀི་ལི་ཡ་སརྦ་བི་ནན་བ༷་ཧཱུཾ༷་ཕཊ། འདི་གཞུག་ཕྱག་ཆའི་ལས་སྦྱར་བས། །ཕུར་པའི་འཕྲིན་ལས་ཕྱག་ཆར་འགྲོ། །ཡ་མན་ཏ་ཀར་གསལ་བསྒོམས་ལས། 
།འཕྲིན་ལས་གཞུང་བསྲངས་རྩ་སྔགས་བཟླས། །ཨོ་ཨ་ཀྲོ་ཏེ་ཀ་ཡ་མན་ཏ་ཀ་ཧ་ན་མ་ཏ་བྷརྫི་ཕཊ། མུག་ཡེ་འདི་གཞུག་ཕྱག་ཆའི་ལས་སྦྱར་བས། །འཇམ་དཔལ་གཤིན་རྗེའི་ཕྱག་ཆར་འགྲོ། །བྱ་ཁྱུང་འཁྱིལ་པ་གསལ་བསྒོམས་ལ། །འཕྲིན་ལས་གཞུང་བསྲངས་བསྒྲུབ་སྔགས་བཟླས། །ཨོ་ག་རུ་ཌ་བཛྲ་དུན་ཏིང་ཀ་ཨ་ཀུ་ཤ་ཛ་ཛ། འདི་གཞུག་ཕྱག་ཆའི་ལས་སྦྱར་བས། །བྱ་ཁྱུང་གསེར་གྱི་ཕྱག་ཆར་འགྲོ། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། བཀའ་འཁོར་དྲུག་གི་ཐུན་མོང་སྤྱིའི་ལེའུ་སྟེ་ཉེར་བཞི་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །ཆེ་མཆོག་པདྨ་ཕྱག་རྡོར་དང་། །ཕུར་པ་གཤིན་རྗེ་བྱ་ཁྱུང་གསལ། །བསད་དང་གསོ་དང་དབབ་དང་བཅད། །ལྟས་ངན་གཟིར་དང་ནན་དང་བདུན། །གཅིག་ལའང་འབྱུང་བར་ཤེས་པར་བྱ། །དྲག་པོ་ངན་སྔགས་བསྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། སྤྱིའི་ལས་ཀྱི་ཐོ་བྱང་བསྟན་པའི་ལེའུ་སྟེ་ཉེར་ལྔ་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །སྤྱིར་ནི་རྡོ་རྗེ་གཙུག་ལག་དང་། །དྲག་སྔགས་གཙུག་ལས་གཉིས་ལས་རྡོ་རྗེ་གཙུག་གི་ནི། ཆོས་ཉིད་སྤྱི་གཏན་ལ་འབེབས་ཤིང་སྤྱི་རྒྱ་གཅོད་དོ། །དྲག་སྔགས་གཙུག་ལག་གི་ནི། གཙོ་བོར་ལས་རྣམ་པ་བདུན་དང་། སྤོ་བརྒྱ་དང་ལྔ་བཅུ་རྩ་གཅིག །སྡེ་དཔོན་བཅུ་དགུའི་རྫས་དང་། སྔགས་དང་། ཏིང་ངེ་འཛིན་དང་། རྟགས་དང་། འཁྲུལ་འཁོར་དང་། སྲོག་དང་ངེས་ཚིག་གཏན་ལ་
