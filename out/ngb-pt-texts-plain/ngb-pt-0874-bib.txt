དོན་འགྲོལ་རྫོགས་སོ།། །། ༄༅། །རྒྱ་གར་སྐད་དུ། བཛྲ་ཀྲི་ཡི་ཨུ་ཀྲ་ན་བི་ཤན་ཏིང་ཀ་ར་ཨ་མྲི་ཏ་ཀ་ར་ད་ཡ་ཙཀྲ་ཏན་ཏྲ་ནཱ་མ། བོད་སྐད་དུ། རྡོ་རྗེ་གཙུག་ལག་དྲག་པོ་ངན་སྔགས་ཞི་བར་བྱེད་པ་བདུད་རྩི་འབྱུང་བ་འཁོར་ལོའི་རྒྱུད་ཅེས་བྱ་བ། བཅོམ་ལྡན་འདས་དཔལ་ཀུན་ཏུ་བཟང་པོ་ལ་ཕྱག་འཚལ་ལོ། །དེ་ནས་བྱང་ཆུབ་སེམས་དཔའ་ཞི་བའི་ཚུལ་ཅན་ཕྱག་ན་རྡོ་རྗེ་བདེ་བའི་མཆོད་པ་ཐོགས་ཏེ། ཨོ་དཔལ་ཀུན་ཏུ་བཟང་པོ་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །ཨོ། བདེ་བར་གཤེགས་པ་བཅོམ་ལྡན་འདས། །བཀྲ་ཤིས་བརྒྱད་ཀྱིས་མཆོད་པར་བཞེས། །མ་འོངས་དུས་ཀྱི་སྙིགས་མ་སྲ །སེམས་ཅན་བསོད་ནམས་ཆུང་དུས་སུ། །རིན་ཆེན་བུམ་བཟང་དཔག་བསམ་ཤིང་། །འདོད་པ་ཇི་ལྟར་བསྐང་བར་བགྱི། །ཞེས་གསོལ་པས། ཨོ་དཔལ་ཀུན་ཏུ་བཟང་པོས་ཞི་བའི་ཚུལ་དགུ་སེམས་ཅན་མཐའ་ཡས་བསྐྱབ་པའི་ཕྱིར། 
ཚེ་བསྲིང་པ་དང་གསོ་བ་དྲག་པོའི་དབྱིངས་རྣམ་པར་དག་པ་རྡོ་རྗེ་ལྟར་བསྲུང་བའི་ཏིང་ངེ་འཛིན་ལ་སྙོམས་པར་ཞུགས་ནས། དེའི་རྣམ་རྟོག་བཀའ་སྩལ་པ། ཨེ་མ་གསང་བ་སྙིང་རྗེ་མཆོག །བདུད་རྩི་རྣམ་དག་ཆེར་གསང་བའི། །རིན་ཆེན་བུམ་བཟང་དཔག་བསམ་ཤིང་། །རྣམ་ཤེས་འཁོར་ཡུལ་བཅས་པ་ཡི། །གསང་བའི་འཁོར་ལོ་ལས་བྱང་བ། །ཡི་གེ་བཏུ་ བ་བསྟན་པ་ནི། །འཁོར་ལོ་རང་འདྲེན་བཏུ་སྒྲ་ཡིས། །དགྲ་ཡི་བདུད་རྩི་གཏན་ལ་འབེབས། །འཁོར་ལོ་དབུས་ཀྱི་མི་འགྱུར་ལས། །འཁོར་ལོ་རང་འདྲེན་ཨ་བླངས་ཏེ། །ཡུལ་འཁོར་ཁྲ་གླང་སྟེང་ནས་བརྒྱན། །ཡུལ་འཁོར་ཆུ་སྲིན་ཐིག་ལེའི་སྟེང་། །འོག་ཏུ་ཁྱི་ཐ་སྦྲུལ་གྱིས་མཛེས། །ཡེ་ཤེས་ལྔ་རྫོགས་གསང་སྔགས་ཏེ། །གསང་སྔགས་ཀུན་གྱི་མགོ་ལ་བཞག །འཁོར་ལོ་ནུབ་བྱང་རང་འདྲེན་ཡ། །ནུབ་ཀྱི་འཁོར་ལོ་རང་དྲེན་མ། །དེ་དང་ལྷོ་ཕྱོགས་འཁོར་ལོ་ཡི། །རང་འདྲེན་ན་རུ་སྦྱར་བར་བྱ། །ལྷོ་ཕྱོགས་འཁོར་ལོ་རང་འདྲེན་དང་། །བྱང་ཤར་འཁོར་ལོ་རང་འོག་བརྒྱན། །སྟེང་དུ་རྟགས་དང་རི་རབ་བརྒྱན། །རེ་ལ་ལྷོ་ཕྱོགས་འཁོར་ལོ་ཡིས། །རང་འདྲེན་ད་ཡིས་བརྒྱན་པར་བྱ། །ནུབ་བྱང་འཁོར་ལོ་རང་འདྲེན་ཧ། །ཡུལ་འཁོར་གྱི་སྦྲུལ་ཐིག་ལེ་ཡིས། །སྟེང་དུ་བརྒྱན་པར་བྱ་བ་ཡིན། །དེ་ལ་གླང་སྦལ་མཐའ་ཡིས་བརྒྱན། །ནུབ་ཀྱི་ཕྱོགས་ཀྱི་འཁོར་ལོ་ཡིས། །རང་འདྲེན་ཕ་ནི་བླང་བར་བྱ། །ལྷོ་ནུབ་འཁོར་ལོ་རང་འདྲེན་ཌ།
