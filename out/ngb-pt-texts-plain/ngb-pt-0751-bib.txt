རྣལ་འབྱོར་ཉིད་དབྱིངས་ཀློང་ནས་འཆར་མཛད་མ། །སྣང་བའི་གསུང་མཆོག་རྟགས་ཀྱི་ཕྱག་བརྒྱད་བདུན། །ཞིང་ཆེན་ཙིསྟ་བདུན་ནི་ཤར་ར་ར། །འཕྲིན་ལས་མྱུར་མཛད་མ་ལ་ཕྱག་འཚལ་བསྟོད། །ཅེས་བདག་ཉིད་མཆོད་པ་ཡང་དབུལ་བར་མཛད་དོ། །སྔགས་རྒྱུད་ནག་མོའི་གསུང་གི་བསྒྲུབ་པའི་ལེའུ་སྟེ་གཉིས་པའོ།། །།དེ་ནས་ཐུགས་ཀྱི་བསྒྲུབ་པ་འདི་གསུངས་སོ། །་ག་རཀྟ་ཁ་མུན་ཏྲ་མུ་རུ་ལུ་རུ་ལུ་ཧུ༷་སྨྱོ། རག་མོ་ཏེ་ཀེ་ཛ་ཛ་ཁ་ལ་མ་སྨྱོ། ཞེས་མ་མ་་ག་རག་ཤ༷་ཁ་མུན་ཏྲ་སྲོག་ཙམ་རིལ། རག་མོ་སྙིང་ཤ་ཁ་ལ་རག་མོ་ནི་་ག་དུན་སྨྱོ། རག་མོ་ནི་སྙིང་རྩ་ལོང་་ག་མོ་ཤད་ཤད་སྲོག་རྩ་ལ་དུན་སྨྱོ། ཞེས་ཏྲག་བདུན་གཅོད་པའི་ཡེ་གེའི་སྔགས། །ཐུགས་ལས་རྒྱུད་པའི་་ག་བདུན་པོ། །ཡང་དག་གཏེར་ཆེན་དོན་རྣམས་སྨྲ་ཞིང་སྟོན། །དྲག་སྔགས་ཀུན་གྱིས་གཏན་ལ་འབེབས། །རྦད་ཅེས་བྱ་བ་ཐམས་ཅད་ལས་ཀྱི་སྔགས། །ཧུ༷་ནི་ཐམས་ཅད་སྡུད་པའོ། །སྨྱོ་ནི་གཉིས་ཀའི་ལས་ལ་འཛུག །ཅེས་སྔགས་ཐམས་ཅད་རྣམ་པར་དབྱེ་བའི་སྒྲ་རྒྱུན་དུ་གཅོད་པའོ། །སྔགས་རྒྱུད་ནག་མོ་ལས། ཐུགས་ཀྱི་བསྒྲུབ་པའི་ལེའུ་སྟེ་གསུམ་པའོ།། །།དེ་ནས་ཡང་བསྐང་བའི་དམ་ཚིག་གི་ལེའུ་འདི་གསུངས་སོ། །བསྐང་བ་ཆེན་པོའི་དམ་ཚིག་ནི། །དངོས་གྲུབ་འདོད་པའི་རྣལ་འབྱོར་པས། །གཙོར་ནི་དམཚིགབསྲུངབརབྱ། །རྫས་ནི་མཚན་ལྡན་བམ་གྱིས་བསྐང་། །རྨ་བྱའི་རྒྱལ་མཚན་རུ་དར་དང་། 
།འབྲུ་ཡི་གཏོར་མས་་་། །བསྐང་བར་བྱ། །བྱ་ནག་ལུག་ནག་རྟེན་དུ་བྱ། །ཁྱི་དང་བུད་མེད་སྤང་བར་བྱ། །འདོད་ཆགས་བྱ་ལ་རྫས་མི་བསྟན། །སྔགས་ལ་མིང་ཚིག་སེལ་མི་གཞུག །ཐུན་རྫས་མི་གཅད་བསོག་པར་བྱ། །བསྟོད་དེ་་བཞིན་མཆོད་པར་བྱ། ཁྲ་ཆེན་གཏོར་མ་དེ་བཞིན་དབུལ། །དྲག་པོའི་སྔགས་འདི་དབུལ་བར་བྱ། །ཨོ་་ག་རསྒ་ས་མ་ཡ་དུན་སྨྱོ། སྙིང་སྙིང་ཁྲག་བྱ། །བྱ་ནག་ལུག་ནག་རྟེན་དུ་བྱ། །ཁྱི་དང་བུད་མེད་སྤང་བར་། །འདོད་ཆགས་བྱ་ལ་རྫས་མི་བསྟན། །སྔགས་ལ་མིང་ཚིག་སེལ་མི་གཞུག །ཐུན་རྫས་མི་གཅད་བསོག་པར་བྱ། །བསྟོད་དེ་དེ་བཞིན་མཆོད་པར་བྱ། ཁྲ་ཆེན་གཏོར་མ་དེ་བཞིན་དབུལ། །དྲག་པོའི་སྔགས་འདི་བུལ་བར་བྱ། །ཨོ་་ག་རསྒ་ས་མ་ཡ་དུན་སྨྱོ། སྙིང་སྙིང་ཁྲག་ཁྲག་ཤ་ལ་དུན་སྨྱོ། མཏིཏིནྟྲཛ་ལ་ར༷། འབྲས་བུ་སྣ་ཚོགས་གཏོར་མ་འདི། །ཨེ་ཀ་ཙ་ཏིའི་ཞལ་དུ་ཁ་ར༷་ཁུ་ཧི། ཞེས་དེ་བཞིན་གཤེགས་པས་རྫས་མས་དབུལ་བར་བྱ། །ཞེས་གསུངས་སོ། །སྔགས་རྒྱུད་ནག་མོ་ལས། ཐུགས་དམ་བསྐང་བའི་ལེའུ་སྟེ་བཞི་པའོ།། །།དེ་ནས་ཨེ་ཀ་ཛ་ཏི་ནག་མོའི་ཚུལ་དུ་སྐུ་སྔོན་དུ་དབྱུང་བ་འདི་གསུངས་སོ། །དཀྱིལ་འཁོར་་སྙོམས་ཡིད་འོང་ལ། །རྩ་བ་ལྔ་ཡི་ལྡེ་གུས་བྱུག །དབུ་ནག་དམར་པོ་སེར་པོ་བརྩིགས། །དམར་པོ་ཟླ་ཁམ་མུ་ཁྱུད་ཟླུམ། །ཐོད་པའི་ར་བས་བསྐོར་བ་ཡིན། །སྒོ་སྣམ་སྒོ་ཁྱུདདེབཞིནབྲི།
