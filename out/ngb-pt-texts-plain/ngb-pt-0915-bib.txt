ཁྱབ་བདག་ཡེ་ཤེས་རྒྱལ་པོ་ཆེ། །ཆེ་བའི་ཡོན་ཏན་རྒྱས་མཛོད་ལ། །དཀྱིལ་འཁོར་ཆེན་པོ་འབྱུང་བར་འཚལ། །རང་བཞིན་མི་གནས་མཉམ་ཉིད་དང་། །མི་ཆོད་མི་ཤིག་མུན་པའི་མཆོག །དེ་ལས་གྱུར་པ་ཞེ་སྡང་ཆེ། །ཐུགས་ལས་འབྲུ་རྣམས་ཤར་བའོ། །བ༷་ལས་རྒྱ་མཚོ་ཆེན་པོར་གྱུར། །ར༷་ལས་བསྐལ་པའི་མེ་རི་འབར། །ལ༷་ལས་རིན་ཆེན་རི་རབ་འབར། །སྒྱུ༷་ལས་རིན་ཆེན་གཞལ་ཡས་ཁང་། །ཨེ་ནི་བསྐལ་བའི་རླུད་ནག་ཀློང་། །རྭ་ལས་རག་ཏ་དབང་གིས་བསྒོར། ཁ༷་ལས་ལྕགས་ཀྱི་ཁང་རོམ་འབར། །ཕ་ལས་སྦྲུལ་ནག་ཆུན་འཁོར་བརྒྱན། །སྒྱུ༷་ལས་ཡེ་ཤེས་གཞལ་ཡས་ཁང་། ཁྲར་ཕྱོགས་དངུལ་ལ་ལྷོ་ཕྱོགས་གསེར། །ནུབ་ཕྱོགས་ཟངས་ལ་བྱང་ཕྱོགས་ལྕགས། །སྦྲུལ་ནག་ཆེན་པོ་ཀ་གདུང་ལ། །དམངས་རིགས་ཀླུའི་དྲལ་ཕྱམ་གཏང་། །རུས་སྦལ་ཕོ་མོའི་སྒོ་ཐེམ་བྱས། །སྡོམ་དང་སྡིག་པའི་ཟ་ར་ཚགས། །ཆུ་སྲིན་ཉི་མའི་ཁ་བད་ཅན། བསྐལ་པ་སྦྲུལ་ནག་འཁྱིལ་པའི་ཀློང་། །ཨོ་ཕེ་ཤ་ག་དུ་སྐྱུ༷། རིག་པའི་རྒྱལ་པོ་འཆི་བདག་གཤེད། །དགུ་དགུ་ཉིས་བརྒྱ་ཕྱག་དང་ལྡན། །ཞབས་ཀྱིས་སྡོམ་སྡིག་མང་པོ་བརྫིས། །བདག་པོ་བཞི་དང་གཙོ་བོ་བརྒྱད། །རང་བཞིན་ཉིད་ཀྱི་དྲེགས་པའི་གདན། །དེ་ལ་གདུག་པའི་རིགས་བརྩེགས་ཀུན། །ཟ་ལཁབབཞིནའདུསཤིངབགྱེལ། །སེམས་དཔའ་རྡོ་རྗེ་ཞེས་བྱ་བ། །ཐུགས་ཀྱི་རྡོ་རྗེ་ཡིན་ཞེས་བཤད། །དེ་ལས་ཕྲ་མོའི་ཡིག་འབྲུ་འཕྲོ།

ཁ་གནོན་འཁོར་ལོ་སྟོབས་དཔག་མེད། །རང་ལ་མི་ལྡོག་ཐེབས་པའི་གགས། །ཨོ༷་ནག་པོ་ཏི་པ་ཙ་ཡ། ནཱ་ག་རག་ཤ་ཡག་ཤ །ཡུ་ཤན་ཏེ་ཡུ་ནཱ་ག་ཡ་ནི་ཀུ་རུ་ཙ་ཡ། ཡ༷་ཡ༷་མུ་ར་ཡ་ཛུ་ཛུ་རྦད་ཡ་རྦད་ཡ། ལྕགས་རི་མུ་ཁྱུད་ལྟ་བུ་སྟེ། །ལྕགས་ཀྱི་ཁང་རོམ་ལས་ཀྱིས་གཅོད། །ཨ་ཤ༷་ནག་པོ་ཏི་པ་ནཱ་ག་ཀྲོདྷི་ཤྭ་རི་ཏྲི་ཤ༷་རྦད། བུ་གུ་རག་ཤ་རྦད། ཏི་པ་ཙ་པ་ཙ་ཁ་ལ་བ་ཤ་ཏྲུ༷་མ་ར་ན་ཛུ་ཛུ། ཀླུ་ཡི་གཤིན་རྗེ་སྐུ་ཁྲོས་པའི། །སྙིང་པོ་སྲོག་ཏུ་བཤད་པའོ། །ནཱ་ག་དུ་ན་དུ་ན་ཧི་ལི་ཧི་ལི། འདུས་འདུས། ཀླུ་ཡི་འཁོར་རྣམས་སྡུད་པར་བྱེད། ནག་པོ་བམ་ཙི་ཙི་རྒྱས་ཏྲག་ཏུར་སྙིང་ལ་འགུལ་མྱགས་མྱགས་རྩ་བཞི་བསྐུམ། ཤན་ཤན་སྲོག་ལ་ཤན། ཐུར་སོད་སྲུཧྭ། ཡང་སྙིང་ནག་པོ་ཞེས་བྱ་བ། བསད་པའི་སྤུ་གྲི་ཡིན་པར་བཤད། ནག་པོ་ཏི་པ་ཙ་ཡ་ནཱ་ག་རག་ཤ་ཤན་ཤན་ཀར་རྦད་ནཱ་ག་ཕོས་ཕོས། ཨེ་ཨ་ཀར་ཤ་ཡ་ནན། ཨུ་ནན་ཕེ་ལ༷་ཤག་ཤིག་དྲི་མན་མྱ་ར་ཡ་རྦད་ཨ། སྲོག་ཐུན་ནག་པོ་ཞེས་བྱ་བ། །གཤིན་རྗེ་ཡ་མ་རྭ་ཛ་སྟེ། །ཀླུ་ཡི་དུག་མདའ་ནག་པོ་ཡིན། །ཞེས་གསུངས་སོ། །དེ་ནས་ཀླུའི་ཁྱབ་བདག་ཆེན་པོས། ལས་ཀྱི་སྲོག་ཐུན་ཞེས་བྱ་བ་འདི་དག་རྡོ་རྗེ་ལས་ཐོན་ཏོ། །ནག་པོ་ཏི་པ་ཙ་ཡ་ཕེ་རག་ཤ་ཨུ་གནན། ནག་པོ་ཏི་པ་ཙ་ཡ། ཙིསྟ་ཙ་ར། ཙ་པ་ར་རུ་ཤན། ཆེ་གེ་མོ་ལ་ཡ༷། ནག་པོ་ཏི་པ་ཙ་ཡ་ཙུ་ཏ་ར༷་ཙུ་ཏ་རུ་ཀ་ལག་ཤ་ཨ་ཤན། ནག་པོ་ཏི་པ་ཙ་ཡ་ནཱ་ག་ཀྲོ་དྷི་ཤྭ་རི་ཏེ་ཤན་རྦད། ཕུ་ཤན་རྦད། 
ཀུ་རག་༄༅། ཁྲ་རྦད། ཏི་པ་ཙ་ཡ་ཛ་མ་ཛུ་ཁ་ཡན་ཧུ༷་་། །ནག་པོ་ཏི་པ་ཙ་ཡ་ཨུ་ཕུག་ཤ་རྦད། ནག་པོ་ཤ་པ་ཏི་རྦད། དེ་འདྲ་བ་ཏི་རྦད། བ་རུ་ན་ཕེ་ཤན་ནཱ་ག་རྦད། ནཱ་ག་ཕེས་ཡེ་ཨ་ཀྟ་ཤ །ཕེ་ཤན་ཧུ༷་ནན། ཕེ་ལང་ཤག་ཤིག་ནིར་ཡན་མྱ་ར་ཡ་ཛ་རྦད་ཨ། སྲོག་ཐུན་རྒྱལ་པོ་ལྷ་མེན་མཚོན། །སྤུ་གྲི་ཀ་མ་ལ་ཞེས་བྱའོ། །དགྲ་བོ་དྲི་ཛ། དགྲ་བོ་ཁི་ཛ། ནིར་སྨི་རྫི་ཛུ། དགྲ་བོ་བཱ་ར་ཡ་ཛ། འབྱུང་པོ་མྱ་ར་ཡ་ཛ། ཁྲ་ཙ་ར་ཡ། ཛུ་ཧྲི། ཨུར་ཙུ་ད་ར་དུ་ནན། ཛ་ཧྲི་ཨུ་ཙུ་ད་ར་དུ་ནན། ར༷་ཡ༷་ནན། དུ་རུ་ཡ༷་ནན། ཨོ་བཛྲ་ཙན་ནན། ཁྲ༷་ཤིང་ཏེ་ཡ༷། ཙི་ཏ་བྱི་མ་ལ་ཨ་ཀར་ཤ་ཡ། ཨ་ཀར་ཤ་ཡུ། བཛྲ་བྲ་མ་ནི། ཧུ༷་ཛ་མན་ཏྲ་ཧུ༷་ཕཊ། གཤིན་རྗེ་ནག་པོའི་ཞལ་ནས་དབྱུང་། །ཟ་བྱེད་སྨོད་པའི་ཐུན་སྔགས་སོ། །ཨ་ཤན་ནག་པོ་ཏི་པ་ཙ་ཡ། ཀྲོ་དྷི་་།་རི་ཏེ་ཤན་རྦད། ཕུ་ཤུན་ཐུན་རྦད། ཀུ་ར་བ་་།་རྦད། ཏི་པ་ཙ་ཡ་ཛུ་ལ་ཛུ་ལ། དགྲ་བོ་ཙི་ཙི་ཛ་ལ་ཁ་ཡན་ཧུ༷་་།་ཆེ་གེ་མོ་ནཱ་ར་ཡ་ཛུ་ཛུ།ཀླུ་ནག་པོའི་དུག་སྔགས་སོ། །ཨོ་ནག་པོ་ཏི་པ་ཙ་ཡ། ཨུ་ཕུག་ཤ་རྦད། ནག་པོ་ཤ་པ་ཏི། དེ་འདྲ་བ་ཏི་རྦད། བ་རུ་ན་ཕེ་ཤ་ན་ན་ག་དགྲ་བོ་རྦད། ནཱ་ག་ཕོས་ཕོས་ལྭ་ལྭ། ཨེ་ཨ་ཀར་ཤ་ཡ་ཨུ་ནན། ཕེ་ལ༷་ཤག་ཤིག་ནམ་ཡན། དགྲ་བོ་མྱ་ར་ཡ་ཛ་རྦད་ཨ། ཡུམ་ཆེན་ཀླུ་མོ་ནག་མོའི་མྱོགས་པ་ཁྲག་གི་སྔགས་སོ། །ནག་པོ་ཏྲི་རག་ཤ་དགྲ་བོའི་སྙིང་ལ་པྲ་ཤ་ཐུམ་རིལ་མྱ་ར་ཡ་ཛ། དགྲ་བོ་མ་ར་ཡ་ཁུག །བདུད་རོ་ཤ་པྲ་མ་ཡ། 
དགྲ་བོའི་སྙིང་ནག་པོ་པྲ་ཏྲག་ཤ་ཁུག་ཐུམ་རིལ་རྦད། ཁྲག་འཐུང་མུ་ར་ཡ་ཛ། པྲ༷་པྲ༷་ནག་པོ་ཏི་པ་ཙ་ཀྲག་ཤ་རོ་ལ་ཁུག །དགྲ་བོའི་སྙིང་ཙ་རོ་ལ་རྦད། དགྲ་བོ་མྱ་ར་ཡ་ཛ། དེ་ནི་ཀླུའི་སྲོག་གཤེད་ཀྱི་ཕུར་པས་བསད་པའི་སྔགས་སོ། །ནཱ་ག་ཧུ་ཡ་བྷ་ལིང་དུར་བ་ཆེ་གེ་མོ་ཐུན་སོད་ཟུང་ནཱ་ག་ཤིག །རོ་ལ་ཐིབ་ཐིབ། ཀླུའི་སྨོད་པ་ནག་པོའོ། །ནག་པོ་ཏི་པ་ཙ་ཡ། དགྲ་བོའི་སྙིང་ལ་མུ་ར་ཡ་ཛུ། ཀླུ་སྦྲུལ་ནག་པོ་ནཱ་ག་སྙིང་ལ་ཁྲུམ་སུ་ཡེ་ཛ། ནག་པོ་ཏི་པ་ཙ་ཡ། དགྲ་བོའི་སྙིང་ལ་ཡ། རྦད་པུ་ཡེ་ཐུམ་རིལ་ཡེ་ཛ། ནཱ་ག་རྭ་ཛ་དགྲ་བོ་དྲི་རྦད། ཤན་ཏི་རྦད། ཕྲོག་ཏི་རྦད། སྣར་མ་ཏི་རྦད། སྙིང་ལ་རིལ་རིལ་རྦད། ཤ༷་ཤ༷་རྦད་རྦད། མྱ་ར་ཡ་རྦད་རྦད། ཀླུ་བདུན་སྦྲུལ་ནག་པོའི་བསད་པའི་སྤུ་གྲི་ཁོ་ནའོ། །ཨོ་ནག་པོ་ཏི་པ་ཙ་ཡ། ནཱ་ག་རག་ཤ་ལོས་བ། ཀྲི་སྣ། ན་ག་ས་སྟིར་ཝ་བྷ་ཧུ་ཏ། ནཱ་ག་ཏྲིས་སྣ་རག་ཤ་ཏྲི་སྨྱོ། དར་བན་ད་སྟུ་ཧུ་ཏ། ཀ་ལ་སྟིར་སརྦ་རག་ཤ་ཛ་ལ། ཆེ་གེ་མོ་ལ་སྨྱོ་སྨྱོ།ཙིསྟ་ཁུ་ཧི་ཁུ་ཧི། བན་དྷ་ཕཊ། ན་ལ་ཙིསྟ་ཛོ་ལ་ཆེ་གེ་མོ་ལ་ཡ་ཡ་ནི་སྲོག་ཏིགི་རྦད། ཆེ་གེ་མོའི་སྲོག་ཆོད་ཙིསྟ་ལོང་། །ཡན་ལག་ཆིངས་ཐུམ་རི་ལི་ལི། ནག་པོ་སྲོག་ལ་སྨྱོ། དགྲ་བོའི་ཤ་ལ་ཡ། ཁྲག་ལ་ཡུ། ས་མ་ཡ་རག་ཤ་ཐ༷་སྨྱོ། མ་ར་ཡ་ཕཊ། ས་བདག་ཀླུ་གཉན་མང་པོ་ཡིས་ལས་ཀྱི་ཐུན་ནག་དབབ་པའི་ལས། །གསང་བ་མཛེ་ཟོར་ནག་པོ་ཡིན། །ཨོ་ནག་པོ་ཏི་པ་ཙ་ཡ། ནཱ་ག་རག་ཤ །ཡག་ཤ་ཁ་ཡན་ཧུ༷། ཡ་ཧུ༷་ཡ། 
སྤོ་ཛ་དྲི་ཛ།༄༅། །ལ༷་ཆེ་གེ་མོ་མྱ་ར་ཙཀྲ་ཐུམས་ཐུམས་ལ༷་རྦད་རྦད། རྦད་ཡ་རྦད་ཡ། ཁམ་ཆེ་གེ་མོ་མྱ་ར་ཡ་ཙཀྲ་ཐུམས་ཐུམས་ཁམ་རྦད་རྦད། རྦད་ཡ་རྦད་ཡ། ར༷་ཆེ་གེ་མོ་མུ་ར་ཡ་ཙཀྲ་ཐུམས་ཐུམས་ར༷་རྦད་རྦད། རྦད་ཡ་རྦད་ཡ། ཡ༷་ཆེ་གེ་མོ་མྱ་ར་ཡ་ཙཀྲ་ཐུམས་ཐུམས་ཡ༷་རྦད་རྦད། རྦད་ཡ་རྦད་ཡ། འཁོར་ལོ་ནག་པོའི་གནད་གཅོད་པའོ། །དྲི་ཛ་དྲི་ཛ། ལ༷་ཁ༷་ར༷་ཡེ་ཆེ་གེ་མོ་སྨྱུ་ར་ཡ་ཙཀྲ་ཐུམས་ཐུམས་ལ༷་ར༷་ཡ༷་རྦད་རྦད། རྦད་ཡ་རྦད་ཡ།གནད་གཅོད་སྤུ་གྲིརཁྱབ་བདག་བཤད། །ནཱ་ག་ཤན་ཏི་དུ༷་ཏི་སིབས། ནཱ་ག་རྭ་ཛ་ནན་ནཱ་ག་མི་ར་རྦད། ཤན་ཏི་ཚ་རུ་ཚུབས། ནག་པོ་ཨོ་ར་ལི་གཤམ། ན་ག་དུན་དུན། གྱེན་གྱེན། ནཱ་ག་དུ་ཡ་ཡ། ཙུབས་ཙུབས་ཤ །ཆེ་གེ་མོ་ལ༷་བ༷་ར༷་ཡ༷་མུ་ར་ཡ་རྦད། ལ༷ བ༷་ར༷་ཡེ་ཆེགེ་མོ་རྦད་ཡ་རྦད་ཡ། ཡ༷་ར༷་བ༷་ལ༷་ཆེ་གེ་མོ་རྦུད་རྦུད། འབྱུང་བ་སྙིང་ཏིབ་སྤྱིར་རྩུགས་ཀྱིས། །ཕུང་གཤིག་དྲག་པོ་མནན་གཅོད་ཡིན། །ཨ་ཤ༷་ནག་པོ་ཏི་པ་ཙ། ནཱ་ག་ཀྲོ་ཏི་ཤྭ་རི་ཏྲི། ཤ༷་རྦད། ཆེ་གེ་མོ་ལ༷་ཁ༷་ར༷་ཡ༷་ནག་པོ་བ༷་ཙི་ཙི་རྒྱས། ཏྲག་ཏུར་སྙིང་འགུལ་མྱགས་མྱགས་ལ༷་ཁ༷་ར༷་ཡ༷་ཆེ་གེ་མོ་རྩ་བཞི་བསྐུམ་ཤན་ཤན། ཡ༷་ར༷་ཁ༷་ལ༷་སྲོགས་ཤན་ཐུར་སོད་རྦད་ཡ་རྦད་ཡ། འབྱུང་བ་གནད་ཀྱི་ལྟོ་དཀྲུག་པ། །ལས་ཀྱི་སྦལ་ནག་ལྷོ་སྦྱོར་ཡིན། །ལ༷་ཆེ་གེ་མོ་སྨྱུ་ར་ཡ་ཐུན་སྨྱོ། ལ༷་ཆེ་གེ་མོ་བཱ་ར་ཡ་ཐུན་སྨྱོ། ལ༷་ཆེ་གེ་མོ་མྱ་ར་ཡ་ཐུན་སྨྱོ། ལ༷་རྦད་ཐུམ་རི་ལི་ལི་སྨྱོ། ཁ༷་ཆེ་གེ་མོ་མྱ་ར་ཡ་ཐུན་སྨྱོ། 
ཁ་རྦད་ཐུ༷་རི་ལི་ལི་སྨྱོ། ར༷་ཆེ་གེ་མོ་མུ་ར་ཡ་ཐུན་སྨྱོ། ར༷་རྦད་ཐུ༷་རིལ་ལི་ལི་སྨྱོ། ཡ༷་ཆེ་གེ་མོ་མྱ་ར་ཡ་ཐུན་སྨྱོ། ཡ༷་རྦད་ཐུ༷་རི་ལི་ལི་སྨྱོ། ཡ༷་ར༷་ཁ༷་ལ༷་ཆེ་གེ་མོ་སྨྱུ་ར་ཡ་རྦད་ཡ། ལ༷་ཁ༷་རེ་ཡ༷་བཱ་ར་ཡ་ཐུན་སྨྱོ། གནམ་འདྲེ་སྤྱི་རྩའི་བསད་པ་ཡིས། །འབྱུང་བ་ལས་ཀྱི་གནད་གཅོད་ཐུན། །ཅེས་གསུངས་སོ། །དེ་ནས་ཡང་ཀླུའི་ཁྱབ་བད༷ག་ཆེན་པོས། དྲེགས་པའི་བསད་ཁྲམས་འབར་བས་གནད་བསྐུལ་བ་འདི་དག་གསུངས་སོ། །ཡ་ནཱ་རྭ་ཛ་ནག་པོ་དགྲ་སྙིང་ལ་ཡ་ཡ། ཡ་མ་ཀྲི་སྣང་ཛ་ཐུར་དགྲ་ལ་སྨྱོ། ནཱ་ག་ཀེན་ཙེ་འདུས་འདུས་ཛ་ཐུར་དགྲ་ལ་སྨྱོ། ཨོ།་བེ་ཤེ་ཧེ་ཐུམ་ཧེ་ཐུམ་ཛ་ཐུར་དགྲ་ལ་སྨྱོ། ཀ་ལ་མུན་ཏྲི་དེ་ཆོད་དགྲ་ལ་ཡ། ཛ་ཐུར་དགྲ་ལ་སྨྱོ། ཛ་ཐུར་སྙིང་ལ་ཡ༷། བཛྲ་ས་དུ་སྙིང་ཐུར་དགྲ་ལ་ཡ༷་སྨྱོ། ཛ་ཐུར་སྙིང་ལ་ཡ༷་མ་སྙིང་ཐུར་དགྲ་ལ་སྨྱོ། དྲེགས་པ་ལས་ཀྱི་འཁོར་ལོ་སྟེ། །གསད་པ་སྤུ་གྲི་ཁོ་ནའོ། །ཀླུའི་འཁོར་ལོ་ནག་པོའི་རྒྱུད་ལས། ལེའུ་སྟེ་རྟོག་པ་གསུམ་པའོ།། །།དེ་ནས་སྤྲུལ་པའི་ཁྲོ་བོས་ཁྱབ་བདག་ཡེ་ཤེས་ལ་བསྟན་པའི་དགྲ་བོ་འདུལ་བའི་ཕྱིར། བསྒྲུབ་པའི་རིམ་པ་བཤད་པར་འཚལ། །ཅེས་པས། ཀླུའི་བདག་པོས་དེ་ཉིད་བསྒྲུབ་པའི་རིམ་པ་བཤད། །ལུང་གསུམ་མདོའམ་བྲག་ཆུ་འཐབ། །བྱང་དང་བྱང་ཤར་གནས་མཆོག་སྟེ། །ཀླུ་སྲིན་གདུག་པ་གནས་པར་བསྒྲུབ། །དབྱར་གྱི་ཟླ་གསུམ་ཁྱད་པར་དུས། །རྩི་ཐོག་རྒྱས་སམ་འབྲས་བུ་སྨིན། །བསྒྲུབ་པའི་དུས་དང་ཟླ་བའོ།
