1་་་་་་་ཇོ་་་་་ཁྱེ་སྟི་་་་་་་་་1 །་ཁྲི་་་་་་་་་་་ཛི་་་་་་ཛི། །་ཉུ་་་ཁུ་སྟུ་་་་་་ཁུ་རྩོ་་། ་བྱུ་་་་་ཁུ་་་་་༦་་་་་་་་་1 1ཚོ་་་་་་་་ཚོ་ཁུ་་་་་་་་1 1་་་་་་་ཁུ་་་་ཁྱི་་་་་་་་་1 1་་་་་་་་་་་་་་་ཁྱི་་་་ཁྲོ་། །་༦་་་་་ཉུ་་་་་ ་ཁྱི་་་་༷་ ་་། ་་་་་་༷་ ་་རྨྱུ་་་་ཁྱི་་་སྐྱུ་་་སྐོ1 1ཆི་་་་་་་་་་་་་་་་་་་གོ 1ཁུ་་་་་་ཁུ་་་་་་་སྐོ་ཁུ། 1་ཁྱེ་་་་་་་་་ཉུ་་མི་་་་་་་་་1 །་ཉུ་་ཙྭ་་་་་་་་་་་་་་ཟློ་།་་ཅུ་་་་་་་་ཁུ་་ཁུ་་ཉུ་་1 1ཚོ་་་་་་་་྅་་སྩོ་་་་་ཇོ་1 1་་་་་་་་བྱོ་་་་་་་་་་་4་1 1་་་་་་་་་་་་་་་་་་་་་་་་1 1་་་་་་་་་་་་་་ཁུཁུ་་ཁྲི་་1 1་་་་་་་་་་་་ཁྱི་་་1 །་་་ཁྱི་་་་་་་་་་་་་ཟི་་་་1 1་་་་་་་་སྤྱོ་་་་་ཁྱི་་ཉུ་ཕྱུ་1 1་ཁྱི་་ཁྱི་་་་་་་་་་་་་་་ཁྱོ་1 1་་་་་་་ཇོཁྱེ་་་་་་་་ཁྱི་་་་1 །་་་་་་་་་་་་ཁུ་་་་་། 1་་་་སྐྱུ་་་་་ཁྱུ་་ཟི་་་་་་་1 ་་་་་་་་་་་་་་་་་ཁྱི 1་་་་་་་་་ཁྱེ་་་་ཉུ་་་ཚོ་་སྦྲི་ཟློཁྲི་སྦྱི་་་ལྗོ་་ཟོ་་་་་་་སྤྲོ1 1་ཁྲི་་་་སྐྱི་་་་ལྗོསྟོ་་པྱོ་་་་་་ལྗོ་སྣྲོ1 1་་་་ཁྲི་་་ལྗོ་་་སྐྱི་་་་ཚོ་་ཁྱི་་ སྐྱི་་་་་སྐྱི1 །་་སྟི་་་་1 
1་་་་ཁྱི་ཛི་་ཟི་་་་་་1 1་་་ཁྱི་་་་་་ཁྲུ་་ཉུ་་་ཁུ་་་་་ཁྱེ་་་་་་་་་་་་ཛི1 14་་ཁྱི་་་་་་་་་ཁྱེ་་་ཟོ་་་་1 ཁུ་་་་་་་ཚི་་་་་་་་་1 1་ཁུ་་ཉུ་་་་་་་་ཁྱི་་་་་་་། ་། ་་་ཆོ་་་་་་་་་ཁུ་་་་། 1་་་་་་་་་་་་་་་་་་1་1་་བྱུ་་་་་་་་་་་་ཁྱི་ཁྱི1 1་་་་་་་་་་་་་་་་་་་་1 1་་་་ཁྱི་་་་་་་ཟློ་་་་་་1 1་་་ལྗུརྭ་ ཇོ་་་་ཐི་་་ཆོ་ 1ཁུཉུ་4་་་་་་་ཟློ་ཚོཇོ་་་་་་་་1 །་་༦་་་་་་་་་་་་་ཁྱི་་་་སྤྱོ་ 1 1་་་་ཕྱུ་ ཁྱི་་་་་་་་བློ་ཁྱེ་་1་།་སྒྲི་་་་་་་ཁྱི་་་་་་་་་ཉུ་1 1་་་4་་་་སྟུ་་་་་4་་ཁུ་་་། །་་་་རྨྱུ་ ་་་་་་་་་་སྨྱོ1 1སྟུཉུ་་་་་་་་་་་་་་ཁྱི་་་་་་་་་1 1་་་་ ་་་༔་་སྟི་་སྟུ་ཁྱོ་་1 1་་་་་་་་་་་་བྱུ་ ཁུ་་་་་་་། 1་་༷་་་སྟུ་་་་་་་་་་་་་་་་་་་། ་བྱུ་་་་་་༷་་་་་་་ཟི་་་ཁུ1 1་་་་་སྦྱི་་་་་ཁྱི་་ཁུ་་་་གློ་1 །་་་་་་རྐྱུཁྱི་་་་ཁུ༷་ ་་་་་་1 1་་་་་་ཁྱེ་་་་་སྒྲོསྨྱུ་་་་ཟློ་1 1་་་་་་་་་་ཁུ་་་་་་་་1 །་་་་་་་་་་་་་་་་་བྱུ་སྦྱོཁྱེ1 1་་་་་་་་༦་་་་ཚོ་་་་1 1་་་་་་་་ཏོ་་་་་་་ཁྲུ་ཁུ1 1་ཞུ་་སྐྱུ་ཉུ་་ཁྱི་་་ཛི། 1་་་་ ་་་་སྐྱི་་་་གྱི་1 ་1་་་་་ཁྱི་་ཟི་་་ཟོ་་་་་་ཛི་་་་༦། །་་་ཙྭ་ ་ཁུ་་་་
