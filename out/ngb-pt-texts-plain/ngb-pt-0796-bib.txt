ཕྲད་པར་ཤོག། །། ༄༅། །རྒྱ་གར་སྐད་དུ། ཀྲོ་དྷ་རྭ་ཛ་ཤི་བཛྲ་བཱ་ནི་ཙཎྜ་མ་ཧྭ་རོ་ཥ་ཎ་ཨ་སྟི་ཁི་྅་ཨུ་པ་དེ་ཤ་གུ་རྒྱ་ནཱ་མ་ཏན་ཏྲ། བོད་སྐད་དུ། ཁྲོ་བོའི་རྒྱལ་པོ་ཕྱག་ན་རྡོ་རྗེ་གཏུམ་པོ་འཇིགས་སུ་རུང་བའི་དབང་བསྐུར་བ་མན་ངག་གསང་བའི་རྒྱུད་ཅེས་བྱ་བ། བཅོམ་ལྡན་འདས་དཔལ་རྡོ་རྗེ་གཏུམ་པོ་ལ་ཕྱག་འཚལ་ལོ། །འདི་སྐད་བདག་གི་ཐོས་པ་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་གསང་བའི་བདག་པོ་ཕྱག་ན་རྡོ་རྗེ་ལ། དེ་བཞིན་གཤེགས་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་གདྷ་ཀླུང་བྱེ་བ་ཕྲག་འབུམ་གྱི་སངས་རྒྱས་རྣམས་ཀྱིས་དབང་བསྐུར་ཏེ། དེ་བཞིན་གཤེགས་པ་ཤཱཀྱ་ཐུབ་པ་དང་། རྒྱལ་བ་རྣམ་པར་སྣང་མཛད་དང་། སངས་རྒྱས་མི་བསྐྱོད་པ་རྣམས་ཀྱིས་དབང་བསྐུར་བས། ཁམས་གསུམ་ཆོས་ཀྱི་རྒྱལ་པོར་གྱུར་ནས། འོག་མིན་གྱི་གནས་ལྕང་ལོ་ཅན་གྱི་ཕོ་བྲང་ན། འཁོར་སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་དང་། བཀྲ་ཤིས་ལྷ་མོ་བརྒྱད་དང་། ཁྲོ་བོ་བདུད་རྩི་འཁྱིལ་པ་ལ་སོགས་པ་ཁྲོ་བོའི་ཚོགས་དང་། གནོད་སྦྱིན་ཆེན་པོ་རྒྱལ་པོ་རྣམ་ཐོས་སྲས་དང་། 
ཛམ་བྷ་ལ་སོགས་པ་རྣམས་དང་ཐབས་ཅིག་ཏུ་བཞུགས་༄༅། །སོ། །རྡོ་རྗེ་གཏུམ་པོ་དབང་བསྐུར་བ་མན་ངག་གསང་བའི་རྒྱུད་ལས། གླེང་གཞིའི་ལེའུ་སྟེ་དང་པོའོ།། །།དེ་ནས་བཅོམ་ལྡན་འདས་དཔལ་རྡོ་རྗེ་གཏུམ་པོས་དབང་བཞི་བྱིན་གྱིས་རློབ་པ་ཞེས་བྱ་བའི་ཏིང་ངེ་འཛིན་ལ་སྙོམ་པར་ཞུགས་སོ། །དེ་ནས་འཁོར་གྱི་ནང་ནས་ཁྲོ་བོ་བདུད་རྩི་འཁྱིལ་པ་སྟན་ལས་ལངས་ཏེ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་རྡོ་རྗེ་གཏུམ་པོ་ཉིད། །གཟི་བྱིན་འབར་བའི་རྡོ་རྗེ་ཅན། །ཕྱི་རབས་དམ་ཚིག་ཅན་དོན་དུ། །ཉིད་ཀྱི་བསྒྲུབ་པའི་རིམ་པ་དང་། དབང་བསྐུར་དམ་ཚིག་རབ་གནས་དང་། །ལ་སོགས་རིམ་པ་བཤད་དུ་གསོལ། །ཞེས་གསོལ་བ་དང་། །བཅོམ་ལྡན་འདས་རྡོ་རྗེ་གཏུམ་པོ་ཏིང་ངེ་འཛིན་ལས་བཞེངས་ཏེ། བདུད་རྩི་འཁྱིལ་པ་ལ་བཀའ་སྩལ་པ། །བདུད་རྩི་འཁྱིལ་པས་དོན་ཞུས་པ། ང་ཡིས་བཤད་ཀྱིས་ལེགས་པར་ཉོན། །རྡོ་རྗེ་ག་་པོ་བསྒྲུབ་འདོད་པས། ཁེན་ཏུ་རྒྱས་པ་ལྷ་མོས་བསྐོར་བ་དང་། །དེ་ལས་བསྡུས་པས་ག་་པོ་བཞིས་བསྐོར་དང་། །བསྡུས་པ་རྒྱན་གཅིག་དཔའ་བའི་ཚུལ། ཤེན་ཏུ་བསྡུས་པ་བཟླས་ལུང་དང་། །རྣམ་པ་བཞི་རུ་ཤེས་པར་བྱ། །དེ་ལ་རྒྱས་པ་བསྒྲུབ་པ་ནི། །རྣལ་འབྱོར་དམ་ཚིག་ལྡན་པ་ཡིས། ཁེན་ཏུ་དབེན་པ་ཉམས་དགའི་གནས། །ཡིད་དང་འཐད་པ་བསྒྲུབ་པར་བྱ། །སྟེགས་བུ་མདའ་གང་གྲུ་བཞི་ལ། །དྲི་སྤོས་བདུད་རྩིས་ཆག་ཆག་གདབ། །ཐིག་གདབ་ཚོན་གྱིས་བྲི་བ་ནི།

།ཨོ་ཙཎྜམ་ཧྭ་རོ་ཥ་ཎཤརྙིང་ཀུ་རུ་ཧུ༷་ཕཊ།བྲེརྙུ་ཀུ་སྲ པ་ཤང་ཀུ་སྲ མྱ་ར་ཡ་ར་ཀྟ་ཀུ་སྲ ཨོ་བྲུ༷་བཛྲ་ཨུ་ཡུཤེ་ཧུ༷་ཨ། མ་མ་སྤྲུརྙུ་ཀུ་རུ་ཧུ༷་གྲྭ་ཧྭ། མ་ཧྭ་སིདྷི་ཧུ༷། ཨོ་ནཱ་ག་ཀ་ལ་ཧུ༷། ནཱ་ག་ལེ་དུ༷་པ་ལེ་ལེ། ནཱ་ག་རྭ་ཛ་ཧུ༷། ཨོ་ལྷི་ལི་ལྷི་ལ་འབར་ཤ་འབར་ཤ་པ་ཏ་ཡ་པ་ཏ་ཡ་ཧུ༷་ཧུ༷་ཧུ༷་ཧོ་ཧོ་ཧོ་ཕཊ་ཕཊ་ཕཊ་གྲྭ་ཧྭ། སེར་བཨུ་ན་མ་ནི་ཕཊ་གྲྭ་ཧྭ། གཏུམ་པོ་དབང་བསྐུར་བ་མན་ངག་གསང་བའི་རྒྱུད་ལས། ལས་ཁ་ཚར་གྱི་ལེའུ་སྟེ་བརྒྱད་པའོ།། །།དེ་ནས་དཔལ་རྡོ་རྗེ་ག་་པོ་ལ། །འཁོར་རྣམས་དགེས་ནས་བསྟོད་པ་ནི། ཨེ་མ་ཧོ་རྡོ་རྗེ་གཏུམ་པོ་ཡིས། །གསང་རྒྱས་ཀུན་གྱིས་དབང་བསྐྲུར་བས། །སངས་རྒྱས་ཀུན་པས་སྔར་སངས་རྒྱས། །ཁྲོ་རྒྱལ་དེ་ལ་ཕྱག་འཚལ་བསྟོད། །ཅེས་པས་བསྟོད་ནས། །མཆོད་པས་མཉེས་པར་བྱས་པར་གྱུར་ཏོ། །གཏུམ་པོ་གསང་དབང་བསྐུར་བ་མན་ངག་གི་རྒྱུད་ལས། ཁྲོ་བོ་རྣམས་ཀྱིས་བསྟོད་པའི་ལེའུ་སྟེ་དགུ་པའོ།། །།དེ་ནས་ཁྲོ་བོ་བདུད་རྩི་འཁྱིལ་པས། བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་རྡོ་རྗེ་གཏུམ་ཉིད། བདག་གིས་རྒྱུད་འདི་ཇི་ལྟར་བསྟན་ཅིང་གཟུང་བར་བགྱི། ཞེས་གསོལ་པས། རྡོ་རྗེ་ག་་པོས་བཀའ་སྩལ་པ། རྒྱུད་འདི་རྣལ་འབྱོར་པ་དམ་ལ་གནས་པ། བློ་སྣ་ཉུང་བ་བསྟན་པ་ཉམས་སུ་ལེན་པ། རིན་པོ་ཆེ་ལྟ་བུ་ལ་བསྟན་ཅིང་ཤིང་ཏུ་གཅེས་པར་གཟུང་ཞིང་བསྟན་པར་བྱའོ།
