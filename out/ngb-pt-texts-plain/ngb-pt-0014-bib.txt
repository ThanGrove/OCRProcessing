སྔོན་བྱུང་གི་རྒྱུ་རྐྱེན་གཏམ་རྒྱུད་ཀྱི་།ལེའུ་སྟེ་གཉིས་པའོ།། །།དེ་ནས་རྡོ་རྗེ་སེམས་དཔའ་དང་། །ལག་ན་རྡོ་རྗེ་སྟོབས་ཆེན་གྱིས། །སྐད་མཐུན་ཚིག་སྒོ་གཅིག་པ་ཡིས། །རྒྱལ་བའི་ཚོགས་ལ་འདི་སྐད་བརྗོད། །དེ་ལྟར་འདུལ་བའི་དུས་ཐབས་མཆོག་གང་གིས་དྲང་བར་བྱ། །སླར་གསོལ་པ་དེ་བསྟན་དུ་གསོལ། །ཞེས་ཞུས་པ་ལས། ཀུན་བཟང་རྡོ་རྗེ་རྣམ་སྣང་གིས། །སྤྲུལ་པ་ཉིད་ལ་བཀའ་སྩལ་པ། །ལེགས་སོ་སེམས་དཔའ་རྡོ་རྗེ་།མཁྱེན་པར་གྱུར་ཀྱང་ཞུས་པ་ལེགས། །རིག་པ་འཛིན་པ་ རིགས་ཀྱི་བུ། །གསང་སྔགས་དམ་ཚིག་ཉམས་པ་ལ། །ན་རག་དོང་ནས་སྤྲུག་པ་ཡི། །འགྱོད་ཚངས་བཤགས་པའི་རྒྱལ་པོ་སྒྲོགས། །ཆོས་ཉིད་དོན་་མ་ཐེག །ལུས་ངག་ཡིད་ཀྱི་བརྩོན་འགྲུས་ཕྱུང་། །སྐུ་གསུང་ཐུགས་ཀྱི་ཚངས་པ་ཐོབ། །ངན་སོང་དོང་སྤྲུག་སྐུ་གསུམ་མཐོང་། །ཕྱི་ནང་གསང་བའི་དབང་བསྐུར་གྱིས། །རྒྱུད་དྲུག་གནས་ནས་ངེས་པར་།ཡེ་ཤེས་ལྔ་ལྡན་ཆོས་སྐུ་ཐོབ། །ཡུལ་མེད་མིང་གི་སྒྲོ་འདོགས་ཆོད། །གསང་སྔགས་བླ་མེད་རྡོ་རྗེ་ལ། །ཐབས་མཆོག་བླ་ན་མེད་པ་བཞུགས། །ཉན་ཐོས་རང་རྒྱལ་ལྟ་བུ་མིན། །དཔེར་ན་ཉན་ཐོས་འདུལ་ནི། །རྫ་གསར་སོ་མ་བཏང་པ་འདྲ། །ཆག་པར་གྱུར་ན་གསོ་ཐབས་མེད། །གསང་སྔགས་རྡོ་རྗེ་ཐེག་པ་འདི། །རིན་ཆེན་གསེར་འདྲ་རྒྱུ་མཆོག་བཟང་། །གསེར་དངུལ་སྣོད་འདྲ་ཤོམས་ཡང་བཅོས།

མ་ནུ་བཱ་ལ་ཡ།བཛྲ་སཏྭ། ཏེ་ནོ་པ་ཏིམོ དྲི་རྙོ་མེ་བྷ་བ། སུ་ཏོཧྲོ་མེ་བྷ་བ། ཨ་ནུ་རལྟོ་མེ་བྷ་བ། སུ་པོསྩོ་མེ་བྷ་བ། སརྦ་སིདྷི་སྨེ་པྲ་ཡསྒྲོ སརྦ་ཀརྨ་སུ་ཙ་མེ། ཙིརྟི༷་ཤེ་ཡ་ཀུ་རུ་ཧུ༷། ཧ་ཧ་ཧ་ཧ་ཧོ། བྷ་ག་བཱན། སརྦ་ཏ་ཐུ་ག་ཏ་བཛྲ་མུ་མེ་མུ྅་བརྡོ་བྷ་བ། མ་ཧྭ་ས་མ་ཡ་ས་ཏྭ་ཨ། འདི་ནི་དུས་ལ་བབ་པའི་ཚེ། །གསུམ་མམ་བདུན་ནམ་བཅུ་བཞིའམ། །ཉི་ཤུ་རྩ་གཅིག་བརྒྱ་རྩ་བརྒྱད། །ལྷ་བསྒོམ་གསལ་བས་བརྗོད་བྱས་ན། །ཆག་ཉམས་ཐམས་ཅད་བསྐང་བར་བྱེད། །ངན་སོང་གསུམ་ལས་ཐར་བར་འགྱུར། །ཡིད་དམ་དག་ཏུ་འདི་བརྗོད་ན། །དུས་གསུམ་རྒྱལ་བའི་སྲས་སུ་འགྱུར། །ཞེས་གསུངས་པས། ལས་ངན་སེམས་ཅན་དམྱལ་བ་རྣམས་ཀྱིས། །མགོན་པོ་སེམས་དཔའ་ཨ་ལ་ལ་ཧོ། ཞེས་སྨྲས་སོ། །དེ་ནས་དུས་གསུམ་གྱི་སངས་རྒྱས་རྣམས་ཀྱིས། རྡོ་རྗེ་སེམས་དཔའ་ལ། ཐུགས་ཀྱི་ཡང་སྙིང་བརྗོད་པ་ཁྱོད། ཤིན་ཏུ་ལེགས་སོ། །བཤགས་པའི་རྒྱལ་པོ་བརྗོད་པ་ལེགས་སོ་ལེགས་སོ། །ཞེས་གནང་བ་བྱིན་ནོ། །དབང་མཆོག་རྒྱལ་པོའི་རྒྱུད་ལས། ཡང་སྙིང་གསང་སྔགས་བཤགས་པའི་ལེའུ་སྟེ་བརྒྱད་པའོ།། །།དེ་ནས་ཕྱག་ན་རྡོ་རྗེ་ཡིས། །ལས་ངན་སེམས་ཅན་དེ་ལ་ནི། །སྔགས་ཀྱི་སྣོད་དུ་རུང་བྱའི་ཕྱིར། །དང་པོ་སེམས་བསྐྱེད་དད་བྱས་ནས། །བུམ་པའི་ལྷ་ཡི་དབང་བསྐུར་དང་། །ན་རཀ་གནས་ནས་དྲང་བ་ཡི། །གསང་བ་སྔགས་ཀྱི་དབང་བསྐུར་རོ།
