་1་་་་་་་་་་་་་་་་་ཁུ་་་་་་་1 1་་་་་་་་་་་སྦྱི་་་སྐྱི་་་སྟུ་་་་་ཁྱེ། 1་་་་་གྱི་་ཉུ་་་་་་་་་་། 1་་་་་་་་་ཁྱི་་་་་་་་་་་1 1་ཁྱི་་་་་་་ཁྱི་་་་་་་་་་་་་བྱོ། 1་་་་་་་་་སྟུ་་་་་་་་་་་1 །་་་་་་་་་་་ཁྱེ་་་་་་གོ 1་་་་་་་་་་་་་ཁྱི་་་་་་། 1་་་་་་་སྣོ་་་་་་་་སྤྲེ་་་ཟློ1 1་་་་་ཁྱི་་ཁྱི་་་་་་་ ་་། 1་ཁུ་་་་་༷་་་་ཁྱི་་ཁུ་་ཁུ1 ་1་་་་་་4་་་་་་་་་་་་་་་ཁྱི་1 1་་་་བྱུ་ ་་་་་ཁྱེ་་་་་ཁྱེ་་་ཁུ་ཁྱེ་་1 1་་ཟི་་་་་་་་་་་་་་་་་་1 1་་་་་་་་་་གོ་་་ཇོ་་་་་་་1 །་་སྡུ་་ཁྱི་་་ཛི་་་་་་་་་་1་1་་་་་་་་་་་རྙུ་་་་་་་་1 14་་་་་་་་་་ཁྱི་་ཁུ་ཁྱེཁུ1 1་་་་་་ཁྱི་་་་་་་་་་་་་1་།ཁྱི་་་་ཁྱེརྨྱུ་་་ཁྱི་་་་་་་་་་་1 1་་་་་་སྩོ་་ཇོ་་་་ཁྱི་་ཚི་་་་1 1་་་་ཁྱེ་་་་་་་་སྨྱོ་་ཁྱི་་ཟློ་།གོ་་་་་་་་་་་་་་་་་་་་། 1་་་་བྱོ་ཁྱི་་་་་་་་་་་་། །་་་་་་བྱོ་་་་་་་་་ཁྲི་་་་་་1་1་་་སློ་་་་་་་་་་་་་ཚོ་་་་1 1་་་་་་སྤྱོ་་་་་སྟུ་་སྦྲེ་་་སྤྲི1་1་་་་་་་་་་་་སྟུཁྱེ་་སྨོ་ཁྱིདྲི་་་་སྤྱ་1 1་་་་་་་་་་་་་་གོ་སྤྲི་་་་1 1་ཐུ་་་ཁྱིཇོ་་་་་་རྩི་་ཛི། 1༷་་་་་་་་་་་་ཁྲི་་་་ཁུ་།

1་་་་ཁྱི་ཛི་་ཟི་་་་་་1 1་་་ཁྱི་་་་་་ཁྲུ་་ཉུ་་་ཁུ་་་་་ཁྱེ་་་་་་་་་་་་ཛི1 14་་ཁྱི་་་་་་་་་ཁྱེ་་་ཟོ་་་་1 ཁུ་་་་་་་ཚི་་་་་་་་་1 1་ཁུ་་ཉུ་་་་་་་་ཁྱི་་་་་་་། ་། ་་་ཆོ་་་་་་་་་ཁུ་་་་། 1་་་་་་་་་་་་་་་་་་1་1་་བྱུ་་་་་་་་་་་་ཁྱི་ཁྱི1 1་་་་་་་་་་་་་་་་་་་་1 1་་་་ཁྱི་་་་་་་ཟློ་་་་་་1 1་་་ལྗུརྭ་ ཇོ་་་་ཐི་་་ཆོ་ 1ཁུཉུ་4་་་་་་་ཟློ་ཚོཇོ་་་་་་་་1 །་་༦་་་་་་་་་་་་་ཁྱི་་་་སྤྱོ་ 1 1་་་་ཕྱུ་ ཁྱི་་་་་་་་བློ་ཁྱེ་་1་།་སྒྲི་་་་་་་ཁྱི་་་་་་་་་ཉུ་1 1་་་4་་་་སྟུ་་་་་4་་ཁུ་་་། །་་་་རྨྱུ་ ་་་་་་་་་་སྨྱོ1 1སྟུཉུ་་་་་་་་་་་་་་ཁྱི་་་་་་་་་1 1་་་་ ་་་༔་་སྟི་་སྟུ་ཁྱོ་་1 1་་་་་་་་་་་་བྱུ་ ཁུ་་་་་་་། 1་་༷་་་སྟུ་་་་་་་་་་་་་་་་་་་། ་བྱུ་་་་་་༷་་་་་་་ཟི་་་ཁུ1 1་་་་་སྦྱི་་་་་ཁྱི་་ཁུ་་་་གློ་1 །་་་་་་རྐྱུཁྱི་་་་ཁུ༷་ ་་་་་་1 1་་་་་་ཁྱེ་་་་་སྒྲོསྨྱུ་་་་ཟློ་1 1་་་་་་་་་་ཁུ་་་་་་་་1 །་་་་་་་་་་་་་་་་་བྱུ་སྦྱོཁྱེ1 1་་་་་་་་༦་་་་ཚོ་་་་1 1་་་་་་་་ཏོ་་་་་་་ཁྲུ་ཁུ1 1་ཞུ་་སྐྱུ་ཉུ་་ཁྱི་་་ཛི། 1་་་་ ་་་་སྐྱི་་་་གྱི་1 ་1་་་་་ཁྱི་་ཟི་་་ཟོ་་་་་་ཛི་་་་༦། །་་་ཙྭ་ 
་ཁུ་་་་་་་1 1་་་་་་ཁུ་་་་་སྐྲུ་་་སྟུ་་། 1་ཚི་ཉུ་་་་་སྐྱི་་་་་་་ཁྱི་་་་ཚི་་་་་་་་ཁྱི1 །་་་་་ཐུ་་་་་་་་་་་་་་་་་ཚི་་་་་་་་་1 1་་ཁྱི་ ་་ཆི་་་་་་་་་་་་་་་་་་་་་་་་་སྟུ1 1ཙྭ་སྤྱོ་་་་་་་ཉི་ ་་་་1 །་བྱུ་རྙུ་ ་ཆི་་་་སྐྲུ་་་་་་། 1་ཁྱེ་་་་རྙུ་་་སྐྲུ་་་་་་1 1་་་་ཁྱེ་་་ཟི་་་་ཁྱེ་་་་་1 1་་་་ཟློ་་་་་་་་ཟི་་་་་་། 1་་་་ཁྱི༷་ ཁུ་་་་་་༷་་་ཙྭ་་་་་1 1་་་་་་་་་་་ཁྱི་་་་་་་་་་༦། །ཁུ་་་་་་་་་་་་་་་་་1།་་་་་་་་་་་་་་་་་་་་ཁྱི་1 1་་་་་་བྱུ་་་སྐྲུ་་་་་་1 1་་་་་ཁྱི་་་་་་་་་་་་1 1་་ཆོ་་་་་་་་་་་་་་་་་་1 1་་་་་་་་་་་་་་་སྤྱེབྱུ་་་་།།་ ་ 11་་་་་་་ཙུ་ ་་སྟུ་་་་་ཙྭ་་་་་་་་་་་་ཉུ་་་ཅྭ་ ་་1 1་་་་་་་་་་་་་་་ཁྱི་་་་་་་་་། 1་་་་་་་་་་ཁུ་་་་་་་་ཟི་1 1་སྤྱོ་་་་་་རློ་་་་་་་་་་1 1ཁྱི་་་་་་་་་་་་་་རྐྱུ་་སྟེ1་1་་་་་་་་་་་་་་་རྙུཅྭ་ ་་ཙུ1 །་་་་་་་་ཁྱི་་་་་་་་་1 1་སྐྱེབྱུ་་་་་་་་་་་ཆི་་་རྙུ1 །ཟློ་་་ཁྱི་་རྨྱུ་་་་་་ཁྲི་་་རྭ་་ཉུ1 །རྩི་་4་་་་ཙོ་་་་་་་་་་ཚོ1 1་ཁུཙྭཅོ་ཚོ་་ཅོ་སྐྱི་་་་་1 1་་་་་་ཙྭ་་རྭ་་་རྙུ་༽ཕྱོ་སྦྱོ་1 1་ཁུ་་་སྐྱི་་1 
།ཟློ་་་་་་་ཁུ་་་་་1 1ཛི་་་་་་་་་་་་་་་ཁུ 1་་་་་ཁྱི་་་་ཁུ་་་་་་1 1་་་་་་་་ཁུ་་་་་་་་་ཁྱི1 1་་་་་་་་ཁུ་ཁུ་་་་་་་་་1 1ཛི་་་་་ཁྱི་་་་་་་་སྣྲི་་་་1 1་་་རྭ་ ་་་་་་བྱུཛི་་་་་1 1་་་་་་་་་་་་་་་ཟི་་1 1་་་་་མི༷་་ཇོ་་་་་་་སྟུ་་་་་1་14་་་་་་་་་་་་་ཁྱི་་་ཁུརྭ་ ཁུ1 1་སྒྲོ་་་་ཁྱི་་༷་་་ཁུ་་་་་་ 1་་་ཙྭ་་་་་་་་་་་་་་་་ཁྱེ་་་ཟི་་་་་་་་་་་་་་1 1ཁྱི་་་་་ཁྱི་་་་་་་་་་་་་1 1་ཁུ་་་་་་་་་་ཙྭ་་་་་་་1 1ཁུ་་་་་་་་་་་་་་་་་1 1་་་་་་་་་་་་་ཉུ་་་་་སྟུ་1 1་ཟློ་་་་་་ཉུ་་་་ ་་་་་་་1 1་་་་་་་་་་་་་་་་་གོ་་་་་་ཙྭ1 1་་་་་་་ཁྱི་་་་་་སྡུ་ ཅོ་སྔོཚོཟློ1 1་་་་་་་ཉུ་་་་་རྨྱུསྟུ་་་་་་་། 1་སྤྱོ་་་་ཁྲུ་་་་ཟི་་་་་ཁུ1 1་་་་་༦་ཞུརྨི་་་་་་་་ཁུ་་་ཙྭ་་1 1་་་ཁྱི་་་་་་་་་་་་་་་་་་1གོ་་་་་་་་་1གོ་་་་་་་ཟི་་་་་་་་་ཁུ་་་་1 1སྐྱུ་་་་སྟུ་་་་་་་་་་་1 1སློ་ཆི་་་་་་་་་་་་་བྱུ1་1་་་་་་༦གོ་་་་་་་་་་་སྤྱོ1གོཁྱི་་་ཛི་་་་་་་་་་་་་1 1་སྦྱི་་་་་་་ཟློ་་སྦྱི་་་་་ཟློ་་་་་་་་ཚི་་་ཟོ་1 1རྩོ་་བྱོ་་ཉུ་་་གློ་ཁུ་་སྟེ་ཙྭ་1 1སྦྱིསྩི་་རྙུཉུ་ཙྭཟི་་བྱུ་་་1 
1་་ཁྱི་ཉུ་་སྐྱི་་་་་་་ཟི་སྟོ་་་་་་་་་་་་་་་་་་་་་1 1་ཁུ་་་་་་་་་་་་་་ཁུ་སྣྲི་་1 1་་་་ཅྭ་ཙོ་ཉུ་་ཙོ་་་སྔོ་་་་1 1་་་་་་་་་་ཁུ་་་་་་་་ཁུ1 1་ཁྱི་་་་་རྨྱུ་་་་་་་་ཁུ1 1་་་་་་་་་་་་་་་་་་་1 1་ཁུ་་་་་ཁྱེཁུ་ཟི་་་་་ཁྱི། 1་་་སྟུ་་་རྨྱུ་་་་་་་་་་་་་་་་1 1་་་་་་་་་་་་་་་་་་ཀློ་་་་་1 ་སྩོ་་ཁུ་་་་་་་ཚི་་་་་ཁྲུ1 1་་།་ ་་་་་་་་་་་་ཅྭ་་་་་1 1༷་ ་་་་་་་་་་ཁུབྱུ་སྒྲོ་། 1་་་་ ་་་་་་་་་་་་སྟུ་་་་་་1 1་ཁྱི་་་་་་་་་་་་་ཁུ་་་ཟི1 1ཟློ་་་་་ཟློ་་་་ཛི་་་།་ ་་་་་་1 1ཁྱི་་་་་སྟུ་་་་༦་་་་་་སྨོ་་སློ་1 1་་་་་་་ཁྱེ་་་སློ་་ཁུ་་་་རྙུ་1 1་་་སློ་་་་་་་་་སྒྲོ་ཇོ་་་་1 1་ཁུ་ཅྭ་ ་ཚོ་་་་་ཁུ་་་་ཟི། 1་་་་་་་་་་་་་་ཟི་་་་་་1 1ཁྱི་་་་་་་་་་་ཁུ་་་་་ཁུ1 1ཁུརྭ་་་་་་་རྭ་་4་་་་་་སྦྱོ1 1་་་་་་་་་་་་་་་་སྟུ་་་་་་། 1་སློ་ཅྭ་ ་ཁྱི་་་་་ཁྱི་་་་་ཙྭ1 1་་སྨོ་་་་་ཏཱ་་་་་་་་་་་༦་1 1ཙྭ་ ་་་་་་ཁྱི་་་་་་་་་་ཁུ1 1་ཁྱེ་་་་་་་་་རྭ་ ་་་་་་་༷་་་ལྗོ1 1་སྙི་་ ཁུ་་ཆོ་ཙྭ་སྐྱི་་་་་་1 །རྩི་སྐྱིཁུཚོ་ཏྲུརྨྱུ་སྟུ་་་་་སྤྲི1 ་་་ཅི་་་་ཚི་་་ཁྲི་་་་ཆི་་་སྟུཇོ་ཁྱི་་་་1 1་ཁྱི་་་་་ཀེ་་་་་་༷་ ་་་་་གྱི་1 1་་་་་་་རྙུ1 
1་་་་ཁྱི་་་་་་1 1་་་ཛི་་་་་༦་་་་་་་བྱུ་་་་་1 1་་་་་སྦྱི་ཅི་་སྒྲོ་་་་་་་་་་་་་་་1 1་་་་་་་་་་་་་ཛི་་་༦་་་1 1་་་་་་་་་་་་་་་་་་ཛི1 1ཛི་་་་་་་་་་་་་་་་་་1 1་་་་4་་་་་་་་་་་་ཉུ་རྭ་ ་་་1 ་1་་་་་་་་་་་་་རྙུ་་་་་ཏོ1 1་་་་་་་ཟི་་་ཟོ་་་་་་ཙྭ་་་་་1 1་་་་་་་་་་་་་་་་་་་་་་ཁུ1 ་་་་་་་སྨོ་་སྨོ་་་་་ཆོ་་་་༷1 1་་་་་་་་་་་་་་་་་4་་1 །ཤཱ་་་་་༦་་་་་་ཁྱ་་་་་་་1 1་་་་་་་་་་་་་་་ཟློ་྅་་། 1་་ཁྱི་་་་་་སྟུ་་་་ཁུ་་་་་་་་1 །་་་་་་་་་་་་་་་་ཁྱི་་། 1་༷་་་་་་་་་ཁུ་་་་་་་་ཁྱི་་1 1་་་་ཁྱི་་་།་་་་་་་་་་་ཟོ་ཁྱེ་་1 1་་་་་་༦་ཙྭ་་་་་་ཟློ་་་ཟིཉུ་ ་་1 1་་་་་་་་་4་་་་་་་་་་1 །་་ཟློ་་་་་་་་ཙྭ་་བྱུཉུ་ཞུསྦྱི་་1 1་་་་་་་སྣྲོ་་ཙྭ་་་་་་་་་་1 1་་་་་་་་་་ཉུ་་་་་་བྱུ1 ་1 ་་་་་་་་་སྟུ་་་་་་ཁྱི1 1་་་་བྱོ་་་་་་་་་4 ་ཁྱི་་1 1་ ་་་་་་་་་ཆི་་ཉུ་་་་་་གོ1་ །1་་་་་ཁྱི་ཙི་ ་་ཁྱི་་སྣྲོ་་་་་་་་་་་་་་་་་་ཟློ་་ཉུ།ཅྭ་ ་རྨྱུ་་་་1 1ལྗོ་ཆོཁྱི་་་་་་་་ཛི1 1་་༦་་་་༷ནུ་་་ཁྲུ་་རྩོ་་ཁྱི་་ཁྱི1 1་་་་་་་་་་་་་་ཁྱི་ཇོསྣྲི་་་་1

་་1་རྙུ་་་་་་་་་་་་་་་་་་་་་1 1་་་ཅྭ་ ་ཁྱི་་་་་་་་་་་་1 1་་་་་་་་་སྟུ་་་ཟོ་་ཁྱི་་་་1 1་་་་ཁྱེ་་་་་་་་་་་་་་་་ཁུ1 1ཛི་་་་ཁུཁྱི་་ཅོ་་་་་་་་1 །་་་ཅྭ་ ་ཁྱི་་་་་་་་་་་ཁུ་1 1་་་་་་་་་་་་སྟུ་ཙུ་་་སྟུ1 1་་་ཁྱེ་་་་་་་་་་ཁྱི་་་གློ་་་1་་་་་་ཁུཁྱིབྱོ་་་་་ཆི་་གོ་ ་།་1་ཁུ་་ཅྭ་ ་ཁྱི་་་་་་་་་་་་1 1ཟློ་་་ཆི་་་་་་་སྟུ་་་་་། 1་་་་ཁྱེ་་་་་་་་་་་་་་་ཇོ1་1་་་་་་་་་་་་་་་་་་་་་་1 1་་་ཁྱི་་་་་་་་་་་་་་ཟོ་་་་་1 །་ཏོ་་་༦་་་་་་་་་་་་་ཟི་་་་1 །་་་་་་་་་་་་་་ཁུཇོ་ཟློ་1 1་་་་ཁུ་་རྨྱུ་་་་ཁུ་་་སྐྱུ1 །་་་་་་་་་་་སྔོ་་་4་ཁྱི། 1་ཁྱེ་་་ཁུ་་ཆོ་་་་་་ཁུ་་1་་་་་་་་་ཇོ་་་་་ཁྱི་་་་1 1་ཁུ་་་་་་་་་་་རྨྱོ་་་་1 །་་་་་་་་་་ལྟོསྙོ་་་་་1 1་་་་་་་་་་་་ཁྱེ་ཁུཉུ་ ་ 1་་་༷་་་་་་་སྐྲུ་་་་རྒྱོ་་1 1་་་་ཁྱི་་་་་་་་་་་་་་་་་་་་1 1ཁུ་་་་་་་་་་ཉུ་ ་་1 །ཁུ་་ཆོ་་་་་་་་གློ་། །ཁུསྤྱོ་་་་་ཉུ་ ་་ཛི་་་བྱུ1་།ཁུ་་་་་་་ཛི་4་ཧཱུཾ་་1 རྙུ་་་་ཅི་་རྩོ་་་་་1 ་1ཁྱི་་་་་་ཅི་་སྐྱི་་་ཁུ་མོ 1ཉུ་་་་ཅི་སྙི་་་ཉིསྦྱི། 1་་་་་་༦་་་་་་་་་་་ཟློ་སྤྲི་་ཟོ་། །་་་་་་་་ཙོ་་སྣྲི་་་་་་་་་1 1་་་ཚོ་་་་་་་་ཉུསྣོ་ 
་་ཉུཅོ་་་་་སྦྱི་་ ་་་་་་་་ཐོ་སྐྱུ་་་་་་་་་་་ཅྭ་ ་རྨྱུ1 1་་་་་་་་་་་་་་་་་ཁྱེ་། 1་་་་་་་་་་༦་་་་་་སྐྲུ་་་་་་་་་་་་ཁྱེ་་་་་་་་་་་་་ཉུ་་་་་་་་ཁྱི་་་་་་་་་་་་1 1ཙྭཁུ་་་་་་་་་་་་་་་ 1་་་་ཁྱི་་་ཟློ་་་་་་་་་་་་1 1་་་་་་་་༷་ཉུ་ ་་་་་་་སྐྲུ་་ཇོ་་་྅1 1་་་་་་་་་་་ཁུ་་ཇོ་་་་སྐྲུ་་་་་་ཁྲི་། 1་་་་ཆི་་་་་བྱུ་་ཇོ་་་་སྐྲུ་་་་་་།1 1་་་ཁྱི་་་་་་་རྙུ་་་་་པྱོ་་་སྐྲུ1 1ཁྲུ་་བྱུ་ ་་་ཁྱི་་་་་་་་་་་་་་་། 1་་་་་་་་་་་་་་་་་་1 1སྨོ་་་་་་ཆོ་་་་་་་་་རྭ་་་་1 1་་་་་་་་ཁུ་་རྙུ་་་་་་་་སྤྱོ་། 1་་ཁྱི་་་ཉུ་་་་་་་་་དྲོ། །སྨོ་་་་་་་་་་་སྨྱུ་་་་་་། 1་་་་་་་ ་་་་་་་་་་་1་1་་་་་་ཟློ་་་་སྟུ་་་་་། 1་་་སྟུ་་་་་་་་་ཁུ་་་1 1་་་ཏོ་་་་་་སྟོ་ ་་་་་། 1་་་ཁྱི་་་ཁྱེ་ཆོ་་་་་་་ཉུ་ ཁུ1 །་་་་་ཛི་་་་་་་་་་སྟུ་སྣོ་་་་ཁྱི1 1་ཁྲི་ཉུ་་་་་་་་་སྟུ་རྩོ་་་ཁྱི1 1་་ཇོ་་སྨོ་་་་་་་་་་་༔1 1་་་་་་་་་་་སྦྱི་་་་་སློ1 1་་་་་་སྤྲི་ཚེ་་ཁུ་་་་་1 1་་་རྭ་་་་་་ སྤྲོ་་ཙྭ་་་་་ཁྱི༷་ཐོ་ཙྭ། 1་སྐྱིཇོ་་་་་ཕོ་་་་་་་ཟློ་་་་་་་1 1་་་་་་་་ཉུཉུ་ ་་་སྤྲོ་་ཁྲི1 1ཛི་་་་ཁྱི་་་་་སྦྱི་་་་1 1་༷་1 
་།་་་་་་་་1 1ཁུ་་་་་་྅་་་་་རྩོ་་1 1་་་་་་་་་་་་གློ་་ཉུ་་་་་1 1་་་་་་་་་་ཁུ་་་་་་་ཛི1 1་་་་ཁྱི་་་་་་་་་གློ་ཚོ་་ཁྱི་། ་་་ ་རྨྱུ་྅་་་་་་་་་་་ཛི་་1 །་ཅྭ་ ་རྨྱུ་ནྟྲ་ཁུཔྱོ་་་་ཅོ་་་ཁུ་་་་་་་་1 ་་ཅྭ་ ་་་་་་་་་་་ཁྱི་་་༦་་་1 ་་ཅྭ་ ་་་ཁུ་་་་་་ཉུ་་་་་་་་ཁུརྭ། ་ཅྭ་ ་་་་་་་ཁྱི་་སྟུ་་་་་་་། །་་་་་་ཚི་་་་་་་་སྦྱོ་་་་་་་་་་ཇོ1 ་ཅྭ་ ་་་བྱུཟློ་་སྐྱོ་་་་་་་་་་1 1་་་་རྭ་་་་་་་་སྤྲི་་ཁྲི་་་་་་་་་་་་་1 ་ཅྭ་ ་་་སྤྱི་་་་་་་་་་་་་་1 །་་་་་་་ཁྱི་་་་་་་་་་སྟུ་་་་་་་ཁྱེ་་་། ་ཅྭ་ ་་་ཎྜ་་༷ཚི་་་་་་ཆི་་1 །་་་་་་་་་་་་ལྗོ་་་ཙྭཁྲི་་་་་་ཇོ་་། ་ཅྭ་ ་་་རྨྱུཁུ་་ཁྲི་་་་་་་་་་། 1ཁྱེ་་་་་་་གློ་་་་་་་་་ཁྲི་་རྭ་་་་་4་1་།་་་ཁྱི་་་་་་་་་་་་་ཟི་་རྨྱུ་། །་་་་་་་་་་་་་་་་་་་ཟོ་་་ཁྱེ1་1སྨོ་་་་་་ཅྭ་ ་་་྅་་་1 1ཁུ་་་་་་་ཅྭ་ ་་་་་ཁུ། །ཁུ་་་་་་གོ་ཅྭ་ ་་་ཉུ་1 །ཁུ་་་་་ཅྭ་ ་་་་་། །སྟུ་་་་་ཅྭ་ ་་་བྱུ་། 1ཁུ་་་་་་ཅྭ་ ༷་ ་རྒྱོ༷1 1ཁུ་་་་་ཅྭ་་རྨྱུ་རྨྱུཁུ1 1སྦྱི་་་་ཅི་ཅྭ་ ་རྨྱུ་་ཐོ་11སྦྱི་་་་་་་་་་་ཁུ་་་རྗོ་་་1 1སྦྱི་་་་་་་་ཛིགོ་་་ཁྲུ་་སྤྲོ1 1་་ཟློ་་ཙོ་ཟློཁུ་་་ཁྲུ་་་་། 1་་་་་་་་་སྙོ་་་་ཙྭབྱུ་སྣྲོ་11 ་ 
།1ཏོ་་་ཁྲི་་་་་་་་ཁྱེ་་1 1་་་་་་་་་ཚི་་་་་་་་་་་1 1་་4་ཛི་་་་་ཚི་་་་་་་་ཛི་་1 1ཁྱི་་་ཁུ་་་་་ཚི་་་་་་་་་་་1 1་་རྩིཙི་ ་་་་་་ཁྲི་་་་་་་་སྦྱིཁུ1 1་་་་་་་་་་ཁྲི་་་་་་་་་་ཁྲི། །་་་སྤྱ་་་ཚོ་་་་་་་་་་ཉུ་ ་་1 །་་་་་་རྩོ་་་་་་་་་་་་ཟི་་་་1་1ཁུ་་་་་་་་་་་་་་1 སྟུ1 ་་་་་་་་རྭ་ ་་་ཁུ1 1ཁུ་་་་་་་་་་་ཙྭ་སྟུཆིགོ 1ཁུ་་་་་་་་་་་་་1 །ཁུ་་་་་་་་་་་་་4་་1 སྟུ1 ་་་་་་་་་་་་་་་ཇོ1 1ཁུ་་་་་་་་་་ཟློ་་་་ ཟློ། 1ཁུ་་་་་ཁྱེ་་་་་་་་་། 1་་ཆོ་་་་་་་་་་གློ་་་་་་་་1་1 ་ 11ཕྱུ་་་ཁྱེ་་་་་་་་་་་་་་་་་་་་ཁུ་་་བོ་་་་་་་རྨྱུ་ ་་གོ་་་་་་་་བྱེ་ཙྭ་ ་་་ཐོ་ཞུསྐྱུ། 1་་་་་་་་ཟློ1 ་ཉུ་་་་་་་ཁུ་་་་་་་སྟུ༷་ ་་་་1་1་་་་་་ཧཱུཾ་་་་་་་༷་ཆི་་་་བོ1 1་སྩོ་་་་ཁུ་་་་་་་སྩིབྱུ་་་བློ། 1ཟློ་་་་༷་་ཁྲི་་་་སྤྲོ་་་ཁུ༷་་་1།་་་་་་་་་་་་་་་ཁྱི་་་་་་་་1 1་྅་ཞུ་་རྭ་ ་རྭ་ ་་་་་་ཁྱི1 1སློ་ཟི་་་་་་་་་་སྐྱོ་་་1་1་་ཆོརྨྱེ༷་ ་་་་་་༷་་་་སྣྲོ་་ཟོ་་། ་1་་པྱོ་་༦་ཁྱེ་་་་་་་་་་་་་ཟོ་1 1ཉུ་་་་་་་་་་་་སྐྱི་་་་་ཁུ1 1ཁྱི་་་་་་་་་ཆི་་ཟློ་་་སྤྲོ་1 །རྟུ་་་་བྱུ་་་ཆི་་་་་སྦྲིསྤྲི་ 1 1རྟུ་་་་་་་་་་་་གྱི་་་ཁྱེ1 
1ཟློརྨྱུ་ག ་་་སྟུ1 1་་་་་་་་་་་་་་་་་་་་1 1་་ཁུ་་་་་་་་་ཁྱི་་་་་་་་་སློ་་་1 1་་ཇོ་༦་་་་་་་་་་་་་་་་་ཁུ། 1་་་་་་་་་་་་་་་་་ཟོ་་་་་་་་་1 །་ཁྱི་ག། ་་་་་་་་་་་་་་་་་་་་་་་་་་1 1་་་ཁྱི་་་སྤྱོ་་་་་་་་་ཟོ་་་་1་1ཁུཟློ་་་་་་་་་་་སྨོ་༦1 །ཁུབྱོ་་་་་་་་་་་་་་་་་ཟི། 1ཁུ༷་་་་་་་་ཟློ་་་་་་་1 ་་་་།་་་་་་་ཕྱུ་་་རྙུ1 1་་་་་་་་་་་་་་་ཉུ་་་་་། 1་་་་་་ཡི་་་་་་་་་་་་སྟུ1 1་་་་་་་་་་་་ཁྱི་་་་་་་་་1 ་།་་ཛི་་་་་་་་་་་་་བོ་་་ཉུ། 1་་སྦྱི་་ཁུ་་་་་ཛི་་་་་་་ ་1 1་ ་ཇོ ་ཏོ་་་་་་ཁུ་་་་་་ ་་1 1་་་་ཁུ་་་་་རྩུ་་་་་་་། 1་་གློ་བྱུ་་་་་་་་་་་་་་་་་1 1་་རྨྱུཇོ་་་་་་་་་་་་་་་བྱུ་་1 1་་་་་་་་་་་་་་་་་་་་་བྱུ་།་།་་་ ཚེ་་༦་་་་་་་་་་་་་ཟི་་་་། 1ཁུཐོ་་་་་་་་་་་་་སྤྲོ 1ཁུ་་་་་་་་་་་་་་ཁུ1 1ཁུ་་་་་་་་་ཟློ་་་་་།་1ཁུ་་་་་་་་་་་སྦྱིཁུ1 1ཁུ་་་་སྤྱོ་་ཛི་་་སྤྱོ་མོ་1ཁུ་་་་་་་་་་་སྐྱོ1 1སྦྱི་་་་སྨྱོ་་་་་་་ཆི་བྱུགོ 1སྦྱི་་་་་ཅི་་་་་་་སྦྱིབྱུ1གོ་་་་་་་ཟློ་་་་་་་་ཙུ་་ལྗོ་་1 1སྐྱིརྩོ་་ཙོ་་་ཚི་་སྟུ་་་་སྙོ་ཙོ1་1སྤྱི་་་་ཁུ་ཙོཅོ་་ཁྲི་་ཤཱ་་་་་སྒྲོ་་། 1རྩི་་་ཆི་ཛི་

1་ཁྱེ་་་་་སྤྱོ་་ཕྲི་་་་་ཟློ་་་ཁྱི་རྩི་་་1་1་་་་་་་་སྤྲེ་་་་་་་་སྦྱི་་ཟློ་་་རྩོ་་ཇུ་་་་་༦་་སྦྱི་་་1 1་་ཁུཅྭ་ ་་་་་་་་་སྡིཚི་་་1 1་་་་ཁྱི་སྐྱི་་་ཁྱི་་་་་ཁྱི་1 1་་་་་་སྦྱི་་་་་སྦྱི་ཁྱི་་1 1་་་་ཁྱི་སྐྱི་་་ཁུ་་་་་སྟུ1 1་་་ཁྱི་་་་་་་་་་་་་ཟི་་་་1 1་་་་་་ཙྭ་་་་་་་་ཁུ་་་་ཛི། 1་་་་་་༦་་་་་ཚོ་་་་་་ཟི་་ཙྭ་1 1ཙྭཁུ་་་གོ་་་་་ཁུ་ཁྱི་་་༷་་1གོཁུ་་་་་་་་་་་་་ཇོ་་་་་1 1ཁུ་་་་་་་་་་་་ཁུ་་རྭ་ ་་ཁྱེ་1 1ཁུ་་་་་་་ཁུ་་་ཁུ༷་ ་་་སྟུ། །་་་་་་་་་་་་གློ་་ཉུ་་་་་1 ཉུ། སློ་་་་་་ཙོ་་ཏོ་་ཁྱི་་་་་1 ་་་་་་་་་་་་་་ཁུ་ཁྱི་་་་་་་1 1་སྟུ་ ་་་་་་་སྐྱུཁྱི་་་་་ཙྭཁྱི1 1སྟུ་་་སྤྱོ་་་་་་་་་་་་ཁྱེསྤྲོསྦྱོ་་ཁུ་་ཁུ་་་1་1་་་་་་་་་་་་་་་་་་་་་་་་་་་1 1་་་་་་་་་་ཁུ་་ཉུ་་་་་་་་་ཁུ་་ཁུ1 1་་་་སྤྱི་་ཁུ་་་ ་་་་་་་་ཟི་ཁྱེ་་1 1་ཏོ་་་ཅོ་་་་་་་ཕྱུ་་་་་་་༦1 1་ཙོ་་་་་་་་་་རྭ་ ་་་་་་་1 1རྭ་་༷་ ཆི་ཇོ་་སྒྲོ་་ཟློ་་་་ཙྭཁྱི་1 1་་་་་་ཆི་་་སྤྱོཉུ་་་་་་ཁྱི1 1་་་་་་་་་་ཉུ་་་་་་་་་1 ་1་་་་་་་་་་་་་ཆི་་ཉུ་་་ལྗོ་།1 ་ 11་་་་ཅི་་ཆིཟློ་ཁྱིཙི་ ་་ཛི་་ཕྲི༷་་་་་་་་་་ཆོ་་ཛི་ཅྭ་ སྟོ་ ་་་ཁྱིསྐྱུ1 1ལྗོ་་ཁྱི་་་་་་་ཛི1 1་་སྒྲོབྱོ་་་་་་་སྐྱི་གོགྱིཁྱི་ཐོ་གོ 1་་་་སྤྲི་༷་་་ཕྲི་1་ 
1་་་་་་་་་་་ཟི་་་་1 1་་་་་་་་་་་་་་ཁུ་་་་ཛི1 1་་་་་༦་་་་་་་་་་་་ཟི་་་་1 1ཁུ་་་་་་་ཚི་་ཅུ་་་ཙོ་་་་་་1 1ཁུ་་་་ཅི་ཚི་་ཁྱུ་་་་་་་་་་ ཁུ་་་་་་ཆི་་བྱུ་་་་་་་་་་྅1 1ཁུ་་་་་་ཚི་་ཏྲི་་་་༷་ ་་་1 1ཁུ་་་་་་ཚི་་བྱུ་་་་་་་་ཉུ1 1་་་ ་པྱོ་་་ཚི་་བྱུ་ཁྱེ་་་་༦་་1 ་་་ཚོ་་་་་་་་་་་་ཉུ་་་་་1་།་གོ་་་་་་་་་ཁུ་ཁྱི་་་་་1 །བྱུ་ སྟུ་་་་་་་་་་་་་་་གོ་་1 1་་སྤྱ་་་་་་་་་་་ཇོརྭ་ ་་་་་1་།སྟུ་་་་་་་་་་་་་་་་་་་་་་་1་་་་1་།་་་་་་་།་་་་་ཁྱི་ཉུ་ ་་་་་་ཁུ་་་་་་་་་་་་1 ་1་་་་་་་་་་་་་སྟུ་་ཁྱེ་་་་་་་་སྟུ་་་་1 1་་་་སྨྱོ་་་་་་་་་་་་་་ཟི་སྨོ་་་1 1ཚོ་་་་་༦་་་སྦྱོ་་་་་་་་སྨོ་་་་1 རྭ1 ་་་་་རྭ་ ་་ཁུ་་་་་་་་་་1 1་་་་་་་ཇོ་་་་་ཁྲོ་་་་་1 ་1་་་་་ལྷོ་་་་་་་་་་་་་ཕྱུཁྱི་1་1་་་་་་་་་་་་་་་་་་་་སྟུ།་1་་ཟློ་་་་་་་་་་་་ཉུ་་་་་11 ་ 1་1་་་་༦་་་ཆི་་་ཙི་ ་་་སྟོ་་་ཇོ་་ཉུ་་་་་་་་་་་་་་་ཙྭ་ ་རྨྱུ་ཐོ་་་1གོ་ཁུ་་་་་༄་ ཙོ་་་་ཁུ་སྦྱི་ཆོ་་་1 1་་་་་་་་་་་་་་ཁུ་སྦྱིཁྱི་ཚི1 1་ཏོ་་སྐྱི་་་་་བྱུ་་་སྤྱི་རྩི་་1 1་་་་་སྤྲི་་་་་སྐྱི་ཉུ་་1་1རྩི་་་་་ཁྱི་་སྐྱི་་སྐྱུསྐྱི་སྦྱི་་་1

1ཇོརྗོ་་་་1 1་་་་་་་་ཟི་་་་་ཁྱི་་་་་་་1 1་་་་་་་་་་་་་ཁྱི་་་་་་་་་སློ་་་་་་་་་་་སྦྱོ1 1་་་་་་་་ཁྱི་་་་་སྟུ་་་་་་་ཧཱུཾ་་་ཁུ་་་1 1་་་་གློ་་་་་་་་་ཁྱི་་་་་ཟློ་་་་་་སྤྱོ1 1་་ཉི་་་་་ཐོ་་་༦་་་་་་་་ཉུབྱུ་སྒྲོ་་1 1་་་་་་་་་་་་་་་་་4་་བྱུཁྱི་་ཁྲུ་་་1 1་་་་་་་་་་་་ཁྱི་་ཁུ་་་་་་་ཁྱེཇོ༷་་ཚིཙུ་ ཚེ། །་་་ཁྱེ་་ཟློ་ཁྱི་་་་་་སྟུ་ཕྱུ་་་་༷་༷སྤྲོ་་ 1་་་་་་ཁུ་་་ ་་་་ཁུརྭ་་་སྣྲོ་་་1 1་་་ཉུ་་་་་་ཚོ་་་ཁྱི་་་་། 1་་་་་སྨྱོ་་་བྱོ་་་་་ཟློ་་་སྐྱུ་གློ1 །་་༦་་་་་་་་་ཅུ་་་་་་1 1་་་་་་གློ་་་་་་་་་ཏྲི་་ཁྱི1 །་་་་་་བྲེ་་་་་་་་་་་1 1གློ་ཁྱེ་ཁྱེ་་་་ཀེ་སྣྲོ་་་་་་ཁུ་1 1་་་་་་་་་ཏྲི་་རྟོ་ཉུ་ ་་་༦1 1་་་་ཁུ་་་་་་་་་་་་ཅོཁྱེ་་བློ1གོ་་་ཁྱི་ཙུ་་་་རྭ་་་་་་ཟི་་་་1 1་་་་་རྨྱེ་་་་་་་་་་་་ཟི་་་་། ་1་་་་་་་་་་་་་སློ་་་་་་་་་་་སྦྱོ་་་་་་1 1་་་་་་གློ་་་་་ཆོ་གློ་་་་་་་་ཟི་་་་་་་་་་། །་་ཛི་་་་་་་་་བྱུ་ཁུ་4་ཁུ་1 1་གོ་་་་་་་་་་་་་་་་་་་་སྟུ་་སྒྲི1 ་་་་་་སྟུ་་གོ་་ཟློ་་༷་སྦྱི་་་་་སྦྱོ་1་1་་་་་ཆི་སྟུ་་་་་གློ་་ཁུ་་ཏྲུ་་་་སྤྲི་༩1་་་་་་་སྟུ་་་་་གློ་་་ཐིབྱུ་ རྗེ་་་་སྦྱོ། 
།་་་་་་་སྟོ་་་་་ཉུ་་་་་་་་་་་་་ །་་་་་་་་་་་་་ཁྱི་་་ཁུ། །་་་་་་་ཟི་་་་་་གློ་་སྦྱོ་ཁྱི1 1་་་་་་་་་་་་་གློ་་་སྦྱི་་་་་སྦྱི་1 1་་་་་་་་་་་་་་་་་་་་་1 1་་་་་་་་་་བྱོ་་་་་་་། །་་་་་་་་་་་་་་ཁྱི་་་་་་་་་1 1་་་་་ཚི་་་་ཁྱི་་ཁྱེ་་་་་་་་ཁྱི། 1་་་་ཅོ་་་་་་་་་་་ཁུ་སྤྱོ་1 །་ཇོ་་་་་་ཁྱི་་་་་རྩོ་ཁྱི་་ཁྱི1 1་ཏོ་་་་ཁུ་་་་ཁྱི་་་ཇོ་་། །་་་་་་་་་ཟློ་་ཉུ་་་་་་་ཉུ་ཁུ་་་་་ཁྲི1 1༷་་་་་་་་ཁྱི་་་༦་་ཟློ་ཞུཁྱེ་་་་།་ཚེ་ཉུ་་་ཟི་་་་རྭ་་་་་་ཅི་་་་་1 1་་ཉི་་་་་་ཙུ་་༦་་་་་་་་སྦྱོ1 1་་་་་་་ཁྱི་་་༦་་་ཟློ་སྟུ་་བྱུ་་། །་་་སྨྱོ་་་་་་་་་་་་་་ཟི་་་་1 1་་་་་༦་་་་་་་་ཟློ་་་་ཟི་ཁྲུ་་1་།་་་་་་་་ ་་་་་་་་་བྱུ་་་་་སྟུམི1 1་་་་་ལྷོ་ཟི་་་་་་་་་་ཁུ་་་་ཁུ1 1༷ཁུཆོ་་་་་་་གྱུ་ ཙྭ་་་་་་ཅུ་ཁུཆི། 1་༽་་་་ཆི་སྨོ་་་སྤྱོ་་་ཁྱི་་ཁྱི་་ཟློ་་་་་་་། 1་་ཟློ་་གློ་ཇོ་་་་་་་་ཁྱེབྱོ་་་་ཁུ་་ཁྱི་4་་་1 1་་་་་་་་་་་་ཁྱིབྱུ་་་་་ཆོཁྲི་་ཁུ་ཚི་་་ཁུ1 1་་་་༷་་སྟུ་་་་་ཁྱི་་ཉུ་་་་་་་་་་་སྟོཙི་ ་1་།་་་་རྨྱུ་སྙི་ ་་་ཐྲེ་་ཁྱི་་་སྦྱི་་་་་་་ཁྱི་ཅོ་་ ་1 1་་་་་་་་་་་་ཉུ་་གློ་་ཉུ་་་ལྗོ་1 1་་་་་ཁྲི་་ཙོ་་་་སྙོ་་་1 
1་་་་༦་་་་་་་་་སྐྱི་་་་1 1་་་་་ཉུ་་་་ཁུ་་་་་་་ཁྱི་་་་1 1་་་་་་་་་་་་་་་་་་་་་། 1་ཁུ་་་་་་་ཁྱི་་་་་་་་ཁུ་་་1 །་་་་ཇོ་་་སྐྲུ་་་་་་་་་་་། 1་་་་་་ཐོསྟུ་་་་་་ཛི་༦1 1་༷བྱུ་་་་་་༦་་་་་་་་་་་་་1 1་་་་་་་ ་་་་་་རྩོ་་ཁྱི་་་ཆི་1 1་་་ཁྱི་་་་་་་་་་ཆོ་་་ཁྲོ་་༦1 །་་་་་༦་་་་་་་ཚོ་་་་ཟི་་་་1 1་་་་་་་ཇོ་་ཁྲུ་་་སྤྲོ་་་་་་སྦྱོ1 1་་་་༷་་་་ཇོ་་་་་་་་་1 1ཁྱེ་་་་་་་་ཁྱེ་་་་་་་་་་་1 1་་་་་་་ཙྭ་ཁྱི་་་་ཆི་་་་་སྔོ1 1་་་་་་་་་་་་་་་སྣེ་ཉུ་ ་་1 1་་་་་་ཇོསྟུ་་་་་་་་་་་ཁྱི་་་ཡོ 1་ཇོ་་༷་ཆི་་་་་༷་་་ཁྱེ་་་་་་་་་་ཅོ་ཁྱི་་་་1 1་༷་་༦་་་་་་་་་་་་་་་་་་་་་1 ་1་་་་་་་ཁྲི་སྨོ་་་་་་་རྭ་་་ཁུ་་སྦྱི་ཇོ1 1་་་་་་་་་༦་་སྤྲི་་་་་མོ་1་་་་ཁྱི་་་་་༦་་་་་་ཁུ1 ༷1 ་་་་་་་་ཅོ་་་སྟུ་་་་་གློ་1 1་་་་་་་་་་་་་་་ཁྱིཙྭ་་་་་གློ་1 1་་་་ཁྱི་་རྭ་་ཟློ་་་་་་་་་་་་་་། །་་་་་་སྐྱི་་ཙྭ་་་་་་སྒྲོ་་་་ཕྱུ་། 1ཟློ་་་་་་་སྟུ་་་བྱུ་་སྒྲོ་་་་སྤྱེ། 1་་་་་་་སྟུ་སྟུ་དྷི་་་་་་་་1༽་་་་་་་་སྣྲོ་་་རྨྱུ་ སྐྱི་་ ་་་1 1་་་་་་་སྟུརྭ་ 
དྲི་་ཁྱི་སྦྱི་་ཁྱི1་1་་་་་་་་་་་་་་ཁྱེ་་་་ཆི1 1་་་་་་་་ཚོ་་་་་་་་་་1 1་་་་ཁྱི་་་་་་་ཛི་་སྐྱུ་་་ 1་་་་་ཁྱེ་་་་་་་་་་་་་་༦་་་་ཁུ་1 1་་་ཁྱི་་་་་་་་་་་་་ཟི་་་་1 1་་་་་་་ཁུ་་་་་་་་་་་་་༦1 1ཁྲུ་་་་་་་་་་་4་་་་་་་་1 1་་་་་་སྟུ་་་་་ཇོ་་་་་ཁུ1 1་་་་་་་་་་་་གློ་་་་་རྩུ་་11 11་་་ཚོ་་་་་་་་ ་་་་ཁྱེ་་་ཁྱི1 1་་་་་་་་་་ཅྭ་ ་་་་ཁྱེ་1 །་་་་་་་ལྟོ་་་་་་་་་་་1 ཆི1 ་་རྭ་་ཁུ་་་་་་་་་་་་ཁྱི1 1་་་༷་་་ཉུ་་ཙྭཁུ་་་་་་་་་། 1་ཁུ་་་་ཁུརྨི་་་་་་༦་་་་་་་་1 1་ཁྲུ་་་༦་་ཉུ་་་་རྙུ་་་༦་་་་་་ཇོ་1གོ་ཁུ་་་་་་་ཚོ་་་་་་་་་་་་་་་1 1་་་་་་ཁུ་་་་་་་་ཁྱེ་་ཁྱི1 1་་་་་་ཛི་་ཁྱི་་་་་་་་་1 1་་་་་་་ཁུ་་་་་་་་་་་་་་་1 1་ཙྭ་་་་་་་ཙྭ་་་ཐོ་་ཙྭ་་ཟློ་་་ཁྱི1 །་་ཛི་ཁྱེ་་ཁྱི་་་་་་་་ཁྱི་ཙྭ་་1 ་་་་་་་་་་་་ཟློ་སྒྲོ་་་་་་1 1་་་་་་་་་་་་་་ཁྱེ་་ཁྱི1 1་་་་རྭ་་༦་་་་རྭ་་སྟུ་་་་་་ཁྱེ1 1་་་་་་་་སྟུ་་་་་ཁྱེ་་ཁྱི1 །་་་ཁྱི་་་་་་སྙི་་་་་་སྒྲོ་གྱོཟོ་1 །་་་་་༦་་་་་་་་་་་་ཅི་་་༦1 1་་་་་ཅི་་སྤྲི་ཉུ་་་་་་་1 1་་་་་ཀྲོ་་་་ཙོ་་་་་་ཁྱེ་་ཁྱི1 1་་ཚོ1 
་1་་་་་་ ་་་་་་་་་་ཅྭ་ ་་་་་་1 1་་4་་་་་་4་་་་གློ་་ཛི་ཡི1 1་ཟི་་་་་་་་་་་་་་ཚི་་་་ཁུ་1 1་་་་་ཛི་་་་་་་་་་་་ཁྱི1 1་་་་་་་་་་ཁྱི་་་་་་་་་རྨྱུ1 1་་་་་ཁྱི་་་ཏྲོ་་་ཛི་་་་་གློ1 1ཇོ་་་་ཇོ་་་ཉུ་་་་་་1 1་ཁྱི་་་་་་ཁུ་་་་་་ཅྭ་ 1 1ཉུ་་་་་་་་་་ཟློ་་་་་་བོ་1 1ཁྱི་་་་་་་་་་་་་་་་་སྡུ་་ནཱ་1 1སྤྱོ་་་་་་་་བྱུ་ ་་་་་་་་་ཉུ་1 1་་་་་་་རྭ་་་་་་མྱུ་་་་༦1 1ཁྱི་་་་་་་་་་་་སྨོ་་་་་་་༦1 1་་་་་་་་ཚོ་ཁུ་་་་གློ་་ ཟོ1 བྱུ1 བྱུ་ ཅྭ་་་་་་་་་་་་1 1་་་་ཁྱི་་་་་་་་ཁྱི་བྱོ་་1།་་་་་་་རྗོ་་ཟློ་་་་་་ཁྱི་་་་་1 1ཁྱི་་ཉུ་་ཉུ་་་་་་་ཁྱི་་་་་1 1་་་་་་་་སྣྲོ་ཚི་ཙྭ་ཉུ་་ཁྱི1 1ཁྱི་་་་་ཉུ་་་་་་་་་་་1 1་སྤྲི་་་རྩོ་་ཟི་རྭ་་སྤྱོ་་ཙྭ། 1རྩོ་་ཉུ་་ཉུ་་་་་་་ཁྱི་་་་1 1་་་་་་་་ཟི་སྨོ་་་ཅྭ་ ་་1་1ཁྱི་་་་་ཉུ་་་ཁུ་་་་་་1་1རྨྱུ་་་ཟློ་་་ཙྭ་་་་་་ཅུ་་་་1 1ཁྱི་་ཉུ་་ཟི་་་་་སྤྱོ་་་་1གོ་་་་་་་་ཟློ་་་་ཚི་་ཁྱི་་་་ཁུ་1 1ཉུ་་་་་་སྤྱོ་་ཟློ་རྭ་ ་་་་ཇོ་་ཅྭ1 ་་་་སྤྲེ་་་་་་སྦྱོསྦྱོ་་་་་སྙོ1 1་གོ་་ཅོ་ཛིརྗོ་ཁུ་་ ་ཅོ་ཧཱུཾ་་་ཅོ་་སྤྲི། 1ཉུ་་་་སྦྱི་སྟུ་་་་བྱོ༷་་ལྷོསྨྱུ་ 1 1་་སྦྱི་་་་་་ཇོ་རྐྱུསྐྱི་ཚི་་་ ་སྦྱི་1 1རྩི་་་་་་་་་་་ཚོ་་་ཛི1 
1་་སྒྲོ་་་་་4་་་་་་་་་རྗོ་་་་་་་་སྐྲུ1 1་་་་་་་་་་ཛི་་་་ཁུ་་་1 1་་་་་་རྩི་་ཁུ་་་་ཁྱི་་ཁུ་ཟི1 1་་ཁྱི་་་་་སྟུ་་་་་་ཁུ་ཁུཚི་་་1 1་་ཁུ་་་་་་་་སྣྲི་་་ཁྱི་་སྦྱི་་་1 1་་་་་་སྨོ་་བྱུཛི་་བྱུ་་་་་་་1 1་་་་་་་་་་་་་་་ཁྲུརྭ་ ཁྱི། ་1 སྦྱི་ཁུ4་་་་་ཁྲི་་་་་་་་་བྱུ །་ཉུ་་་་་་་་་་ཁྲུ་་ཁྱི 1་་ཁུ་་་་་་་་་བྱུཁུ་་་ཛི་་་་་་ 1་་་་་་་་་་་་་་་་་་་་་་་ཁུ་་ཁྱི་1 1་་་ཀྲོ་1གོ4་་ཁྱི་ཕྱུ་་་་རྩི1 ་་་་ཁུ་་་་་ཛི་སྐྲུ་་་་་༷་་་ཁུཛི་་་ཁྱི་་་་ 1་་་་སྤྱོ་་་་་་་་་ཛི་་ཁྱི་་་་ 1་་་་་་་་་་་་ཆི་ཁུ་་་་་་་་་་་་་་་་་་་་་་་་་བྱུ་་་་༷་ ་ 1་་་རྨྱི་་་་ 1རྨྱུ་་་་་་ཆོ་་་ཅྭ་་ཁུ་་་་་་་་་་་་་་་ཁྱི་་་་་ཁྲི་་་ཟི་་་་་བྱོ་་1་་་་་་་་་་་་་་་་་་་་་་་་སློབྱེ་་་་་་་་ཁུ་་ཟློ་་་་་་་་་་་་་་་་་་་་་་་ཟི་་ཁྱི་་་ 1་་་་་་་་གློ 1་་་་་ཁུ་་་ 1་་་སྤྲོ་་་་་་ 1་་་་་་་་རྗོ1་་་་་་བློ1་་་་་་་བོ 1་་་་་་་་་་་་་་་་་་་་་་་་་ཟི་་ཁྱི་ཁུ་་་་ 1་་རྨྱུ་་་ཆོཙྭ་ཉུ་་་་་་ །་་་་ཁྱེ་་་་་་ཛི་་་་་ཙོ 1་་་ཉུ་་་་་་ 1སྐྱི་་སྟོ་་་ཁུ་ཛི་ཅྭ་ ་རྨྱུ་་་་ 1་་་་ཟློསྨྱོ་་ཁུལྗོ་སྩོཇོ་་་་་་་་སྨྱུ་ 11 11སྒྲོ་་་བྱུཆི་་ལྗོ་་ ་་ཛི་་་1 ་ 
1་་་ཆི་་་་་་་་་་་་་་་་་་ཙོ་་་་་་ 1་་་་་་་་་་་་་ཙོ་་་་་1 14་་་་་་་1 1་་ཟོ་ཉུ་་༦་་་་སློ་་་་་་སློ1 1་་་་ཁྲི་་་་་་་་་་་་་་་་1 1་་་་་་་་ཁུ་་་་་་ཁྱི་་་་་་་1 །་༷་་་་་་་་་སྦྱོ་་་་་་ཁུ་་་གློ1 །་་་་་་ཚོ་་་་་་་་་ཉུ་་་་་ཁྱི1 14་་་་་་་་་་་་་་་་་་་་་་ཛི་་་་་་་་་1 །4་་ཁྲི་་་་་་་་་་་་་་4་་་་ །་་་་་་་་་་་། 14་་་་་་་་་་༷་ ་་་་་་་་སྟུ་་་་་་སྟུ་་། 14་ཆོརྨྱི་་་སྤྱོ་་་་་་་་་་་1 1་རྙུ་་་་་ཁུ་་་་་་་་ཐོ་་་་་་རྙུ་་ཁྱི་་་་་་་1 །་་་་་་་་་་་་་་་་ཀེ་་་་་་་་བྱུརྨྱེ་་ཁྱི་་་་༷་ ་1་1་་་་་་་་་་་་་་་་་་་་་་་་་་་་ཟློ་་་1 1་་ཁྱི་་་་་་་་་་་་ཁྱེ་་་་་་་་་་་་4་་་་་་་ཁྱེཁུ1 1་་་་་་་་་་་་་་སློ་་༦་་་།་1་་་་་སྟུཁྲུ་་་་ཁྲི་་་་་་སྤྱོ་༷་་་སློ1 1་་་ཉུ་་་་་་་་་་་རྩུ་་་1 1སྟུ་་་་་་་་་་་་་་་་ཟློ་་་་་་་་་་1 1་་་་་་་་་་རྭ་་་་་ཟི་་ཁྱི། 1སློ་་་་་་་་སྟུ་་་་་་་ ་་་་་་་་་་་་1་1་་ཁྱི་་་་་་་་་༷་ ་སྙོ་་་་་་བྱུ་་་་་་་་་་1་1ཚོ་་་སྦྱི་་་་་་ཟློ་་སྟེ་་ ་གློ་་་་་་་་་་སྤྲི1 1སྤྲོ་་རྗོཁྱེ་ཇོརྩི་ཙྭཇོ་སྤྲུ་་་་་་ཛི་ཚོ་རྙུ ་ ་ཁྲིཇོ་་ 1་་་ ་ལྗོ་་༷་་་་་1 
།ལྗོ་་་་་་་་རྗོ་་་་་་་་སྐྲུ་་་་ཛི11 11་་་་་་་ཟི་་ཁྱི་་ ་་་་་་4་་་་་་་ 1་་་་་་ཁུ་་་ཅྭ་ ་རྨྱུ་ཐོ་་41 1་་ཟི་་་་་་་ཛི་་ཛི་་ཁྱེ་་་་་་་་༦1 1་་་་་་་་་་་་ཁྱི་་་་་་་་་་1་།་་་་་་་་་་་་་་་་་་་་་་་1 1་་་་་་་་་་་་ཁྲི་་ཁུ་་ཆི1་1་་་་་་་་་་་ཙུ་་་ཙོ་་་་་ཅོ1 1་༷་་་་ཁྱེ་་་་་་་་་་་1་1་་་་་་་ཁུ་་་་་ཟློ་་་་་1 1་་ཁྱེ་་ཡི་ཙྭཁྱི་་ཇོ་་་་་་་་་་་་་1 1་་་་ཚོ་བྱུ་་་་རྭ་ ཙོ་་་་་རྨྱུ1 1་་་་་་་་་་་་་་ཁུ་་་་་ཁུ1 1རྨྱུ༷་ ་་་་ཅོ་་་་་་་་་་ཟི་་་ཁྱི1 1ཚོ་་་་་ཅིབྱོ་་སྩོ་་སྤྲི་ ་་1 1་་་་་ཐོཚི་་ཁྱི་་་་་སློ་་་་་ཉུ་་་་་་སྤྱོ་1 1་་་ཁྱེ་་ཁྱི་་་་་ཐོཁྱེ་་་་་་་། 1ཚོ་་་་ཅི་་་་ཐོ་ཁྱི་་་་་ཐོབྱུ1 1་༦་ཉུ་་་་གོ་ ་་་་་་་བྱུ་་་་་ 1་་་ཁྱི་་་་་་་་རྐྱུ་་ 1་་་་སྣྲི་་་་ཁྲི་་་་ཁྲི་་་་་་་་་སློ་་་་་༷1 14་་་་་་་་་་་སློ་་་1གོ་་་་་་་་ཁུ་་་་་་་་་་་མོ1་1་་་ཟི་་་་ཙྭ་་ཁྲོ་རྭ་་་་་་་སྟུ་གོ 1་་་ཉུ་་ཁྱི་གོ་་་་་་་་་་་་་་། 1་་་་སྤྲིཚོ་ཁྱེ་་་་་་་་་སྤྲི1 1བྱུསྐྱི་་་་་་ཉུ་་སྤྲེ་་ཁྱི་་་་་1་་བྱི་་སྦྱི་་་་་་་་་་་་་་ཅི། 1ལྗོ་་་་་་རྙི་་་་གྱིཔྱོ་་་་་ཚོ་་་་་་་སྒྲོ 1རྨྱུ་་་སྟུ་་་་་་་་་་ཁུ་ཕྱུ་1 
1ཟློ་་་་་ ཛི 1་་་་ཇོ་་་ 1་་་་ཁྱི་་་་ 1་་་་་་་་་་ 1་་་་་་་་་་་ཁྱེ་་་་་་་་་་་ཁུ་་་་། 1་་་་་་་་ཇོ་་་་༦་་་ཛི་་་་་ཛི་་་་ཛི1 ་་་ཁྱི་་་་་་་་་་་་ཛི་་་ཅྭ་ ་་ཚི1སྦྱི་་་་་་་་་་ ་་་་་་་་་་་་་་་་་་ཅྭ་ ་་་་༷་ ་་་་་སྐྲུ་་་་བྱུབྱུ་་་་་་་་1་་་་ཁུ་་་་་་་་་་་སྐྱུ་གོ4་་་་གོ་སྒྲི་་་་་་་་་ཁྲི་་་་་་་་་་་་་1་ཅྭ་ ་་་་་་1 1་་་ཁུ་་་་་སྩོ་་་་་་་ཟློ་་་1་1་་་རྭ་ ་་་་་་་་་་་་གོ་་་་་་གློ 1་་་་ཁྱི་ཕྱུ་་་་་་་་་་་་་་་ཉུ་་་་་་་གློ་1་་་ཆོཁྱེ་་་་་ཙྭ་ཁྱི་་་་་སྐྱུ་་་1 14་་ཅོ་་་་་་་་་་སྩོ་་་་་་་་་་སློ1་་་་ཁྱི་་་་་་་་་་་་་་ཡི་་་་ 1་་་་་༷་ ་1 1་་་་་་་ཁྱི་་་་་་་༦་་རྨྱི་1 1་་་༷་ ་་༦་་ཁུ་་་་་་་སྟུ་་1་1་གྱོ་་་་་་་་ཟློ་་་་་་་་་1 1་་་་་་༷་་་་་་ཁྱེ་་་་་་ཁྲི1 །་་་སྤྲི་་་་་་་ཐོབྱོ་་་སྟུ་་༷་1་1་་་་་་་་་་་་་་་་་རྭ་་་རྭ་ ༦1 ་ 1་་་་་་་་་་་་་་་་སྦྱོ་་་ཁྱེ་་་་1 1་་་་་་སྨོ་་་་་༦་་་་་་་་1 1་་་་་ཚི་རྩི་་ཕྱུ ་ཏོ་ཁྲུ་་སྣྲི1་།་བྱུ་་ཟི་ཁྱི་ ་ཁྲུཉུསྤྲོ་ཉུ1 1་་་་ ་སྦྱི་་ཉུརྗོ་སྣྲོ་་་་སྦྱོ་་་་་སྐྱུསྒྲོ1་1་་་་་་་་་་སྒྲོ་་་་་་ཙོ་་ཅྭ་ 
་རྨྱེ་་ཟློ་ཟློ་་་་སྦྱི་་་ཛི་་་་་་་་་་་་་་་ཁུ1 1་ཁུ་་་་་ཁྱི་་ཙོ་་༦་་་ཁྲུ་་་1 1་ཁུ་་ཁྲུ་ཛི་་་སྟུ་་་་་་་་1 1་ཙོ་ཁུ་་ཁུ་་ཁྱེ་་་་་་་་་་1 1་་་་་་་་་་་་་་་་ཁྱེ་་་་1 1་་་་་་་་་་་་་་་་1 1་་་ཁྲི་་་་་་་སྔོ་་་གྱོ་་་་་སྟུ་། །་་་་་་་་་་་་་་་ཁུ་་་་་་་་་་་་་་་་་་་་་་་་་་་་སྟུ4 1་་་་་་་་ཁྱི་་་་་་་ཁུ་་ཉུ་་སློ་་་་་་1 1བྱུ་་་་གོ་་་་་་་་་་ཉུ་་མི་་་་ཁུ་་བྱུ་་ཚོ་་་རྭསྟུ་་་་་སྩོ4་་1་་་་་་༷་ སྟུ་་་་་་་་་་ཁྱི་་་་་་་་ཟློ་་་྅་ཙྭཁྱི་༷་་་ཁྲི་་་་་་་་་་་་་་་་ ་རྨྱུསྟུཉུ་ ་་་1་1་་་་་་་་་་་་་་་་་་་་་་་་་་་་་་སྩོ4 1་་་་་་་་སྦྱ་་་་ཙྭ་་་་་་་་་ཁྱེ་་་་་་་་་་་་ །་་་་་་་་་་་་་་་སྩོ་་་་་་་་་་་་་སྒྲོ4 རྭ1 ་་་་་་་་་ཟློ་་་་་་་་་ཁྱེ་་་་་་་་་ཟློསྟུ་ཉུ་་སྒྲོ་་ཟི་་་ཙྭ་་་་སྨོ།་་་་་སྩི་་སྟུ་་་་་་་་་་ཁུ་་་སྔོ་་སྟོ་་་་་སློ །་་་ཁྱི་་་་་་་་་་་་་་ཟློ་་ཉུ་་་་་་ཏོ1་1་་་་་་ཁྱེ་་་་་་སྣྲོ་་་་་་སྤྱོ་་1 1་་་་་་་་་ཁུ་1 1་་་་་་་་་་་་་་་་་་ཛི་་་་་་ཁྱེ་ཉུ་་རྨྱོ་་སྩོ1 1སྒྲོ་་སྨྲོ་་སྤྲོ་་ནྟྲ་་ལྗོ་་་ཁྱི 1ཁྱི་་་་་་་་་སྐྱི1 1་་་་ལྗོ་་་ཁྱི་ཉུ༷་་སྤྲི་་་་རྨྱི་ཛི1སྙོ་་སྐྱི་ཙོཁྱེསྦྱི་ 1་བྲེ་་་་་་་་ཁྱི ཁྲེ་་ཅོ་་་་་་་1 ་
