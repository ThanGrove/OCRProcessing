།མ་ཧྭ་དེ་བ་ལྷ་ཡི་༄༅། །གདོན། །བདེན་པའི་སྟོབས་ཀྱིས་བཟློག་ཏུ་གསོལ། །ཀ་ལ་ཡསྒེ་མུ་སྟེགས་གདོན། ཁྲོས་པའི་སྟོབས་ཀྱིས་བཟློག་ཏུ་གསོལ། །སྨྱོ་སྨྱོ་སྨྱོ། རྦད་པའི་ལེའུ་སྟེ་དྲུག་པའོ།། །།ཧཱུཾ༷། འཇིགས་ཚུལ་རྔམ་པའི་ཆ་བྱད་ཅན། །སྐུ་ནི་རི་རབ་ཁྱོན་དང་མཉམ། །སྤྱན་ནི་བར་སྣང་ཁམས་སུ་གཟིགས། །ལྗགས་ནི་སྤྲུལ་པ་གློག་ལྟར་འཁྲུག །དབུ་སྐྲ་ཐོར་ཅོག་ཟུར་བུར་འཁྱིལ། །ཐོད་པ་སྐམ་རློན་རྒྱན་དུ་བྱས། །སྦྲུལ་ནག་འབར་བས་རལ་པ་བཅིངས། །སྐུ་ལ་རཀྟའི་ཚོམ་བུས་བརྒྱན། ཁྲག་དང་ཞིག་གི་ཐིག་ལེ་ཅན། །དབུ་དང་ཕྱག་གཉིས་སྤྲུལ་པའི་ཚེ། །སྤྲུལ་པའི་ཁྲོ་བོས་སྟོང་ཁམས་ཁྱབ། །གསུང་གི་སྤྲུལ་པ་ཧ་ཧའི་སྒྲས། །གསུང་གི་ཁྲོ་བོས་སྟོང་གསུམ་གང་། །ཐུགས་ནི་མཉམ་པའི་ངང་ལ་བཞག །དོན་དམ་དབྱིངས་ཀྱི་ལྷ་རུ་གསལ། །དེ་ལ་ཡོན་ཏན་བསམ་མི་ཁྱབ། །འཕྲིན་ལས་བཅོལ་བའི་དོན་ཀུན་འགྲུབ། །སྐུ་ཡི་རབ་ཏུ་རྟགས་བསྟན་པ། །སྐུ་ཡི་སྤྲུལ་པས་ཚངས་པ་འདུལ། །སྤྱན་ནི་ཉི་མའི་འོད་ཟེར་འདྲ། །ཉི་ཟླ་གཟའ་སྐར་འདུལ་བའི་རྟགས། ཁྲངས་ནས་ཚུབ་མས་སྟོང་གསུམ་གང་། །དགྲ་བགེགས་རླུ་་ལ་བསྐུར་བའི་རྟགས། །ཚེམས་ནི་གཅོད་པའི་སྟ་གྲི་སྟེ། །བདུད་དཔུང་ཐམས་ཅད་གཅོད་པའི་རྟགས། །ལྗག་ནི་འཁྱུག་པའི་གློག་ཡིན་ཏེ། །སྲིད་གསུམ་རྨང་ནས་འཇིག་པའི་རྟགས། །སྐུ་ནི་རབ་ཏུ་འཇིགས་པའི་ཚུལ། 
།ཀླུ་ཕྲན་དབང་དུ་སྡུད་པའི་རྟགས། །གསུང་ནི་ཚངས་པའི་སྒྲ་ལྟར་སྒྲོགས། །ཆོས་སྒྲ་ཕྱོགས་བཅུར་སྒྲོགས་པའི་རྟགས། །ཐུགས་ནི་མི་རྟོགས་དང་ལ་བཞག །བྱང་ཆུབ་སེམས་དང་ལྡན་པའི་རྟགས། །གསུས་པ་རབ་ཏུ་འཁྱིལ་བ་ནི། །སྲིད་གསུམ་ལྷོ་ན་གནས་པའི་རྟགས། །ཞབས་གཉིས་དོར་སྟབས་བགྲད་པ་ནི། །བདུད་བཞི་རབ་ཏུ་གནོན་པའི་རྟགས། །དགྲ་བོ་བཟློག་པའི་ཧཱུཾ༷་སྒྲ་ཅན། །བགེགས་ཀུན་བཟློག་པའི་ཕཊ་སྒྲ་ཅན། །བྱད་ཁ་བཟློག་པའི་སྨྱོ་སྒྲ་ཅན། །འབྱུང་པོ་འདུལ་བའི་སྒྲ་སྒྲོགས་ཤིང་། །དགྲ་བགེགས་གསོད་པའི་སྒྲ་སྒྲོགས་པ། །མ་རུངས་པ་ལ་རྒྱོབ་རྒྱོབ་སྒྲོགས། །ཧཱུཾ་ཧཱུཾ༷་ཕཊ་ཕཊ་སྨྱོ་སྨྱོ་དང་། །དུལ་དུལ་ཧེ་ཧེ་སྒྲ་སྒྲོགས་པ། །བཟློག་བསྒྱུར་བྱེད་པའི་མཐུ་བོ་ཆེ། །ཁྲམ་ཁ་བསྒྱུར་བའི་སྟོབས་པོ་ཆེ། །བདུད་ཁ་བསྒྱུར་བའི་བརྗོད་པོ་ཆེ། །ཤོད་ཁ་ཆེན་པོ་ད་ཕྱེ་ཞིག །རྩིས་ཁ་ཆེན་པོ་ད་བསྒྱུར་ཅིག །དོ་ནུབ་སྲིད་པའི་ཁྲོམ་སྣ་བསྒྱུར། །དོ་ནུབ་ལྷ་འབྲེའི་དམག་སྣ་བསྒྱུར། །ཕས་ཀྱི་ཕུར་པ་འཕེན་པ་བཟློག །ཚེ་ཟོར་ཉུངས་ཟོར་འཕེན་བཟློག །གཤིན་རྗེའི་འཁོར་ལོ་བསྐོས་བ་བཟློག །བདུད་ཀྱི་ཁྲ་ལ་འདེབས་པ་བཟློག །མ་མོའི་ཁྲག་ཟོར་འཕེན་པ་བཟློག །ཀླུ་ཡི་བྱང་དུ་བཅུག་པ་བཟློག །ཐེའུ་རང་ཚེ་ལ་རྐུ་པ་བཟློག །འགོང་པོ་སྲིལ་འཇོག་པ་བཟློག །མུ་སྟེགས་པར་དུ་བྱེད་པ་བཟློག །ལྟས་ངན་བརྒྱ་བཅུ་རྩ་བཞི་བཟློག །མོ་ངན་སོགས་སྡིག་བྱེད་པ་བཟློག །རྨི་ལམ་ངན་པ་ཐམས་
