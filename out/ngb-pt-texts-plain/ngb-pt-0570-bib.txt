༄༅། །རྒྱ་གར་སྐད་དུ། ཨ་ཨ་ཨུ༷། བོད་སྐད་དུ། སྟོང་པའི་སྒྲ། བཅོམ་ལྡན་འདས་ཀུན་སྣང་འདུས་མ་བྱས་ལ་ཕྱག་འཚལ་ལོ། །འདི་སྐད་ཅེས་སྣང་བའི་དུས་ན་ཀུན་ཁྱབ་བརྡལ་བ་ཆེན་པོའི་ཞིང་ཁམས་སུ།སྤྲུལ་པའི་སྐུ་ཉིད་དུ་ཤར་ཏེ། ཐབས་ཤེས་གཉིས་སུ་མེད་པའི་ཁྱེའུ་ཆུང་རྡོ་རྗེ་འཆང་གིས། འཕྲིན་ལས་རྣམ་པ་བཞིའི་འཁོར་ལོ་ལ་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། །ཨེ་མ། ངོ་བོ་ཆོས་རྣམས་ངང་གིས་སྟོང་། །ལྟ་བ་མཐའ་གྲོལ་ལ་ཟློས་ཅིག །རིན་ཆེན་ཆོས་རྣམས་ངང་གིས་གསལ། །སྒོམ་པ་ཐོད་རྒལ་ལ་ཟློས་ཅིག །སྤྱོད་པ་ཐུགས་རྗེ་རང་གྲོལ་ངང་། །སྐུ་གསུམ་རང་གསལ་ལ་ཟློས་ཅིག །སྣང་བ་ཕྱི་རུ་བཟོ་མ་བཅོས། །རིག་པ་ནང་དུ་རྩིས་མ་འདེབས། །སྣང་རིག་བར་ན་ཧྲིག་གེ་བ། །དབྱིངས་ཆེན་ངང་ན་ལམ་མེ་བ། །རྐྱེན་བྲལ་ཡུལ་ལ་ཁྱུག་གེ་བ། །མ་བཅོས་མ་ཡངས་ལ་ཟློས་ཅིག །ཚོགས་དྲུག་དབང་ཐོབ་གཞན་དོན་བྱེད། །ལུས་གློང་ངག་སྡོམ་སེམས་མི་འཕྲོ། །འདི་ཉིད་བསྟན་པའི་ཐོག་མའོ། །འཁོར་འདས་གཉིས་མེད་གདེངས་ཆོས་ཅིག ་༑་ལོ་ཀ་ནི་བཞི་པ་འདི། །འབྱུང་བཞིའི་སྒྲ་རུ་རང་བྱུང་བས། །འདི་དང་སུ་འཕྲད་གདོད་མར་གྲོལ། 
།ལྔ་༄༅། །ལྡན་བླ་མ་འདི་སྟོན་ཅིག །ས་མ་ཡ། སྔགས་ཀྱི་སྲུངས་མས་སྲུངས་ཅིག གཏེར་རྒྱ། སྦས་རྒྱ། ཟབ་རྒྱ། དམ་རྒྱའོ། །རྡོ་རྗེ་འཆང་དང་པོའི་དགོངས་པ་བསྟན་པའི་བུ་གཅིག་པོ་རྫོགས་སོ།། །།༄༅། །དྷ་ཧྭར་ཉྭ། ཧ་མ་ཊ་རྟ། རྒྱ་གར་སྐད་དུ། སརྦ་དྷརྨ་སྙེ། བོད་སྐད་དུ། ཐམས་ཅད་ཀྱི་གནད། རང་ཤར་གྱིས་ལྷ་བཞི་སྣང་བརྡལ་ལ་ཕྱག་འཚལ་ལོ། །སེམས་ཅན་གྱི་ཚེ་ལོ་དཔག་ཏུ་མེད་པའི་དུས་ན།
