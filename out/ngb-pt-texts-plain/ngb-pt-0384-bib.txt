གཟུངས་འདོན་ཅིང་སྨོན་ལམ་གདབ་པའི་ལེའུ་སྟེ་བཅོ་༄༅། །བརྒྱད་པའོ།། །།དེ་ནས་འཁོར་ལྷ་མ་སྲིན་རྣམས་ཀྱིས་བཅོམ་ལྡན་འདས་ཕྱག་ན་རྡོ་རྗེ་དང་། །རྡོ་རྗེ་སེམས་དཔའ་ལ་བསྟོད་པ། ཧུ༷་རྡོ་རྗེ་སེམས་དཔའ་སེམས་དཔའ་ཆེ། །རྡོ་རྗེ་ཡེ་ཤེས་ཆེན་པོའི་མཆོག །ཕྱག་ན་རྡོ་རྗེ་མཐུ་སྟོབས་བདག །གསང་བའི་བདག་པོ་རྡོ་རྗེ་ཅན། །ཡ་མཚན་རྨད་བྱུང་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །གྲི་གུག་གི་རྒྱུད་ལས། འཁོར་གྱིས་གཙོ་བོ་ལ་བསྟོད་པའི་ལེའུ་སྟེ་བཅུ་དགུ་པའོ།། །།དེ་ནས་ཡང་འཁོར་གྱིས་གཙོ་བོ་ལ་བསྟོད་པ། བསྐལ་པ་མེ་ལྟར་འབར་བ་ཡི། །དུར་ཁྲོད་ཆེན་པོའི་ཀློང་དཀྱིལ་ནས། །གདུག་པར་གྱུར་པ་ཐམས་ཅད་ཀུན། །འདུལ་བྱེད་མགོན་པོ་ནག་པོ་སྟེ། །སྐུ་ནི་མཐིང་ནག་ཆེན་པོ་ལ། །དཔའ་བོ་སྟག་གི་སླག་པ་ཅན། །བདུད་ཀྱི་བེར་ནག་སྐུ་ལ་གསོལ། །ཐལ་བའི་མཚན་མས་ལུས་བྱུགས་པ། །གཡས་ནག་གཅོད་པའི་གྲི་གུག་བསྣམས། །གཡོན་པ་བྷརྙའི་ཁྲག་ཆེན་གསོལ། །ཞལ་ནས་ཁྲག་དང་ཞག་ཀྱང་འཛག །མཆེ་བ་ཟླ་བ་ཆོས་པའི་ཚུལ། །སྤྱན་གསུམ་སྡང་མིག་གློག་ལྟར་འགྱུ། །རལ་པ་ཁམ་ནག་གྱེན་དུ་འཁྱིལ། །ཁྲོ་གཉེར་བསྡུས་པའི་གཟི་བརྗིད་ཅན། །བྷསྟོའི་ཕྲེང་བས་དོ་ཤལ་བྱས། །གསུས་ཁྱིམས་ཆེ་ཞིང་གསུས་པ་འཕྱང་། །མི་ཐུད་ཁྲོ་གཉེར་དྲག་པོ་ཡིས། །མཆུ་དང་སོ་ལ་ཁྲག་ཀྱང་འཛག །མི་རོ་སྐྲ་ཡི་སྐུད་པས་དཀྲིས། །བམཆེནསྟེངནབཙན་དཀྱིལ་གནས།

།།གསང་བའི་བདག་པོའི་རྒྱུད་འདི་འཇིག་རྟེན་གྱི་ཁམས་༄༅། །སུ་གྲགས་པར་སྟོན་ཅིག །རྒྱུད་འདི་རྣལ་འབྱོར་པ་རྣམས་ཀྱི་སྲོག་ཡིན་ནོ། རྒྱུད་འདི་ནི་ཡུལ་དྷ་ན་ཀོ་ཤའི་གླིང་མཁའ་འགྲོ་མ་མང་པོ་ཚོགས་པའི་གནས་སུ་བབ་པར་གྱུར་ཅིང་། ཕྱག་ན་རྡོ་རྗེ་རི་རབ་རིའི་རྩེ་མོ་ན་འཁོར་ལྷ་སྲིན་སྡེ་བརྒྱད་ལ་བཤད་པའོ། །དེ་ནས་གྱད་ཀྱི་སྐུ་མཆོག་ཆེན་པོས་མ་ཧྭ་བོ་དྷིའི་མཆོད་རྟེན་གྱི་རྩར་མ་བདུན་ལ་བཤད། དེ་ནས་ལྷའི་གཙོ་བོ་རྣམས་ཀྱིས་རྡོ་རྗེ་གདན་དུ་བདུད་བརྒྱད་ལ་བཤད་དོ། །དེ་ནས་རྒྱ་གར་གྱི་ཡུལ་གསེར་མིག་བྱ་མཆུ་ཅན་དུ་སྦས་སོ། །དེ་ནས་རྒྱ་གར་གྱི་སློན་དཔོན་གྲུབ་པ་ཅན་རྣམས་ཀྱིས་རྙེད་དེ། རིམ་གྱིས་བརྒྱུད་པ་ལགས་སོ། །མགོན་པོ་ནག་པོ་མན་ངག་བརྩེགས་པའི་རྒྱུད་ལས། །བརྒྱུད་པའི་ལེའུ་སྟེ་ཉི་ཤུ་གཅིག་པའོ།། །།སྔོན་གྱི་ལས་དང་སྨོན་ལམ་ད་ལྟའི་སྐལ་བར་ལྡན། །སྐྱེས་མཆོག་དེ་ལྟ་བུ་དང་རྒྱུད་འདི་འཕྲད་པར་ཤོག །མན་ངག་གྲི་གུག་བརྩེགས་པའི་རྒྱུད་རྫོགས་སོ།། །།བླ་མ་དམ་པ་མི་བསྐྱོད་རྡོ་རྗེ་ལ་གནང་བའོ། །རྗེ་བཙུན་ཤཱརྙི་བྷ་དྲས་ཞུས་པ་ལས། དགེ་སློང་ཚུལ་ཁྲིམས་རྒྱལ་བས་བསྒྱུར་བའོ།། །།༄༅། །རྒྱ་གར་སྐད་དུ། ཤི་ཀྭ་ལ་མ་ཧྭ་ཀྭ་ཡ་བཱཀ་ཙིཉི་ཀྲོ་དྷ་ཏན་ཏྲ་ནཱ་མ། བོད་སྐད་དུ། དཔལ་མགོན་པོ་ནག་པོ་ཆེན་པོ་སྐུ་གསུང་ཐུགས་ཁྲོས་པའི་རྒྱུད་ཅེས་བྱ་བ། ཁྲག་འཐུང་འདུས་པའི་གཙོ་བོ་བུརྙ་ཧེ་རུ་ཀ་ལ་
