རག་ཏ་༄༅། །ཚལ་པ་དུ་སྙིང་དུ་ར་དུ། མ་ཡ་མ་རྭ་ཛ། ནཱ་ག་ལེ་ནཱ་ག་ལེ། དགྲ་བོ་ཡི་སྙིང་དུ་ཡ་ནི། །དུ་བ༷། བ་དུ་ཧུ༷་ཕཊ། ཅེས་ནི་གདབ་ཅིང་མནན་པར་བྱ། །ས་ཁམས་ཁྲམ་ལ་གདབ་པའི་སྐབས་ཏེ་བདུན་པའོ།། །ཁེརྒོང་ཁམས་ཁྲམ་ལ་གདབ་པ་ནི། །བྱང་དུ་བལྟས་པའི་ཆུ་མིག་ནས། །དེ་ནས་འདོམ་གང་གཅལ་བ་རུ།བ་རུ་དམར་པོའི་ལིང་ག་གཞུག །སྤྱི་གཙར་བཅུག་ལ་སྔགས་འདིས་གཏད། །ཨོ་ཡ་ཧུ༷་ཡ། ན་ག་ལེ་ནཱ་ག་ལེ། ར་དུ་བ༷་ས་ཀུ་མ་ལ། ག་ཧ་ཡ་ནི། དུ་ཀྲ་བ༷་དུ། དུ་པ་ཡ་ནི། ར་དུ་བ༷་པ་ཀྲ་ཧྭ་ཡ་ནི། དུ་ཡ་དུ་ཀྲ་ཡ་ནི། རུ་ལུ་བ༷་དུ། དུ་པ་ཡ་ནི། ཚིག་ཏུ་འདི་དག་བརྗོད་པར་བྱ། །ན་མོ་དཀོན་མཆོག་གསུམ་ལ་ཕྱག་འཚལ་ལོ། །ཀླུ་འཇོག་པོའི་རིགས་རྣམས་ཀྱི་དགྲ་བོ་འདིའི་སྙིང་ནས་དྲོངས་ཤིག །ནཱ་ག་རྭ་ཛ་ས་མ་ཡ། ནི་རག་ཤ་ནི་ལ་གྲ་ཧ་ཡ་དུ། བ༷་ནི་བ༷་བ་ནི། སན་ཙ་ཡ། མེ་ནན། བ༷་དུ། པྲ་ཡ་ནན། མུ་ར་རག་ཤ །མེ་སན་ཙ་ཡ་ནི། ཨ་ལོ་བ༷་པ་ཀྲ་གྲྭཧྭ། ཞེས་བརྗོད་ལ་གཏད་དོ། །ཤིང་ཁམས་ཁྲམ་ལ་དབབ་པའི་སྐབས་ཏེ་བརྒྱད་པའོ།། །།ལྕགས་ཁམས་གཏད་པའི་ལས་བྱ་ན། །ཆུ་མིག་བྱང་བལྟས་ཁར་ཕྱིན་ཏེ། །འདོམ་བཞི་ཕྱིར་འགྱེད་ས་རུ་གཏད། །ཧི་མི་ལེ་བེ་བེ་བེ་དུ་ཀྲ་ཡ། ཁྲ་ར་ནཱ་ག་ཡ། ཧ་ར་ན་ག་ཡ་གསན་ཙ་ནི། རག་ཏ་བ༷་པ། ཡ་ཏྲ་ཏ་ཡ་ནི། བཱ་ར་ཡ། དུ་པ་ལི་པ༷་བ་ཡ། དུ་ལིང་སན་ཙ་ཡ། ནི་ལུ་ཡ། ཡ་བཱ་ར་ཡ། ཨེ་མ་ནི་ར་དུ་ཛ། 
ཕ་མེས་མགོ་རབས་ཡང་ན་བསྟོད། །དེ་ལ་ངན་སྔགས་སྦྱོར་བ་བརྡེག །ནི་སན་ཙ་ཡ། མ་རག་ཏ། དུ་ལིང་སྙིང་མ་ར་ཡ། དུ་བ༷་བ་ཤ་སྙིང་མྱ་ར་ཡ། ཀུ་མ་རཀྟ་མེ་ཛུ་ཛ། ལྕགས་ཁམས་ཁྲམ་ལ་གདབ་པའི་སྐབས་ཏེ་དགུ་པའོ།། །།མེ་ཁམས་གཏད་པའི་ལས་བྱ་ན། །ཆུ་མིག་ཁ་ནི་ལྷོ་ར་བལྟས་པའི། །གོན་ནས་འདོམ་གསུམ་གཅལ་བར་གཏད། །ཐོད་པ་མེད་ན་བསེ་བྱང་ལ། །མིང་དང་རུས་སུ་བཅས་པ་གཏད། །དཀོན་མཆོག་གསུམ་ལ་ཕྱག་འཚལ་ནས། །ངན་སྔགས་འདི་དག་བརྡེག་པར་བྱ། །མུ་ར་ཡ་ནི་དུ་ཡ་ནི་མུ་ར་ཡ་ནི། དུ་ཡ་ནི་མྱ་ར་ཡ་དུ་ཀྲ། མྱ་ར་ཡ་ས་སྙིང་། མ་མ་རག་ཏ་སྨྱོ། ནཱ་ག་ལེ་ནཱ་ག་ལེ། དུ་ཡ་དུ་ཡ། ཧུ་ཡ་ཧུ་ཡ། ཏི་ཧུ་་་་་་ཕ་མེས་ཐོད་པ་ཡང་ན་ཁྱི། །དེ་ན་ནི་ངན་སྔགས་བརྡེག་པར་བྱ། ཨེ་ཨེ་རི་ཤི་ཤི །མང་ས་སན་ཙ་ཡ། ནི་རག་ཤ་ཏ་ཡ་སྙིང་དུ་ཡ་ཀྲ་ཧ་ཡ་ནི། དུ་ཡ་སན་ཙ་ཡ། ཞེས་པས་བརྡེག་པར་བྱ་བ་ཡིན། །མེ་ཁམས་ཁྲམ་ལ་གདབ་པའི་སྐབས་ཏེ་བཅུ་པའོ།། །།ཆུ་ཁམས་གཏད་པའི་ལས་བྱ་ན། །ཆུ་མིག་ཕྱོགས་བཅུར་བལྟས་པ་རུ། །གཉའ་ནས་འདོམ་གང་གཅལ་བ་རུ། །དཀོན་མཆོག་གསུམ་ལ་ཕྱག་འཚལ་ལ། །ངན་སྔགས་འདི་དག་བརྡེག་པར་བྱ། །ནཱ་ག་ལེ་རུ་ཏྲ་ཧུ༷་ཡ་ནི། རུ་ཏྲ་ཧུ་ཡ་དུ་ར་ཡ་སྲོ་ག་ལ། ཕྱིན་ཅི་མ་འོ་ག་སྲོ་ག་གི་ལུང་། །དམ་ཉམས་སྙིང་ཕྱུང་ཞལ་དུ་གསོལ། །མུ་ཧྭ་རག་ཏ་དུ་ཡ་ནི། ཡ་དུ་ཀྲ་ཧ་ཡ་ནི་སན་ས་ཡ་དུ་ནན། 
ཆུ་ཡི་༄༅། །ཁམས་ཀྱིས་མགོ་བོ་སྐྱོན། །བར་དུ་གླང་གི་མགོ་བཅུག་ལ། །བདུད་ཀྱི་སྔགས་ཀྱི་གདབ་བར་བྱ། །རག་ཏ་ནི་སན་ཙ་ཡ། མ་མ་ད༷་ད༷་དུ། ལི་སན་ཛ། ཆུ་ཁམས་ཁྲམ་ལ་གདབ་པའི་སྐབས་ཏེ་བཅུ་གཅིག་པའོ།། །།འཕྲིན་ལས་ཁམས་ནི་ཤིང་བུ་ལ། ཁྲམ་དཀར་ཁྲམ་ནག་དང་པོ་སྟེ། ཁྲམ་དམར་ཁྲམ་ནག་གཉིས་པའོ། ཁྲམ་སྔོན་ཁྲམ་ནག་གསུམ་པ་ཡིན། །མགོ་ཅན་ཉིད་ལ་རྫོགས་ཀྱང་རབ། །ཧུ༷་བཛྲ་ནི་དུ་དུ་དུ་ནི་དུ་ནི་དུ་ཤི་ཁྲེ་ཁྲེ་ཁྲེ་ཤི །ནཱ་ར་ཡ་དུང་དུ། རྡོ་རྗེ་ཁྲམ་འདེབས་མའོ། །རུ་ཀ་ནི་དུ་དུ་དུ་ནེ་དུ་ནེ་ཁྲ༷་ཁྲ༷་ཤི་མྱ་ར་ཡ་དུང་དུ། དྲང་སྲོང་ཁྲམ་འདེབས་མའོ། ཁུ་ཡ་ནི་དུ་དུ་དུ་ནེ་དུ་ནེ་ཁྲ༷་ཁྲ༷་ཤི་ནཱ་ར་ཡ་དུང་དུ། གཤིན་རྗེའི་ཁྲམ་འདེབས་མའོ། །རག་ཤ་ནི་ཡ་དུང་དུ་དུ་དུ་ནེ་དུ་ནེ་ཁྲ༷་ཁྲ༷་ཤི་མུ་ར་ཡ། སྲིན་མོ་ཁྲ༷་འདེབས་མའོ། །ནཱ་ག་ནི་དུང་དུ་དུ་ནེ་དུ་ནེ་ཁྲ༷་ཁྲ༷་ཤི་ནཱ་ར་ཡ་དུང་དུ། ཀླུའི་ཁྲམ་འདེབས་མའོ། །རིའི་ནི་དུང་དུ་དུ་ནེ་དུ་ནེ་ཁྲ༷་ཁྲ༷་ཤི་མྱ་ར་ཡ་དུང་དུ། མ་མོ་ཁྲམ་འདེབས་མའོ། །འདི་དག་ནི་ཁྲམ་ཤིང་རེ་རེ་ལའོ། །འཕྲིན་ལས་ཀྱི་བདག་མོ་དྲུག་གིས་ཁྲམ་ལ་གདབ་པའི་སྐབས་ཏེ་བཅུ་གཉིས་པའོ།། །།ས་བདག་ལྟོ་འཕྱེ་ནག་པོ་སྟེ། །ཉི་མ་འདུན་འགྲོའི་རིམ་པ་བཞིན། །རྒྱབ་ཀྱི་གཡས་སུ་འཁོར་བར་བཤད། །ས་བདག་ལྷོ་འཔྱེ་སྐྱེས་བུ་ཡི། །ལྟོ་བར་གཏད་ན་ཐམས་ཅད་འགྲུབ། །འོག་མ་དང་པོ་བསྐོས་པ་ན། །ཕ་དང་བུ་དང་ཆུང་མ་འཆི།

མཐའ་ཡས་འཇོག་པོ་ཀར་ཀོ་ཏ། །བ་སུ་གི་དང་སྟོབས་ཀྱི་རྒྱུ། །ནོར་རྒྱས་དུང་སྐྱོང་བ་རུ་སྟེ། །ཕོ་ནག་པོ་ནག་པོ་སྲོག་ལ་ཡ། ཐ་ཐ་བུར་བུར། དགྲ་བོ་མཐར་བརླག །དྲུང་བུར་དྲུང་བུར། དགྲ་བོ་གྱེན་ཤ་སྟམ་པ་ཡ། མཐའ་ཡས་དགྲ་བོ་རྩ་བ་བདུན་རྒྱུད་ལ་ཡ། ཀླུ་ཡི་དམོད་པ་ནག་པོ་ཡིན། གུ་རུ་པདྨ་འབྱུང་གནས་ཀྱིས་ཀླུའི་དམོད་པ་བསྡེབས་པའི་སྐབས་ཏེ་བཅུ་དྲུག་པའོ།། །།དགྲ་གཟུགས་བྲི་ཞིང་བཅའ་བ་ནི། ཁ་མིག་ཡོན་པོར་དོད་པར་བྱ། །ལག་པ་བྲང་ལ་ཞ་བར་བྱ། །གཡས་དང་གཡོན་ན་སྦྲུལ་གྱིས་འཁྱུད། །སྦྲུལ་མགོ་སྤྱི་བོར་གཟུགས་པར་བྱ། །བུ་རམ་ཀ་ར་འདམ་བུ་དང་། །མངར་སྒྱུར་རོ་རྣམས་ཐམས་ཅད་ལ། །བད་ཀར་ར་དཀར་འོ་མས་སྦྱར། །དགྲ་བོའི་མིང་རུས་གཟུགས་ལ་བྱུག །ཀླུ་དང་དགྲ་གཟུགས་ལྟོ་སྦྱར་ལ། །ཚོན་སྐུད་སྣ་ལྔས་བྱང་དུ་གཞུག །གོང་མའི་སྔགས་དང་བཅིངས་པའི་སྔགས། །མགོ་བོར་ཚེར་མི་འཕུར་པས་གདབ། །ཕུར་པ་ཀེ་རེ་སླར་མི་འབྱུང་། །རིགས་བཞི་གནས་པའི་ས་རུ་ནི། ཁྲུ་གང་བརྐོས་བའི་ས་རུ་བྱ། །ཀླུ་གཏོར་དྲི་དང་ལྡན་པ་བསྔོ། །སྔགས་དང་བཅོལ་བ་ལས་བཞིན་བྱ། །རྒྱུན་དུ་ཕྱོགས་དེར་ཆུ་གཏོར་བཏང་། །ཀླུ་ཡི་མིང་དང་གཟུགས་བྱས་ལ། །དགྲ་གཟུགས་གོང་མ་འདྲ་བ་ལ། །གོང་གི་ཐབས་དང་ལས་བཞིན་བྱ། །མང་པོས་བྱས་ན་གདོན་མི་ཟ། །ཧུ༷་ཧུ༷་བ་བ་ནི་ར་ཛུ་ཛུ་ལྭ༷་ཡ་ཞི་ཞི༷་པྲ་ལ་ཤེར་ཤེར་ཐུ༷་ཐུ༷་སོད་སོད། རུ་ལུ་ཤཱ་ག་ཧུ༷་ཕཊ། 
དགྲ་བོ་ལ་ཀླུ་༄༅། །ནད་ཐོང་། མཛེ་ཐོང་། ཡུངས་ཀར་བསྔགས་ལ་གཟུགས་ལ་བརྡེག །ཀ་མ་ཀ་མེད་བརླག །ནག་པོ་ཀ་མེད་བརླག །རུག་པ་ཀ་མེད་བརླག །གཟིར་ཅིང་མནན་པར་བྱའོ། །ཛ༷༷ཧུ༷་བ༷་ཧོ། དགྲ་བོ་འདི་སྤྲ་ཏ་ཧ་རྣུ་ཆུ༷། སྲོག་ལུས་དབང་པོ་ལྔ་ཆུམ་ཆུམ། ཀླུ་རྒྱལ་གདུག་པ་ཅན་གྱིས་ཆུམ་ཞེས་བཟླས་ཏེ་མནན་ནས། །རྡོ་རྗེ་རྒྱ་གྲམ་གྱིས་གཅད་དོ། །མནན་པའི་སྐབས་ཏེ་བཅུ་བདུན་པའོ།། །།དེ་ནས་ཀླུ་དགྲ་ལ་ལས་བྱེད་པའི་སྔགས་འདིའོ། །གམ་འཛིན་ཆེན་མོས་སྟ་ལ་ལོང་ལ་སོད་ཅིག །ལྕགས་སྲིན་ཆེན་པོས་སྟ་ལ་ལོང་ལ་སོད་ཅིག །གནུགས་སྲིན་ཆེན་པོས་སྟ་ལ་ལོང་ལ་སོད་ཅིག །དགྲ་བོ་འདི་སྟ་ལ་ལོང་ལ་སོད་ཅིག །ཀུ་རུ་སོད་ལ་ལ་སོད། ཡང་ཀུ་སོད། པུ་ཏི་སོད། ཆིན་ཀ་ར་སོད། ཁ་ཊ༷་ག་གུར་ཤ་ན་ཡ་ཕཊ། དགྲ་བོའི་སྲོག་ལུས་རོལ་བར་མརྫོད། །དེ་ནས་ཀླུ་ལ་གཏད་དོ། །ཟླ་བ་ཉ་གསུམ་ལ་ཆུ་གཏོར་ཕྱོགས་སུ་གཏང་ངོ་། །སྲ་བྲུ་བའི་སྔགས་ནི། ཨོ༷་མར་མེར་ཐབས་ཐིབས་གནམ་ས་སྡོད་ཕྱེ། ཞེས་པས་བརྐོས་ལ་འཁོར་ལོས་མནན། འགོང་པོ་སྣ་མེད་ཡན་པའི་སྐབས་ཏེ་བཅོ་བརྒྱད་པའོ།། །།འཛམ་གླིང་ཀླུ་རྣམས་ཐམས་ཅད་ཀུན། །དབང་དུ་བསྡུ་ན་གང་གིས་བསྡུ། །ཟིལ་གྱིས་མནན་ན་གང་གིས་མནན། །ལས་ལ་ན་གཟུད་ན་གང་གིས་གཟུད། །དབང་དུ་རྟ་མགྲིན་རྒྱལ་པོས་བསྡུས། །ཟིལ་གྱིས་བྱ་ཁྱུང་ག་རུས་མནན།

།བྲམ་ཟེ་ལྐུག་པ་བྱ་བ་གནས། །བྲག་སྣ་འདོར་དང་བ་ཚ་ཅན། །རོ་སྐམས་མཚེ་ན་ཀླུ་བཙན་གནས། །ཆུ་མིག་གདོང་ནི་མིག་ཡིན་ནོ། །འདོམ་དོ་པ་ནི་སྙིང་ཁར་འདོད། །གསུམ་པ་ལྟེ་བ་ཡིན་པར་བཤད། །དོང་ནས་འདོམ་གང་ཁ་ཡིན་ནོ། །ཆུ་མིག་བརྟག་པའི་སྐབས་ཏེ་ཉེར་དགུ་པའོ།། །།ས་ཕྱོགས་དབེན་པ་གནོད་སྤྱིན་གནས། །གཟའ་སྦྱང་ཆེན་པོ་བསྒྲུབ་པར་བྱ། །འབྲས་ཆེན་རྩི་སྨན་འོ་མ་དང་། །ཏིལ་དང་ར་ཁྲག་རྩི་ཏས་མཆོད། །སྲོག་གི་སྙིང་པོ་འདི་དག་གོ །ཨོ་པ་ཤི་མ་གི་ཏུ་ཏུ་མ་ཏི་ཀི་ཛ་ལ་ཏེ་ས་མ་ཡ་ཛུ་ཛུ། རྟགས་རྣམས་བྱུང་ནས་གསུང་པ་ཙམ། ཁྱབ་འཇུག་ཆེན་པོ་ཀུན་ཏུ་རྒྱུག །འཇིགས་བྱེད་ཆེན་པོ་སྤྲུལ་པ་ནི། །ལུས་པོ་གཅིག་ལ་མགོ་བོ་དགུ །མགོ་ལུས་ཐམས་ཅད་མིག་གིས་ཁེངས། །སྐུ་མདོག་བརྗིད་པའི་སྨུག་ནག་ལ། །ཕྱག་གཡས་ཆུ་སྲིན་ཞགས་པ་ཐོགས། །གཡོན་པ་སྦྲུལ་གྱི་ཞགས་པ་ཐོགས། །ཕ་ནི་སྲིན་ལ་མ་ནི་ཀླུ། །ཀླུ་སྲིན་འཚོས་བའི་རྫུ་འཕྲུལ་ཅན། །མིང་བརྒྱ་པོ་ལ་མཁན་སྟོང་པོ། །དུས་ཚོད་ཚེས་གྲངས་ཀུན་གྱི་བདག །བར་སྣང་ཁམས་ན་གནས་པའི་ཚེ། །འཇིག་རྟེན་ཁམས་རྣམས་ཞབྲུག་པར་བྱེད། །ཉི་མ་ཟླ་བ་ཟས་སུ་ཟ། །རྒྱ་མཚོའི་གཏིང་ན་གནས་པའི་ཚེ། །ཀླུ་རྒྱལ་མ་ལུས་ཞབྲུག་ཅིང་བཀྲག །ཐོག་དང་སེར་བ་སྐར་མདའ་སྟོང་། །འཇིགས་བྱེད་རོག་ཏི་མི་བསྒྲུངས་པ། །དམ་རྫས་གཏོར་མ་འབུལ་ལགས་ཀྱིས། །སྡང་བའི་བགེགས་དང་དགྲ་བོ་ལ། 
།གློ་༄༅། །བུར་ནད་དང་ཀི་ཀང་ཕོབ། །དེ་སྐད་བསྐུལ་ལ་བསད་པ་བརྩམ། །ཨོ༷་ཁ་ཤ་བ་རི་ཐུན་རིལ། ཨོ་ཁ་བྲ་དུ་ནི་ལྐུགས། །སྣ་ལ་སེ་ལང་ཀྲི་ཏི་རྨུགས་རྦད་ཡ། གཟའི་ཁྲམ་ཤོད་བཏང་བའི་སྐབས་ཏེ་སུམ་ཅུ་པའོ།། །།དེ་ནས་ཡམ་ཤུད་དམར་པོ་ལ། །འབྲུ་སྐྱེམས་བཤོས་བུས་མཆོད། །ལུག་ཀྱང་བརྒྱན་དུ་བརྩལ་བྱས་ལ། །བཤོས་བུ་རི་རབ་འདྲ་བར་བརྟག །ཧ༷་ཤ༷་ཏྲི་དམ་བཙལ་ཏྲི་རག་མ་ས་མ་ཡ་ཛ་ཛ། དེས་ནི་སྲོག་བདག་མྱུར་དུ་འོང་། །ཀྱེ་འཇིག་རྟེན་འདས་དང་མ་འདས་མཚམས་ཤེད་ནས། །བདུད་དང་གཤིན་རྗེ་དེ་ན་གནས། །གཞལ་ཡས་དམར་པོའི་བསེ་མཁར་ནས། །མི་རེད་དམར་པོ་མདུང་ཙམ་ཅིག །ཆིབས་སུ་བྱ་ཁྱུང་སྨུག་པོ་ཆིབས། །ཕྱག་ན་རྒྱལ་མཚན་རུ་མཚོན་འཕྱར། །སྤྲུལ་པ་མི་དམར་རྟ་དམར་འགྱེད། །མངག་གཞུག་འགོང་པོ་དཔུན་དགུ་འགྱེད། །སྒྲོལ་གིང་ཆེན་པོར་དམ་བཅས་པ། །འབྲུའི་གཏོར་མ་ཆང་དང་ནི། །དམ་རྫས་སྤྱན་གཟིགས་སྣ་ཚོགས་འབུལ། །གསོལ་ལོ་ཡམ་ཤུད་དམར་པོ་མཆོད། །ཏྲི་ཁ་ཐུན་སྨྱོ། ཏྲི་ཁ་ཐུན་སྨྱོ། མུ་ར་ཡ་ཐིག་ཐིག་སྨྱོ། ཞེས་པས་རྦད་པའི་ལས་བྱའོ། །སྲོག་བདག་དམར་བོས་བསད་པའི་སྐབས་ཏེ་སོ་གཅིག་པའོ།། །།གཏད་ཁྲམ་གསང་བའི་ལས་ཅིག་ལ། །ཐོད་ཆེན་ནང་དུ་མིང་རུས་གཞུག །ཐལ་འདག་དག་གིས་ཁ་གཅད་ལ། །དྲག་པོའི་སྔགས་འདི་གདབ་པར་བྱ། ཁྲ་ལ་རྔ་མོ་ཆུམ། ཤ་ལ་རྔ་མོ་ཐུན། ཤ་ལ་
