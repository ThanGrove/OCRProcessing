དེ་ནི་༄༅། །ནོར་གྱི་ལྷ་བརྒྱད་དོ། །ལྷན་ཅིག་སྐྱེས་པའི་འདྲེ་དགུ་པོ། །དེ་ཡི་བཅོས་ཐབས་མ་ཤེས་ན། །བར་ཆོད་སེལ་བའི་དུས་མི་མཆི། །དེ་ལ་དང་པོ་བར་དང་ཐ་མ་གསུམ། །དེ་གསུམ་ངོ་སྤྲོད་མ་ཤེས་ན། །རྣམ་རྟོག་རང་གྲོལ་དུས་མི་མཆི། །ཀྱེ་རིགས་ཀྱི་བུ་ཁྱོད་ཉོན་ཅིག །གསང་བའི་སྐྱེ་དགུ་རྩད་གཅོད་ནི། །སྐྱེ་གཞིའི་རྩ་བ་མ་ཆོད་ན། །རྒྱ་མཚོའི་ལྦུ་བ་སུབ་པ་འདྲ། །སྐྱེ་དགུའི་རྩ་བ་མ་ཆོད་ན། །ཆུ་གཏིང་ཟླ་བར་རྡོ་བརྡེག་འདྲ། །དངོས་པོའི་སྐྱེ་རྒྱུད་མ་ཆོད་ན། །སྨྱོན་པའི་ལག་དུ་སྨན་འགྱེད་འདྲ། །གཞི་ལམ་སྐྱེ་རྒྱུད་གཅོད་པ་ལ། །དགའ་བོ་འཇོག་པོའི་ནོར་བུ་ལྟར། །དུག་ལ་མི་འཇིགས་ནོར་བུ་རྙེད་པ་ལྟར། །དངོས་པོ་སྤང་དང་སྦྱང་པ་དང་། །བསྒྱུར་དང་རང་སྣང་ཡིན་ཟེར་བ། །དེ་ཡི་འབྲས་བུ་སྒྲིབ་པ་ཡིན། །ཀྱེ་རིགས་ཀྱི་བུ་ཉོན་ཅིག །འགྱུས་ནི་རང་གྲེལ་ཟུང་འཇུག་མེད། །ལྷ་བརྒྱད་ངོ་སྤྲོད་དེ་ཡིན་ནོ། །འགྱུས་ན་མ་རིག་བདུད་དུ་འབྱུང་། །འགྱུ་བ་གཅིག་གི་རྩད་ཆོད་ན། །ཚིག་ཆོད་ངོ་སྤྲོད་དེ་ཡིན་ནོ། །གཅིག་པོ་རང་གྲོལ་ཤེས་ཙམ་ན། །ལྷ་འདྲེ་དབང་དུ་འདུས་པས་འབྱུང་པོ་ཐུལ། །མཁའ་འགྲོ་ཆོས་སྐྱོང་དབང་དུ་འདུས་པས་འབྱུང་པོ་ཐུལ། །ཉག་གཅིག་ཉམས་ལས་གྲོལ་བ་རྟོག་གེའི་ཡུལ་ལས་འདས། །མཚན་མ་སྣ་ཚོགས་རང་སར་གྲོལ་བས་ཐིག་ལེ་ཉག་གཅིག་རང་ལ་འཆར། །ལྷ་འདྲེ་གཉིས་མེད་ཤར་བས་དངོས་གྲུབ་བར་ཆོད་མེད། 
།ལྷ་ཡི་སྐྱེ་རྒྱུད་ཆོད་པས་ཟུག་རྔུ་དུས་གཅིག་ཆོམས། །འདྲེ་ཡི་སྐྱེ་རྒྱུད་ཆོད་པས་བར་གཅོད་དུས་གཅིག་ཞི། །བདེ་སྡུག་སྐྱེ་རྒྱུད་ཆོད་པས་འཁོར་བ་རང་སར་གྲོལ། །ཐིག་ལེ་རང་ལ་ཤར་བས་མཐའ་བཞི་དྲས་ཐག་ཆོད། །མེ་ཆུ་གཉིས་ཀྱི་ཚ་གྲང་བསྙམས་པས་བཙོ་བསྲེག་ཁར་མི་ལྟུང་། །ས་དང་རླུད་གཉིས་མཚན་ཉིད་ཤེས་པས་བྱིང་རྒོད་རང་སར་གྲོལ། །མཁའ་དང་འགྱུ་བའི་ཆོས་ཉིད་ཤེས་པས་དགོངས་པ་རྣམ་པར་དག །རྣམ་ཤེས་འོད་གསལ་ཤེས་པས་བར་དོ་ཚད་ལས་གྲོལ། །རིག་པ་ཡེ་ཤེས་ཤེས་པས་མཁྱེན་པའི་ཞིང་ཁམས་འཆར། །ཐིག་ལེ་ཡེ་ཤེས་ཤར་བས་སངས་རྒྱས་དགོངས་པ་ཤར། །འགྱུ་བ་ཟང་ཐལ་ཤར་བས་བྱིན་རླབས་ཕྱོགས་མེད་འཆར། །འགྲོ་དྲུག་རང་སར་གྲོལ་བས་སངས་རྒྱས་ཡུལ་མེད་འཆར། །འཁོར་བ་རང་སར་གྲོལ་བས་འགྲོ་དྲུག་འ་་ས་མེད། །སྣོད་བཅུད་ཕྱི་ནང་མེད་པས་གཟུང་འཛིན་དངས་ས་མེད། །རིག་པ་ཁོང་ནས་ཤར་བས་མ་རིག་དངས་ས་མེད། ཁྲེས་རབ་ཁོང་ནས་ཤར་བས་སྒྲིབ་པ་བྱང་བ་མེད། །འགྱུ་བ་ཟང་ཐལ་ཤར་བས་ལས་ཀྱི་འཁོར་ས་མེད། དུག་ལྔ་རང་སར་གྲོལ་བས་ཉོན་མོངས་སྐྱེས་ས་མེད། །གཟུང་འཛིན་དཀར་ནག་གྲོལ་བས་ཤེས་བྱའི་སྒྲིབ་པ་མེད། །ལུས་ཀྱི་མཐའ་གཞི་ཆོད་པས་རྣམ་སྨིན་གོལ་ས་མེད། །སྒྲིབ་པ་བཅུ་གཉིས་བསྡུས་པས་བཞི་རུ་འདུས། །དེ་ཡང་གསུམ་འདུས་གསུམ་པོ་གཅིག །རྒྱ་མཚོའི་གཏིང་གིས་ལུ་མའི་རྩད་ཆོད་ན། 
།རྒྱ་མཚོ་ཆེ་༄༅། །ཡང་རང་དངས་བསྐམ་པར་འགྱུར། ཁེང་ཕུང་ལྗོན་པའི་རྩ་བ་ཆོད་ཙ་ན། །ལོ་མ་མེ་ཏོག་འབྲས་ལྡན་བསྐམ་པར་འགྱུར། །དེ་བཞིན་སྒྲིབ་པའི་རྩ་བ་ཡིན་པར་ཤེས་པ་ལ། །འགྱུ་བ་ཤེས་པས་གོང་ལྷར་ཐམས་ཅད་ཆོད། །རྩ་བ་གཅིག་གི་དོན་རྟོགས་ན། །དཔེར་ན་ཛེ་ཏའི་སྨན་བཞིན་དུ། །གང་ལ་རེག་པ་སྨན་དུ་འགྱུར། །དེ་བཞིན་ཅིར་སྣང་ཅིར་འགྱུས་ཡང་། །ཡེ་ཤེས་དག་ལ་འདས་པ་ནི། །སྐད་ཅིག་ཙམ་ཡང་མི་སྲིད་དོ། །དབང་པོ་བསམ་པ་དང་སྦྱར་དུ། །འགྱུ་བ་གཅིག་པོའི་སྤྱི་ཁྱབ་ཡིན། །ཁྱབ་པ་གཅིག་པོའི་དོན་ཤེས་ན། །ཐམས་ཅད་སངས་རྒྱས་དགོངས་པ་ཡིན། །དབང་པོ་འདི་ལ་རྣོ་བརྟུལ་མེད། །འགྱུ་བ་བྱིན་རླབས་རང་སྣང་ཕྱིར། །ཐོས་པ་འདི་ལ་ཆེ་ཆུང་མེད། །འགྱུ་བ་ཡེ་ནས་དངོས་མེད་ཕྱིར། །དབང་པོ་བློ་ལ་རྣོ་བརྟུལ་མེད། འགྱུ་བ་རང་གྲོལ་འཆར་བའི་ཕྱིར། །དད་པ་འདི་ལ་ཆེ་ཆུང་མེད། །འགྱུ་བ་རང་ལས་བྱུང་བའི་ཡེ་ཤེས་ལ། །དད་པ་བཟང་པོས་འགྲུབ་པ་མེད །གནད་ཀྱི་གདམས་པས་ངོ་སྤྲད་ན། །སྨན་གྱི་རྒྱལ་པོ་ཛེཏ་བཞིན། །གང་ལ་རེག་པ་སྨན་དུ་འགྱུར། །དེ་བཞིན་རྣམ་པ་ཅིར་སྣང་ཡང་། །རང་སྣང་ཡེ་ཤེས་འབའ་ཞིག་ཡིན། །དཔེར་ན་རྩི་མཆོགམགིབཞིན། །གངལརེག་པ་དྲི་་་་རུ་འགྱུར། །དེབཞིན འགྱུཚདཡེཤེས་འབར། །ཞེས་གསུངས་སོ། །འབྱུང་པོ་ཀུན་འདུལ་གྱི་རྒྱུད་ལས། 
གསང་བའི་སྐྱེ་རྒྱུད་རྩད་བཅད་པའི་ལེའུ་སྟེ་བཅུ་གཉིས་པའོ།། །།དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །ཉོན་ཅིག་རྡོ་རྗེ་ཆོས་ཀྱི་བློ་གྲོས་ཁྱོད། །གསང་བའི་སྐྱེ་དགུ་རྩ་དང་ངོ་སྤྲད་པ། །དང་པོ་རྩ་ཡི་གཞི་མ་བསྟན། །ཐམས་ཅད་ཤེས་རབ་ཡེ་ཤེས་རྩ། །ཡོན་ཏན་རྩ་དང་ཕྲིན་ལས་རྩ། །དང་པོའི་གཞི་རྩ་བསྟན་པ་ནི། །བར་དུ་རླུད་དང་རྩ་དང་ཐིག་ལེ་དང། །རྒྱུ་རྐྱེན་འདུས་པས་འདུས་བྱས་དངོས་པོའི་རྩ། །དེ་ཡང་ཐིག་ལེ་བྱང་ཆུབ་སེམས། །དཀར་དམར་གཉིས་པོ་འདུས་པ་ལས། །རྩ་ལ་བརྟགས་དང་དཔྱད་པ་ལས། །རྩ་ལ་རྩ་རྒྱུད་བཤད་པར་བྱ། །གདུགས་དང་པདྨ་འཁོར་ལོ་དང་། །རིན་པོ་ཆེ་དང་རྩེ་གསུམ་རྡོ་རྗེའི་རྩ། །གནས་པ་དངོས་པོའི་རྩ་ཞེས་བརྗོད། །དེ་ནི་བབ་ལ་གནས་པའི་རྩ། །དེ་ཡང་ཨ་ཝ་དྷ་ཏི་དང་། །རོ་མ་རྐྱང་མ་གསུམ་དུ་འདུས། །ཐོག་མའི་ཆོས་གསུམ་དག་ལ་འདུས། །བར་དུ་མ་ནིང་གསུམ་ལ་འདུས། །རིག་པ་འོད་དང་དཀར་དམར་གསུམ་ལ་འདུས། རྩ་ཡི་ངོ་བོ་ཟང་ཐལ་བས། ཁྲེལ་གྱི་སྦུབས་མི་བྱ་བ་ཡིན། །མཛོད་སྤུར་དངོས་སུ་ཆགས་པ་བག་ཆགས་རྩ། །དངོས་སུ་གྲུབ་པའི་རྩ་ཞེས་བརྗོད། །རོ་རྐྱང་དྷ་ཏི་གསུམ་དུ་གྲགས། །གདུགས་ལྟ་བུ་ཡི་རྩ་ཞེས་པ། །སྤྱི་བོ་བདེ་ཆེན་གཞལ་ཡས་ཡིན། །རྩ་ནི་སུམ་ཅུ་རྩ་གཉིས་གནས། །མཁའ་འགྲོ་སུམ་ཅུ་རྩ་གཉིས་དང་། །ཡེ་ཤེས་རྩ་དང་ཐབས་དང་ཤེས་རབ་རྩ། །ཡོན་ཏན་་རྩ་དང་རང་རྒྱུད་རྩ། །རྣམ་པ་ལྔ་ལས་གྱེས་པ་ཡིན། 
།ཡེ་ཤེས་རྩ་ནི་དབུས་ནས་༄༅། །འགྱུ། །རང་རྒྱུད་རྩ་ཡང་དེ་ལ་བརྟེན། །ཐབས་ཀྱི་རྩ་ལ་ཆ་ལྔ་གནས། །གཉིས་འགྱུར་མཁའ་འགྲོ་བཅུ་རུ་བཤད། ཁྲེས་རབ་རྩ་ལ་རྣོ་བ་ལྔ་རུ་སྣང་། །གཉིས་འགྱུར་མཁའ་འགྲོ་བཅུ་རུ་གནས། །ཡོན་ཏན་རྩ་ལ་རྣམ་པ་ལྔ། །གཉིས་འགྱུར་མཁའ་འགྲོ་བཅུ་རུ་གནས། །ཡེ་ཤེས་མཁའ་འགྲོ་ལས་འགྲོ་གཉིས། །སུམ་ཅུ་རྩ་གཉིས་གདུགས་ཀྱི་གནས། །རྨ་བྱ་མདོངས་དང་བཅས་པ་ལྟར། རྩ་དང་མཁའ་འགྲོ་རང་ཆས་ཡིན། །མགྲིན་པ་ལོངས་སྤྱོད་པདྨ་འདྲ། །པདྨ་ཅན་གྱི་གནས་ཞེས་བྱ། །དེ་ལ་རྩ་ཡང་བཅུ་དྲུག་ཡོད། །ཡེ་ཤེས་ཐབས་དང་ཤེས་རབ་རྩ། །ཡོན་ཏན་རང་རྒྱུད་རྣམ་ལྔའི་རྩ། །ཐབས་ཀྱི་རྩ་ལ་མཁའ་འགྲོ་ལྔ། ཁྲེས་རབ་ལ་ཡང་དེ་ལྟར་རོ། །ཡོན་ཏན་ལ་ཡང་དེ་ལྟར་ཤེས། །ཡེ་ཤེས་རྩ་དང་རང་རྒྱུད་ལ། །མཁའ་འགྲོ་རྣམ་སྣང་གསལ་བས་ན། །དེ་ཡང་རང་ཆས་སྔགས་ཀྱི་རྩ། །སྙིང་ག་ཆོས་ཀྱི་འཁོར་ལོ་ནི། །བྱང་ཆུབ་མཐའ་ཡས་ཕུབ་ཅེས་བྱ། །དེ་ན་རྩ་བཅུས་ཁྱུད་མོ་འདྲ་བ་ཡི། །རྩ་ཡང་རྣམ་པ་བརྒྱད་ཡོད་དེ། །ཡེ་ཤེས་ཐབས་དང་ཤེས་རབ་དང་། །ཡོན་ཏན་རང་རྒྱུད་རྩ་བརྒྱད་གནས། །ཐབས་ཀྱི་རྩ་ལ་མཁའ་འགྲོ་གཉིས། །ལྷོ་ཕྱོགས་ཤར་གྱི་རང་ས་ཟིན། ཁྲེས་རབ་རྩ་ལ་མཁའ་འགྲོ་གཉིས། །ནུབ་བྱང་གཉིས་ཀྱི་རང་ས་ཟིན། །ཡོན་ཏན་རྩ་ལ་མཁའ་འགྲོ་གཉིས། །མཚམས་ཀྱི་མཁའ་འགྲོ་ཞེས་སུ་བཤད། །ཡེ་ཤེས་རྩ་དང་རང་རྒྱུད་ལ། ། 
སྟེང་འོག་མཁའ་འགྲོ་གནས་པ་ཡིན། །ཉི་མ་མདངས་དང་ཆས་པ་བཞིན། རྩ་དང་མཁའ་འགྲོ་རང་ཆས་ཡིན། །ལྟེ་བ་སྤྲུལ་པའི་འཁོར་ལོ་ནི། །འཇིགས་སུ་རུང་བའི་གནས་ཞེས་བྱ། །དེ་ཡང་གོང་དུ་རིམ་པས་ཤེས། དེ་ལ་དྲུག་ཅུ་རྩ་བཞི་གནས། །འཁོར་ལོ་མཚོན་ཆ་འདྲ་བ་ལ། །འཇིགས་སུ་རུང་བའི་གནས་ཞེས་བྱ། །དེ་ཡང་གོང་དང་འདྲ་བ་ལ། །ཐབས་ཀྱི་རྩ་ལ་མཁའ་འགྲོ་བཅུ། ཁྲེས་རབ་རྩ་ལ་དེ་བཞིན་ནོ། །་ཡོན་ཏན་རྩ་ཡང་དེ་བཞིན་ནོ། །རང་རྒྱུད་རྩ་ལ་མཁའ་འགྲོ་བཅུ། །ཡེ་ཤེས་རྩ་ལ་གཉིས་གནས་ཏེ། །དྲུག་ཅུ་རྩ་བཞི་འཁོར་བཅས་རྣམས། །མཚོན་ཆ་འཁོར་ལོ་འབར་བ་བསྣམས། །འགྲོ་དྲུག་གྲོང་ཁྱེར་དམྱལ་ཐག་ཆོད། །འཇིགས་སུ་རུང་བའི་གནས་ཞེས་བྱ། །མན་ཤེས་འོད་དང་ཆས་པ་བཞིན། །རྩ་དང་མཁའ་འགྲོ་རང་ཆས་ཡིན། །གསང་བ་རྡོ་རྗེ་ལྟ་བུའི་རྩ། །རྡོ་རྗེ་རྩེ་གསུམ་ལྟ་བུ་ཡོད། །ཁ་ཊ༷་དུང་ཅན་ཤིན་ཏུ་དཀོན། །དེ་ཡི་མཚན་ཉིད་བསྟན་པ་ནི། །མས་འགེབ་བུས་སེ་མིག་སྨིན་འབྱར། །བྱ་ལག་ཆུ་རྐང་སྐྲ་ལོ་སེར། །ལྟག་རྩ་འཁོར་བར་ཡོད་པ་ལ། །པོ་སྐད་ཅན་དང་གདོང་ཚུགས་ལྟ། །བབ་ཅོལ་སྨྲ་ཞིང་ཟུར་འགྲོས་བྱེད། །དེ་ལ་དུང་ཅན་ཡོད་པ་ནི། །མཁའ་འགྲོ་མ་ཡི་རིགས་སུ་བཤད། །དེ་ཡིས་བདེ་བ་ཁྱད་འཕགས་པས། །བདེ་བ་ཅན་གྱི་གནས་ཞེས་བྱ། །གཡས་ན་ཐབས་ཀྱི་མཁའ་འགྲོ་ནི། །གླང་པོ་གདོང་ཞེས་བྱ་བ་ཡོད། །གཡོན་ན་ཤེས་རབ་མཁའ་འགྲོ་མ།
