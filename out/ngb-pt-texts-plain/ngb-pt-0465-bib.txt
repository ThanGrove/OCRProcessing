ཨ་ཡུ་་་། །ཕ་ལ་སི་དྷི་སྒྱུ༷། བྱང་ཆུབ་ཀྱི་སེམས་སྤྲོ་བསྡུ་བྱ། འདི་ཟོས་ཡོན་ཏན་བརྗོད་མི་ལང་། །བཅུ་ཐུབ་པ་ཡངས་བརྒྱ་ཚེ་འགྱུར། །འཆི་བ་དྲུག་ནི་བསླུ་བར་བྱེད། །བརྒྱད་ཅུ་ལོན་པ་དྲུག་ཅུ་ལྟར། །སེང་གེ་གླང་ན་རྨ་བྱ་ཁྱུང་། །བཀྲག་མདངས་མེ་ཏོག་པདྨ་བཞིན། །སྔགས་འཆང་རིགས་འཛིན་ཀུན་གྱིས་རློབས། །མ་མོ་མཁའ་འགྲོས་སྡོངས་གྲོགས་བྱེད། །བདུད་རྩི་འདི་ཡོན་ཏན་གྱིས། །ཚེ་ལ་དབང་བའི་རིགས་འཛིན་བ། །ཅེས་གསུངས་སོ། །བཅུད་ལེན་བསྟན་པའི་ལེའུ་སྟེ་དང་པོའི་སྐབས་སོ།། །།དེ་ནས་གཞན་ཡང་བཀའ་སྩལ་པ། །འཕོ་བའི་རྒྱུད་དག་རྒྱུད་ལྔར་བསྟན། །དྲན་པས་སངས་རྒྱའི་གདམ་ངག་ནི། །རྩོལ་ད་འཕོ་བའི་དོན་བསྟན་པ། །འགྲོ་བ་རིགས་དྲུག་སེམས་ཅན་ལ། །ཡེ་ཤེས་རེ་རེའི་ཆ་ཡོད་དེ། །འགྲོ་བ་མི་ལ་ཡེ་ཤེས་ལྔ། །འདུས་པའི་ཆ་ནི་ཡོད་པས་ན། །འཛམ་གླིང་ལས་ཀྱི་ས་པ་ཡིན། །དེ་ལ་སངས་རྒྱས་སྟན་ཚང་བས། །དྲན་པ་སངས་རྒྱའི་གདམས་ངག་དོན། །འཆི་ཁར་དྲན་པས་སྐྱེལ་མའི་ཚུལ། །དབུགས་ཆད་ཡེ་ཤེས་འབྱུང་བ་ནི། བསུ་མའི་ཚུལ་དུ་ཕྲད་པ་ཡིན། །ཐ་མ་ཞིང་ཁམས་འཕོ་བ་ལ། །དམ་ཚིག་སྡོམ་པ་ན་པར་བྱ། །དེ་ཡི་ངང་ལ་བསམ་པ་ནི། །བཟང་པོའི་ངང་དུ་འགྲོ་བར་བྱ། །ཞེས་བཀའ་སྩལ་པ་དང་། རྡོ་རྗེ་སེམས་དཔའ་ལ་ཡང་བོས་ནས་བཀའ་སྩལ་པ། 
རྡོ་རྗེ་སེམས་དཔའ་རྒྱུད་འདི་ནི་ཤིན་ཏུ་གསང་བའི་རྒྱུད་ཅེས་བྱ་བ་ཡིན་ཏེ། བར་དོ་གསང་བའི་རྒྱུད་ཕྱི་མ་ཞེས་བྱ་བར་ཟུང་ཤིག །ཅེས་གསུངས་སོ། །རྒྱུད་གཟུང་བའི་ལེའུ་ཡི་སྐབས་ཏེ་གཉིས་པའོ།། །།བར་དོ་གསང་བའི་རྒྱུད་ཕྱི་མ་ཞེས་བྱ་བ་རྫོགས་སོ།། །།༄༅། །རྒྱ་གར་སྐད་དུ། སན་ཏི་མ་ཧྭ་བོ་དྷི་ཙི་ཏ་རད་ན་ཨུ་བ་དྷི་ཤ་གུ་རྒྱ་ནཱ་མ་ཏནྟྲ། བོད་སྐད་དུ། རྫོགས་པ་ཆེན་པོ་བྱང་ཆུབ་ཀྱི་སེམས་རིན་པོ་ཆེ་སྐྲུངས་པ་གསང་བ་མན་ངག་གི་རྒྱུད་ཅེསབྱ་བ། བཅོམ་ལྡན
