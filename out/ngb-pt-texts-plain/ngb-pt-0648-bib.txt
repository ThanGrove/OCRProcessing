མ་ར་ཧུ་ཏི་ཏན་ཧབ་༄༅། །ཁྲག །ལྟེ་ཁྲག་མྱོགས་མུ་ར་ཡ་རྦད། ཨེ་ཨུ་ཀཁོ་ཡ་ཛ་ཡ་ཛ། ཨེ་ཙ་ཙ་ཤཱ་ར་ཡ་བཱ་ར་ཡ་རྦད། སྟསྦྱ་ཡ་ནན། ཨུ་ནན་ཨུ་ནན་ཏང་། ཨ་མར་ཡ་ནན། བི་ཏ་ཡ་ཧ་ཧ་པྲ་མ་ཀ་ར་ས་ད་ཨ། པ་ར་ཡ་ཐུམས་ཛ།ཨི་ཛ་ལ་ཕྱ་ཤག ཁྲག་ཁ༷་ཤག་བི་ཤ་ཕྱེ། པྲ་ཙ་ལི་ཕཊ་ཕཊ་མ་ར་ཡ། འཁོར་ལོ་སྐྲུ་གྲི་གཟས་པ་བསྒྲལ་ཅེས་གསུངས་སོ། །ཕྱིར་བཟློག་ནག་པོ་སྔགས་ཀྱི་རྒྱུད་ལས། འཁོར་ལོ་ལ་གཤེད་མའི་སྔགས་བྲི་བའི་ལེའུ་སྟེ་བཞི་པའོ།། །།དེ་ནས་གཤེད་མའི་ཡི་གེ་རྒྱུད་ལ་བླང་ཞིང་། རང་བདུད་རང་ལ་དབབ་པ་འདི་གསུངས་སོ། །གསང་སྔགས་དྲག་པོའི་ཡི་གེ་སྡེབས། །ཕྲ་མེན་དང་ནི་དམག་དཔོན་དང་། །དམག་མི་དང་རྣི་སྐྱེསབུདང་། །དེ་བཞིན་སྔགས་རྣམས་བསྡུས་ལ་བཟླས། །གཤེད་མའི་ཡི་གེ་འདི་ལྟ་སྟེ། །ཆེ་གེ་མོ་ཨུ་ཨུ་ཨུ་ནེ་ཨུ་ནེ་ཆེ་གེ་མོ་ཙིལྟི་སྦུར་ནན། ཏྲི་སྨྱུ་ར་ཡ་སྟསྟ་ཡ་ནན། ཏྲི་སྨྱུ་ར་ཡ་སྟསྟ་ཡ་མྱོགས་ཐིབས་རོ་མྱགས་མྱ་རཡ་རྦདནན། བྷ་ཏི་དུ་ན་ཏི་མྱ་ར་ཡ་ཧུར་ཐུམས་ཛ། །ཨུ་ནན་ཨེ་ནན་རི་རི་ཏིང་ཏིང་། མི་ལ་ནི་ཛོ་ནི་ཏྲག་ནན་བིག་ནན། རག་ཤེ་ཡག་མོ་ཤུ་ལ་ཡ༷་ཏྲི་དུན་དུན་ཙིསྟ་དུན། སྙིང་ཁྲག་རྦད། རྩ་རི་ཏྲི་ཤམ། དེ་བ་སུམ། ཡག་ཤ་ཁྱེ།རག་ཏི་ཁ༷། སྙིང་ལ་ཐིབས། ཆེ་གེ་མོ་སྙིང་མུ་ར་ཡ་ཧྲིལ་རྦད། ཏྲིད་ཨུ་ནེ་ཨུ་ནེ། རི་རི་ཧ་ཧ་ར་ཡི། ཧུར་ཐུམས་ཛ། སྨྱུ་ར་ཡ། ཧྲི་སྟི་གརྦ། ཁུག་དྲོངས། བཛྲ་སྭ་དུ་གིང་ཀ་ར། ཧུར་ཐུམས་རིལ་སྙིང་ལ་རྦད། 
སྙིང་ལ་ནཱ་ར་ཡ་འདུས། ཏྲི་ཕྲིལ་ཤག་རྦད་ཨེ། གཤེད་མའི་ཡི་གེ་བཏུ་སྟེ། དོན་མེད་བཟླས་ན་རང་ལ་གཅོད། །དབྱེ་དང་དྲིལ་དང་མན་ངག་གཅོད། །དེ་བཞིན་མཁས་པར་ཤེས་པར་བྱ། །གཤེད་མས་གསད་པའི་ཕག་རྒོད་སྟེ། །གསང་ཐུན་བརྡེག་ཅིང་གསད་པའི་ལས། །རང་གི་སྲུང་མ་བདུད་དུ་འབེབས། །ཞེས་གསུངས་སོ། །ཕྱིར་བཟློག་ནག་པོའི་ངན་སྔགས་ཀྱི་སྡེབས་ཀྱི་རྒྱུད་ལས། གཤེད་མའི་ཡི་གེ་རྒྱུད་ལ་བསླང་བའི་ལེའུ་སྟེ་ལྔ་པའོ།། །།དེ་ནས་མགོན་པོ་ཆེན་པོ་དེས། སྐྲུ་གྲི་ནག་པོའི་སྔགས་ཀྱི་ནང་འགྲས་ཀྱི་སྟོབས་གཞོམ་པ་འདི་གསུངས་སོ། །ཨོ་ཤི་མཉུ་བ་ར་ཀྲོདྷ། ཨ་ཀྲོ་ཏེ་ཀ་ཡ་མན་ཏ་ཀ་ཧྲིལྟི་ཀྭ་ལ་རྭ་པ་ཛ། སྙིང་ཁ་རག་ཧྲི་ཏི་དེ་མ།་ག་རག་ཤ་རུ་ལུ་སྨྱོ། ཐུམ་རི་ལི་ལི་སྨྱོ། ཧྲི་ཧྲི་ཁ་ལ་སྨྱོ། ཀ་ལ་ཀྲིན་མྱགས་སྨྱོ། ཧྲི་ཏི་ཨ་པ་ཡེ་རྦད། ཀ་ལ་ཡེ་རྦད། དུ་ཁ་དུ་ས་རྣིར་ཤ་ཀུ་མྱ་ར་ཡ་རྦད་ཨེ། ཆེ་གེ་མོ་ནི་ཤ་ཀུ་རུ་ཐུམ་རི་ལི་ལི་ཁྱེར་སྨྱོ། ཧྲི་ཧྲི་ཁ་ལོང་སྨྱོ།ཙིསྟ་ཧབ་ཤབ་ཏྲི་ཤག་སྨྱོ། པྲ་མ་ཡ་རྦད། མུར་རྩ་ཡ་རྦད་རཀྟ་ཙ་ར་ཡ་རྦད། ཆེ་གེ་མོ་མྱ་ར་ཡ་རྦད། སྲོག་ཁྱེར་མྱ་ར་ཡ་རྦད། སིང་ང་ཡེ་མྱ་ར་ཡ་རྦད། དུ་རུ་ཨེ་མྱ་ར་ཡ་རྦད། བྱ་ཙ་ཡེ་ནཱ་ར་ཡ་རྦད། ཐུམ་ས་རྦད། ཨེ་བན་ཏྲ་སྔགས་པ་ཙི་ཏི་མ་རི་སྨྱོ། སྲོག་ཁྱེར་བཟློག་ཐུམ་རིལ་མུ་ར་ཡ་བཟློག་སྨྱོ། །སྐྲུ་གྲི་ནག་པོ་སྔགས་རྒོད་ཀྱིས། །ནང་འགྲས་པ་དག་འཇོམས་པར་འགྱུར། །གང་ལ་བརྒྱབ་ཀྱང་ཐོགས་པ་མེད།
