1་་ཙྭ་སྐྱིཁྱིཚོ་ཛི་་་་་་་ཉུ་1 1་་་་་་་་་་་་་་་་་་་་་་་ཁུ་་་་1 1་་་་་་་་་ཚོ་་་་ཁུ་་་་་་་་ཟི་་1 1་ཁྱི་་་་ཁུ་་་་་་་་་་་་་ཚོ་་་་་་་1 1་་་ཁྱི་་་་་་ཁུ་་་་་ཁུ་་་་་ཟློ་་ཁུ1 1སྟུ་ཁུ་་་་་་་་་་ཁུ་་ཁྲུ་ཁྱི་་་ཁྲི་་རྨྱུ་1 1་་་་་་བོ་་ཅྭ་་་་་ཁུ་ཟློ་་་་་་་1 ་1་་་སློ་་་་་་་་་་་་ཁྲི་་་་་་་ ་་་ཛི1 ་ཅོ་་་་་་་་་་་་་་་་ཟློ་་་་་་ཉུ་་པྱི་། 1་ཟིརྭ་་་་་་་་་ཁྲུ་གློ་་་་་་་་་་་་་1 །སྒྲོ་་་་་་ཛི་་་་ཁྱི་ཁུཀྱུ་་་བྱུ1 ་1་་ཞུ་་་ ་་་་་་ཟློ་་ཙོ་་་་་་སྨྱོ1 ་་་་་་་་་་་་་༷་་་་་་༦1 ་1 ཁྱེ་་་ཟློ་་་་་་༦་ཚོ་་་་རྟོ། 1ཉུ་་་་་་་་་་་་རློ་་༦་་་་1 1་་་་་་་་༷ཁྱི་་རྩི་་ཁྱི་་་་་1 །་་་་་ཁྱི་་་་་་་་་་་་་ཁུརྨྱུ་ ཆི1 1སྟུ་་་་་་་་་རྨྱུ་་་་་་1 1་་་༦་་་་་སྟུ་་་༷་་་སྦྱོ་་་་་་། །་་ཟི་་་་་གོ་རྨྱི་ཙྭ་་་་་ཟློ་་་་་ཁུབྱོ་་སྨོ1 1་་་་་་་་་་་་་ཙུབྱུ་་ཁུ་་སྦྱ༷་ ་྅་ ་་ཉུ་1 1་༦་་ཛི་་་ཙྭ་་ཁྱི་་་་་སྙོ་་སྩོ་་་་་་ཁུ། ་1་་བྱོ་་གློ་་ཟློ་་་ཁྱི་་་་ཛི་་ཁྱི་་་་་་་་་1 1༷ཁྲི་་་་་་་་ཁྱི་ཙྭཅོ་་ཛི་་སྤྱོ་ཉུཚོ་་1 1་་་་་་་་་ཟི་་་ཁྱེ་་་་་་ཁྱི་་་་་་་་་་ཕྱུ1 །་་་་་་་་གློ་ཁྱི་ཉུ་་གོསྣོ་་ཛི་ཉུ་་ཉུ1 1་་་་་༷1 ་ 
1་་1་་ཁྱི་་་་་་་ཚི་སྐྲུ་་་་་་་་་་་་་་་་་་་་་་ཉུ་ཁྱི 1རྨྱུ་་ཉུ་་་་་་་་་་་་་་་་་་་་ཛི་་་་ཛི1 1་་་་་་་་་་་་་་་་་་་་་་་་་་་་ཁྱི 1ཁྱི་ཛི་་་་་་་་་་་་་་་་་་་་་་་་ཁྱི་་་་་་་་་་་་ཁྱེ་་་་་ 1སྦྱི་་་་་་་་་་་་་་་་་་་་་་་་་་་་་་་ཁྱི་་ཟི་་་་་་་་་་སྦྱོཚི་་ཁྱོ། ་་་་་་་་སྨོ་་་་་་་་་་་་་་་་ཉུ་་་་་་་ཁྱི་་་་་་་་་་ ་་་1 1རྨྱོ་་སྤྱོ་་་་་་་་་ཟློ་་་་་ཟློ་་་་་ཟློ་་་་་་་་་་་་་་་1 14་་ཚི་་ཁྲུ་གློ་་་ཅོ་་་་་་་་་་་་་་་་་་་་ཇོ་སྩོ་་་་་་་་་ཟི་་ཁྱི་སྤྱོ་་་་་་་་་་་་ཁུརྭ་ ཁྱི1 1་་་་ཁྱི་་་་ཁུ་སྤྱོཁྱི་་་་་་་་ཇོ་་་་་་་་་་་་་་་་་་་་་་་་་ ་1་་སྩོ་་གྱོ་་་་་ཁྲུ་་་་་་་་་་་་་་་སྙོ་་་་་་ལྕོ་་་་་་་་་་4་་་་་་་་་་ཁུ་་ཁྱི་་་སྤྱོ་ཁྲི་་་ཐོབྱུ་་་་་་་སྟུ་ ་་་་་་་་་་་་སྦྱོ་་་་་་མི་་་་་་་་་་་ཁྱོ་ཞུ་་་་་༦་་རྨྱུཉུ་་རྐྱུཁྱི་་་་་ཟློ་་་་་་་་་་་་་་་་་1 ་1་་་་་་་ཏོ་་ཟློ་་་་་་་་་་་་་་་་་་་་་་་་ཉུ་་་སྣྲོ་་སྤྱོཛི1་1 ་ 11་་་་་་སྦྱི་་སྟོ་་ཙོསྐྱི་་་ཁྱི 1་་་་་་ཚོ་་སྟེ་་་ཁྱི་་་ཅྭ་རྨྱུ1 1་་་་་་་་་་སྦྱི་་་་་་་ཟློ་1 1ཆོཚོ་སྦྱི་་་ཟློ་་་་་་་ཁུ་་་་་སྐྱི་1 
1ཅོ་་་་་་་སྤྲོ་་་་ཚོ་་་་་་་་་་1 14་་་་་་་ཁྱི་་་་ཁུ་་་་་་་་1 1་ཁྱི་་་་་་་་་་་་་་་་་1 1ཉུ་་སྐྱི་་་ཟི་་་་་་་་་ཁྱི་་་་1 1་་་་་ཁྱི་་ཛི་ཁྱི་་་་་་ཁྱི་1 14་་ཁྱི་་་་བྱུ་་རྗོ་་1 1་་་་རྟོ་་༷་་་་་་་་་་་་་1 1་་་་་རྨྱེ་་་ཙྭ་་་་་་བྱུ་་ཅོཁྱི་་་་1་་་་རྭ་ཟི་་་་་་་་ཁྱི་་་་1 1ཛིརྭ་ ་ཁྱི་་་་ཉུ་ སྦྱི་་་་་ཁུ་་1 1རྭབོ་་་་་་་་་་་་་་་ ་་རྨྱུསྤྱོ1 1་་སྦྱོ་་ལྗོ་་ཁྱེ་་་་་་་་་་་།གོ་ཁྱི་་་་་་་་་ཉུ་་་ཟི་རྗོ་་1 1་སློ་་་་་་་་་་སྦྱ་་ཁུ༷་་སྨོ་་་1 14་་སྟུ་་་་་་ཁུཁྱི་་་ཁུ་་་་་1 1བྱུ་་་པྱོ་་་་་ཟློསྐྲུ་་་་་་་་་་་་་ཛི་་1 1་སྐོཐོ་་་་་་་་་་་སྤྱོ་་་་་ཟི་་་སྦྱི་1 1རྨྱུ་་སྟུ་་་ཟློ་་ཁུཁྱི་་་ཇོ་་་་་ཁུ1 1་་་༷་་་་སྤྲོ ་་ ་་་ ་ཐོ་་་རྨྱོ11་སློ་་་་་སྙོ་་་་་་་་་་་ཁྱི1 14་་སྟུ་་་་་་སྤྱོ་་་་་་་་ཁུགོ 1ཉུ་་་་་་༷་ ཚི་་ཆི་་་་་་་1 ་་་་་་་་་ཟི་།་་་་་བྱོཁྱི་་་་ཁྱི་་་་་་་་་་༦་ཟི་་་་་ཁྱི་་་་་་སྟུ་་་་་་་་་་་་་་་་་་་་་་་ 1་ཟློ་་ཁྱི་་་་་་་་་ཁུ་སྙི་་་ཅོ་1ལྗོ་་ཁྱི་ཙུ་ཇོ་་རྩི་་་་གྱི་་1་་་་གློ་་་་སྟུསྟི་ ་་་་སྒྲོ་་སྤྲི་་་སྟོསྤྱོ་ཛི་ཛི་་་་་་་་་་་་ཟི་་ཁྱི1 1རྨྱུ་་་སྤྱོ་་་ཛི་་་་་་་་་་་ཉུ་་་་་་་སྒྱི་སྦྱི་་་ཚིསྐྱུཁྱེ1

1་རྗོ་་་ཁྱི་་ཛི་་་་་་་་་་་། 1་་་་་་ཙོ་་་་་་་་་་་་་1 །་་ཁྱི་་་་་་་་་་་་་་་་་ཁྱི་། 1རྨྱུ་་་་་་ཁྱི་་་་་་་་་་་་1 1་་ཁྱི་་ཁུ་་་་4་་་་་་་་་1 1་་སྟུ་གློ་་་་་་་་སྩིཀེ་་ཇོ་1 ་་་་་་་་་་་ཛི་་་་་་་་་་་་༽ 1་་་་་་་ཁྱི་་་་་རྨྱུ་་་་་་སྐྱུཁུ་1 1ཁྱི་་་་་་ཁྱི་་་་་་་་་་་སྟུ1 1་་་་ཁྲི་་་་་ཁྱི་་་རྐྱུ་་་་་1།་་་་ཁུ་་ཁུ་་་་་་་་་་་་ཁུ1 །ཇོ་་་་་་་་་་་་་་་་་། 1་་་་་་་་་་་་་་་་ཡི་་་ 1་་ཛི་་་་་་་་་་་་་་་་་ཇོ་་་1 །རྩོ་སྐྱུ་་་་ཁྲི་་ཁྱི་་་་་སྟུ1 1ཉུ་་་ཁྱི་་་་་གློ་་་་་1 1་་སྩོ་་་་ཆི་སྒྲོ་་་ཇོ་་་་་་་1 ་1་་་་་༷་་་་་་་་་་་་་་1 ་1་ཁྱི་་་་་ཁྱི་་་་་་་་་་་་་་་1 1་་་་་་་ཁྱི་་་་་་་་་་་སྔོ་་་་་1 ་་་་་་་་་་་་ཇོ་་་་་་།་1་་་་་ཉུ་་ཕྲི་་་་་་་་་་་་་་་་1 1་་་་ཁྱི་་་་་་་་་༷་སྟུ་་1 1་་་་་་་་་་་་་་་་་་་་གོ་་་གློ་1 1་་་་་་ཁྱི་་་སྟུ་་་་་སྦྱོ1 1་་་་་་་་ཁྱི་་་་་་་སྤྱ་་་་་1 1་་་་་་་་་་་་་་་སྐྱི་་་་་་1 1་ཏོ་་་་་ཅོ་སྦྱི་་་་་་་སྐྱི1 ་1་་་་་་ཚོ་་་་་རྩི་་ཛི1 1་་་རྭ་ ་་སྔོ་་་་ཟློཉུ་ཉུ་་་1 1་སྐྱུ་་་ལྗོ་་གོགྱི་་་སྤྲོ་ཉུ1 1སྦྱི་་་་་་་་་སྙི་་་ཅོ་རྗོ་སྣྲོ1 
1་ཉུ་་ཅུ་་་་་ཁུ་་་་་་་་་ཁུ1 1རྨྱུ་་་་་་་་་་་་་ཁུ་་་1 1བྱི་་ཁུ་སྡིཚི་་ཁུ་་སྐྲུ་་་་་1 1་་་་་་ཁྱི་་་་སློ་་ཅུ་་་ 1་་་ཅི་་་་་་་ཇོ་་་་་་་་1 1་་ཟོཅྭ་ ་་་་ཛི་་་་་་་1 1་་་་་་་་བོ་་་ཚོ་་སྦྱི་་་་། 1ཁྱི་་ནཱ་་ཁྱི་་་་གློབྱུ་ ཁྱི་་། 1སྤྲོ་་་་་་་་་་ཁྱི་ཉུ་༷སྐོ། 1་་་་་ཁྲི་་་་་་་་་་་་་1 1་་་་ཁྱི་་་་་་་་་་ཁུ་་ཉུ། 1ཁྱི་ཉུ་་བྱུ་་་་་་་་་་་་༷1 1་་་་་་་ཁྱི་་་་ཙྭ་་ཇོ་་1 1་་་་་་་་་་་་་་་་་ཁུ་་1 །་བྱུསྔི་་་ཁུ་་ཁྱི་་ཁྱི་་་ཟི་་་་་་་་།་1ཉུ་་་་་་་་་་་་་་་་་་1་།་་་་་་་་་ཁྲི་་་་་༦་་་་་སློ་1 རྭ1 ་་་་་་4་ཁུ་་ཁུ་་རྙུ་ཟློ་་་། །ཁྱི༷་ ་་་་་་་་་་་རྭ་ ་་་་་1 1་་ཁུ་་་་་་་་་་་་་་་་་་་་། 1་་་་ཁྱི་་་་་་་་་4་་1 1་་་་་་་་་སྐྱུ་་་ཁུ་་་་ཁུ1 1་་་་་་་ཁྲོཁུ་་ཁྱི་་་་༷་་བྱུ་་། 1་་་་་་་་་་ཚོ་ཟི་་་་་་་་། 1་་་་་་་་༦་་་་་་སྡུ་་་་་ཁུ་1 1ཁྱི་་་་་་ཐོ་་་་་ཚོསྨོ་་་་་༦གོ 1་་་་་་ཁུ་་་ཇི་་་ཁྱི་་་་་་་1 1ཁྱི་སྐྲུ་ཁྱེ་་་་་་་་ཁྱི་་་་་1 1ཁུ་་་་་་4་ཉུསྣྲིསྣྲོ་་ཁྱི། །་ཁྱི་་ཕོ་་་་ཁྱི་་་ཁྱི་བྱུམི་་་་4་1 1་྅་ཙོ་་་་་་་་་སྦྱི་ཟློཁྱེ་་་་་1 ་1་་་་་་་་་ཚོ་་་སྔོ་་གྱི་་་་རྭཟི1 1ཁྱི་་་་་སྩོ་་སྦྱི་་་་་ཁུ 1་ཙོ་1 ་ 
1་ཇོ་་་་་་རྩི་་་་་་་་ཅྭ་་་1 1་་་་་ཁུ་་་ཚོ་་ཁྱེ་་4་་་1 1་་4་་་ཁུ་ཁྱེ་་་་བྱུསྔོ་་སྦྱི་་་ཁྱི་1 1་་་་་་་་་་་་་་་་་་་་ཁྲི། 1ཁྱི་་་་་་ཁུ་་ཁུ་་་་་་1 1ཁྱི་་་་་ཆོཛི་་༦་་སློ་་་་་རྭ1 1་་་་་ཁུ་་་་་་་་་་་ཁུ་་་་་་1 1་་་ཛ་་་་ཚོ་་་་་1 1ཁྱི་་་་་་་་་་་་་་་་ཁུ1 1་ཁུ་སྐྱུ་་་་་་་་་་་་་་་ཁྱི1 1རྩོ་སྤྱོ་་་་་་་་་་་་་་། །་་་་་་་་་་སློ་་་་་་་་་་་་སྟུ1 1་་་་་་་་་་གློ་་་་་་་་་་་་་1གོཇོ་་་་་་་་་ཁྱི་་་་་་་ཁྱི་་་རྩོཙྭ1་1་་་་་ཁུ་་་་བྱུ་་ཁུ་་ཇོ1་།་༷་་་་་་་་་་་་་བྱོ་ལྗོ་་1་1་་་་་་་་་་་་་་་་་་་ཁྱི་་་1 1ཁྱི༷་ ་་་་་་་་ཛི་་་་པྱོ་་་1་1་་་་༷་་སྤོ་བྱུ་ ་་་ཁྲི་་་་་་་ཁྱི་1གོ་་་་་་་་་་ཁྱེ་་་་་་་་་རྨྱུ་་1 1་་་་ཁྲི་་ཁྱི་་་་སྒྲོ་་་་ཟློ1 1་་་་་ཚོ་་་ཁྱི་་་་་སྤྱོ་་་་་་་རྨྱུ1 1་་་་་་་་་་་་་་་་་་བྱོ1 ་་་་་་་་ ་་་་་་ཁུཁྱི་་་་་་ཁྱེ1 1་་་་ཁུ་་་་་ཚི་་ཚི་་་་་་1 ་།4་་་་ཁྱི་་་་་་་་་་་་་་ཚོ་་་རྟོ། །་་་་་་་ཇོ་ཉུ་གློ་་་་ཁྱི་1 1རྩི་་་་ཁྱིཙོ་་ཁྱི་་་་ཟློ་། །་་་་་་་ཐོ་་་་ཁྲུརྗོ་སྒྱི་1 1་་་་་་་་་་་་་་གྱི་་་སྒྲི་ཙོ1་1་་་་་་སྐྱི་་་་་་་ཁུ་་་སྐྱི་བྱུ1 
1་་་་་་་་་་་ཅྭ་ ་རྨྱུ1 1་་ཁྱི་་་་་ཁྱི་་་ཁུ་་་་་་་་་1 1་་་་་་་་་་་་་ཁུ་་་་་ཁྱི་་་་1 1སྟུ་་་སྟུ་་་ཁུ་་སྒྲོ་་སྒྲོཁྱི། 1་་་་ཁུ་་་་་་་་་་་་་་་ཇོ1 1་་་་་་་་ཉུ་་་་་་སྨོ་་་སྐྱུ་་་1 1ཁུ་ཟློ་་སྐོ་་་་་་་རློ1 །་་་ཁུ་་སྨྱོ་་་་ཏུ་་་་་་1་།ཁྱི་ཟློ་་གློ་་་་་ཁུ་་་་་1 1་་་་ཛི་་་་་་ཁུ་་་་་་༷་ཇོ1 1་་ཉུ་་་་སྤྱོ་་་་་ཁྱི་་་ཟློ་་་་་1 1ཁྱི་་་་ ་་་་་་་་་་་་གོ་༷ཙི་ ཧཱུཾ1་་1་་་་་་་་་ཁྲི་་་་་ཁུ་་ཁྱོ་་་ཁྲོཁྱི1་་་་་ཁྲུ་་ཉུ་་་་་་སྒྲོ་་་་ཁུ་། ་་་་་ཁྱི་་་་་་་་་་་་་་་་་་41་1་་་་་ཁྲི་་ཁྱི་་་་་ཁྱི་་་་་ཇོ་1 1་ཙྭ་་་་་་་་་་་་་་་་་་སྟུ། 1་་་་་་ཙྭཁྱི་་ཁྱེ་་་་་་་་་་་རྟུ་། སྨྱུ1 ་་་་་་་སློ་་་་་་་་་་་ཁྲི་་་་་་་1་1་ཁྱེ་་ཟི་་་་་་་་་༦་་་་་༦་ཁུ་་་་་། 1་་སློ་་ཁྲི་་ཁྱི་་་་ཁུ་་་་སྟུ་་ཕྱུ་་1 1་་་་་་་་སྟུ་་་་་་་་་་་་་་་་་་་་1 1་་་ཁྱི་་་སྟུ་ཁུ་་་་་་་་་་ ་ཁྱི་་སྤྲི་་1 1་་་་་བྱུ་་ཇོ་༦་་་སློ་་་་ 1རྨྱི་ཟློཛ་་ཁུ་ཟློ་་་སྨི་༷་་་ཁུ་་་རྗི1།ཁྲི་སྟི་་་་་་་་་་་་་་་་་ཛི་ཡི1 1་་་་་སྐྲུ་་་་་ཙོ་་་་སྐྱུ་ ་་སྐྱི1 1་ཁྱེ་་་སྦྱི་་་ཚོགོ་་་་་་་སྤྲོ་་ཉུགོ 1ཆིསྟོ་་ཛི་ཁྱེ་་་སྔོ1 1་༷་་་་་་༷སྟུ་ཅོ་ཙོ་་སྤྲོ་་ཉུ1 ་་སྣྲི1 ་ ༷ 
1་་་་ཏོ་སྐྲུ་་་་་སྐྱུ་་་་ 1་་ཁྱི་་སྐྱུ་་་་ཁྱི་ཁུ་ཐོ་་་་ཏོ་་་་་་སྐྲུ་་་་1 1་༷་གྱི་་ཛི་་་་ཁྱི་་་ཐོ་་་་སྨི་སྐྱུ་ཁྱི་་་ཉུ་་་་1 1་་་་་་་་་་་་ཇོ་་་་་་་་་་་་་་་་་་་་་་་་་་་་་་་་ 1་་ཁྱི་་་་་་ཏོ་ཁུ་༷་་་་སྦྱོ་1 1་་ཁྱི་་ཛི་་་་ཁྱི་ཁུ་་་་་ཁུ1་1་༷་ཁྱི་་སྐྱོ་་་་ཁྱི་ཁུ་ཐོ་་་་ཁུ་4་་་ཇོ་4་་་་་་ ཉུ1 ༷༷་་་1 1ཟློ་་་་་བྱོ་་་་་་་་་ཁྱི་་་་་1 1་སྟུ་་་༷་་་་་བྱུ་་༷་ཚི་་་་་་སྟུ1 1་་ཁུ་་ཁུཟོ་་༷་་་་་་་་་་ཁྱེ་་ཁྱི1 །་ཁྱི་་་་་ཟློ་་་་་་་ཁྱེ་ཁུ1 ་་་་་རྗོ་ཟི་་་་་ཁྱི་དྲོ༷་་་༷་སྟུ11་་་་་་་་་་་ཁྲི་་རྨྱུ་སྐྱུ་ ༦་་་་་1་་་༷༷་་སྐྲུ་་་་་་་་་་་ཁྱི་ཎྜ་1 1་་་་་་་་་་་་ 1་༷་སྦྱི་་ ་་་་་་ཐོ་་1་་་་་་4་ཏོ་སྐྱུ་་་་་ཐོ་ །རྨྱེ་་་་་་་ཁྱི་་་ཟོ་་་1་་་་་་་་་་་་ཟོ་ 1་་་་་གོ་ ཟློ་་་་ 1་་་་་་རྨྱུཁུ་་་་1་་་་་་་་་་སྣོ་ །་་་་་་་་ཁུ་ཁུ་ཎྜ་ །་་་་་༷་གློ་རྨྱོ་ཎྜ་ །་་་་་༷་རྨྱེ་་་་ཐོ་ 1་་་་རྨྱུ་མི་་་1་་་་་་ཁུ་ཏོ་ 1་་ཇོ་་་་ཆི་ཎྜ་་།་་ཁུ་་་་ཐོ་ 1་་ཡི་སྐྱུ་་་་་ །་་རྙུ་་་ཁུ་ཎྜ་་ 14་བྱུ་་་་་་ 1་་་་་་ཐོ་་ 1རྨྱི་སྐྱི༷ ་རྨྱི་ཅོཅྭ་ཅྭ་་་1 །་་་་་་་ཁྱེ་་་གྱི་་ཙོ་་་་སྐྲུ་་་་་་་་སྟི་་སྐྱིཛི་སྟུ་་ཅི་་བོ་་་་་་་གྱི11 ་11་་་་་་་ཁྱི་ཙི་ ་་་ཙོ་་་་་་་་་་་ཅི་་་་་་གློ་1་་་་ཁྱི་སྟོཐོ་་ཁྱི་་ཉི་་༷ 
་་་༷་་་་་་་ཉུ་་་་་་་་་་་་། 1་་་་་་་་ཛི་་་་་་་་་་་་་་་1 །་་་་་ཁྱི་་་་་་་་་ཛི་་་་་1 1་་་་་་་་སྔོ་་་་4་་ཇོ་་་1 1་་་་་་་་་་་་་་་་་་་ཁྱེ་་1 1་ཁྱི་་་་་་་་་་་་་་་་་་ཁྱི་1 1་་ཁྱི་་་་་་་་་་་་་་་་་་་་། 1་་་་་་་་་་་ཇོ་ཁུ་་་་་་་རྨྱུ་1་1་་་སྨོ་་་་་་་་་་་་་་་་་་་་1 1་་ཛི་་་སྨོ་་་ཛི་་་་་་་1 1་་་ཙུ་་སྩོ་་་ཟི་་་་་ཁྱེ་་་་1 1་་་་་་་་་་་་་་་་་་་སྦྱི་ཉུ་1 1་་་་་ཛི་་ཉུ་་བྱོ་་་་ཁུ་་་་རྫི་་་1 1ཁྱི་་ཁྱི་་་་་་་་་་་་་་་་་་སྣྲོ་་་་་11 ་ 11་་་་་་་སྐོ་་་་༷་་ཁྱི་་་་་་་ཁྱི་་་་་་་་་་་་1་་་་ཁྱི་་་་་ཁྱི་་ཛི་་་་་་་་ཅྭ་ ་རྨྱུ་་་་1 །་ཁུབྱོ་་་་་་་་་་་་་་་་་་ཁྱེ། 1་་་་་་་་་་་ཁྱི་་སྨི་་1་1་་་་་་གློ་་་སྤྱོཉུ་་་ཁུ་རྙུ1་1་་་་་་་་་་་་་་་་་་་ཁུ་། ་།་་་་་་་་་་་ཛི་་་་ལྗོ་་་1 1་ཁྱི་་་ཐོ་་4་་་་་་་་་ཁུ་་་་བྱུ ཁྲོ་་་4་་ཇོ་་གོ 1་་་་་་་་1་་་་་ 1་་སྐྱོ་་་་་རྙུ་ཚུཁུ་སྐྱི་་་ 1༷༷་་་་་་་་་་་ཅོ་་་་་་ ་་་ཁྱི་་་་ཅྭ་ ་རྗོ་སྐྱོཀེསྦྱི༷་སྒྲི་སྦྱི༷་་ཅྭ་ ་་་་་ ་་ 1་༷་རྨྱོ་ཚེ་་་་་་ཁྱི་ཁུ་ཐོ་་་སྣེ་ ་ཙུཅུ་་་ཁྱི་་་་་སྤྱོ་་་་། 1་༷་རྙོ་་སྐྱུ་་་་་ཁྱི་ཁུ་་ཐོ་་་་སྨི་་ཉུ་ ་ཛི་་སྦྱི་ཁུགོ་་་་་་་ ་།་༷་ཁྱི་་ཛོ་་་་ཁྱི ཁུཐོ་1 ༷ 
་1་རྔོ་་ཁྲུ་་་་1 1་ཁྱི་་་་་་་་་་་ཁུ་་་་་ཛི1 1་་ཁྱི་་་་་་་་་་་་་་གྱི11 11་་་་ཁྱུ་ཁྱིསྩ་ ་་་་་༦་་་་་ཛི 1་་་་་་་་་་ཅྭ་ ་རྨྱུ་་་་1 1་་་་་་་ཁྱི་་་་༦་་་་ཛི1 1་་་་་་་་་་་་་་གོ་་1 ཁུ་་་་ཁུ༷་ ་་་་་ཁུ་་ཟི། 1ཁྱི་་་་་་་་་་་་་་་་་་་་་1 1ཁྱི་་་་་ཁུ་་་་་་་་་་རྩི་་། 1ཁྱི་་་་་་་ཁྲོ་་་་་་ཁུ་་་་་སྤྱ1 །་ཁུ་་་་་ཕྱུ་་་་་་་་་་་ཟློ1 1་་་་་་གློ་་་་་་་་གློ་་་་གློ1 1་་་་་གོ་་་་་་་་ཛི་་་་་རྐྱུ1 །4་་་༷་ ་་་་་་་་ཁུ1 1་་་་་ཁྱི་་་་་གློརྨྱུ་ ་་་ཁྱི1 1་ཁུ་་་་་ཁྱི་་་་་ ་ཙྭ་་ཟློ་་1 ་ཅྭ་ ་་་་་་་་སློ་་གོ་་་སློ1སློ་་ཁྱི་་་་་ཁུ་་་ ་་་་་་༦་་་་་་་་་་་11 །1་་་་་།་་ ་་་་་་སྟུ་་་་་་་་་་་་ཛི་1་་་ཁྱེ་ཁྱི་་་་་་་་་་༷་ཅྭ་ ་་་་་་1 1་་སྩོ་་་་་་ཟློ་སྤྱོ་་་་་རྙུ་་1 1གྱོ་་་་་སྟུ་སྨོ་་་་་་་་་1 1་ཟློ་་་རྩོ་་ཁྱེ་་་་་་་1 1་་་་་་་་་་་་་ཚོ་་་་་་་1 །་་་་་་་་་་་་་་་་་ཟི་་་སྨོ1་1་་་་་་་་་ཁྱེ་་་་་་་སྔོ1 1སྟུ་་་་་༷་་སྤྲེ་་་་་་་་་་་1 1་ཁྱེ་་སྤྲོ་་་་་་་་་་་ཁྱེ་་་་ཛི་་་1 1སྟུ་སྦྱི་་ཟློ་་་ཁྱེ་་་་་་་་སྐྱི་1 ཉི་ཁུ་་་་་ཛི་་་་་ཟི་1 1་ཁུ་་རྙི་རློཆི་་ཉི་་་་་་་་1 1་ཐོ་་་་ཚོ་་་་་ཁྱེ་་་་་སྔོ1 
1་ཟི་་ཁུ་་་་ཁྱི་་་་་་་་སྟུ་་་་་ཛི་གློ 1་་་་ཁྱི་་་་་་་་ཅྭ་ ་རྨྱུ་་་་། 1་་་་1 །་ཁྱི་་་་་་་ཏྲུ་་་་་་་་་སྐྱི་་་1 1་་་་ཁྱི་་་ཆི་་་་་་་ཇོ་༦1 1་་་་་་་་་་་་་་་་་་་་་་་། 1་ཏཱཔྱོ་་ཁྱི་་་ཆི་་་་་་་་་་་་།་་་་་་་་ཙི་་་་་་་་་ཁྱི་་། ་་་ཁུ་སྨོ་་་་་་་་་1གོ་་་་་་་་གློ་་ཁུ༷་ ་་་་་་1 1་་་་སྩོ་་་ཁྱི་་་རྭ་་ཁུ་ལྗོཁུ1 1བྱུ་ ཁྱི་་་་༷་་་་་་་་ཁྱེ་་ 1་་་་་་་ཁྱི་་་་་་་ ་ཁུ1 ་་སློ་་་་་་་གློ་་་་་་་་་གློ་བྱུ1 །་་་་་་་་་གློ་་་་་་་་་་1 1ཟློཁུ་་་་་་་་་་ཁྱི་་་་སྦྱི་་མི་1 1་་་་་་ཁྱི་་་་་ཇོ་་ཁྱི་ཁུ1 1་་ཚོ་་ཙྭ་་་ཁུ་་་་་་བྱུ་་་་ཆི1 1་་སྩོ༷་ཁྱི་་ཁྱི་རྨྱུ་་་་་་་་་།་་་་་ཁྱེ་་་་་་་ཁྱི་སྐྱུ་བོ1 །་་་་་་ཚོ་་་ཁྱེ་་ཁྱི་ཟི་་་་1 1་་་ཟློ་་་་་ཀེ་ ཏོ་་་་་་་་་1་1་་་ཁྱི་་་་ཁྱི་་་་་་ཁྱི་་་་1 1ཁྲི་་་་་ཁྱི་་་་ཁུ་སྒྲི་ཁུ1 །ཁྱིསློ་་་་་་་བྱུ་་་་་་་་་་1 1་སྣྲི་་་་་་་་་་་་་ཛི་་་་་་1 1་་་་་་ ་ཟློ་སྟུ་་༦་་་་ཁྱི་་་་་་་1 1་་ཁྱེ་་རྭ་་་་་་ ་ཁྱི་་་་་་་་ 1་་་་སྤྱོ་ཁྱི་གོ་་སྤྱོ་་་་རྭ་1 1་་་་་་སྨོཙི་ རྩོ་ཧཱུཾ༷་་་རྗོ་སྣྲོ1 1་་་་ཛི་་་ཁུ་་་་་་་་་་་་་་་་་1 1་ཁུ་་སྐྱི་་་སྙོ་ཁྱུ་ཁྱི་་་་་་་རྟོ1 1སྩོ་ཁྱེཀུ་ ཁྱི་་༷བྱུ་ཁུ1 ་ 
།་རྗོ་སྐྱུ་་་་་༦་་་་1 1སྟུ་ཛི་་་ཁྲུ་་ཅུ་་་་་་་་་1 1ཁྱི་་་་་ཁྱི་་་་་་་་་་་1 1་་་་་་ཁུ་་་་་་སྦྱི་་་ཇོ་་་ཇོ་1 1་་་་་་་་་་ཛི་་་་་་་་་་ཁྱི1 1་སྐོ་་་་་་ཁྱི་་་་་་་་་་་་་་་1 1་་་རྨྱོ་་་་་་་་་་་་་་་་་1 1་ཚོ་་་་་་་་་་་་་་་1 1ཚོཁྱི་ཁྲི་་ཁྱི་སྨོཁྱི་་་་་་ཞུ་་༦་1 །་་་་ཁྱི་་་ཁྱི་་ཉུ་་སྟུ་་་་་་་1 1་་་ཅུ་་་་་་་་༷་་་སྟུ་་་སྤྲོ་་ཁྱི1 1་་་་སྒྲོ་་་་་་་་་ཁྱེ་་་ཉི1 1ཁྱི་་་་་་་་་་་་་་་་་་1 1་༦་་་་་་་ཆོ་4་་ཅྭ་་་་་གློ1 ་ཏྲི་་་་ཁྱི་་ཟི་་་་་་བྱུ་་་1་1་སྒྲོ་་་་་་་་་་ཁྱེ་་་ཟླི1་1ཁྱི་་ཟློ་་སྨོ་་་་་་་་པྱོ་་་་་་1 1་་་སྤྱོ་་་་་་་་་་་་་་་1 1ཁྱི་་་་་་་་་་ཁུ་་་་1 1་་་་་་ཁྱི་་་་་་༦་་་ཁྱེ་་ཉུ་1 1ཁྱི་་་་་་་་་་་་་་་་་ཚོ་་་་་1 1་་་་སྡུ་ ་་་་་ཛི་ཏོ་་་་1 ་་་་སྐྲུ་་྅་་་་་ཇོ་ཁྱི་་ཁྱི1 ་ཁྱེ་་ཚོ་་་་་་་་་་་་་ཁྱི་1 1་སྟུསྤྱོ་་་་་་་ཁྱི་ཕྱུ་་་་་་་་་་་་ཁྱི1 1་་ཁུ་་་་་་་་་་་་་་་1 1ཁྱི་་་་་སྟུ་་་་ཐོ་ཁུ་་་་་་་ཏྲི1 1ཛི་་་་་ཁུ་་་་་་སྣྲོ་་་་་1 1་་་་ཁྱི་་་་་་་ཉུསྤྱོ་ཅི་་་བོ1 1རྨྱུ་་ཁྱི་་་་་་རྩི་་་་་་ཇོ1 1་་ཁྱི་་་་་་་་་་་་་གྱི།1 ་ ་11་་་་ཉུ་་སྦྱིཙི་ 
ཁྱི་ཛི་་་གློ་་་་་་ཁྱི་་་ 1ཁྱི་་་་་་་་་་་ཁྱི་་་1 །་་་་་༦་་་་་་་་་་་་་་1 1ཁྱི་ཁུ་་ཁུ་་་་་ཁུ་་་་་་ཉུ1 1་་་་་་་སྣྲི་་་་་༦་་་་་ཚེ་1 1་ཟི་་ཚི་་་་་་་་སྦྱི་་རྨྱུཁྱི1 ་་་་་་རྨྱེ་ཉུ་ ་་་་་ཛི་་། 1་་་་གློ་་ཛི་་་་་་་་་་་ ། ་1་་་་་ཟི་སྤྲོ་་་་་་་་་་་༷། ་་་་་ཏོ་་་་ཁུ་་་༷༑1 1་་་་་ཁྲུ་སྤྲོ་་་མི་་་་1་།་་་་་་སྦྱི་ཉུ་་་་་་་1 ལྗུ1 ་་་་་་སྩི་་་་་་ཉུ་་་་་་ཟློ་1 །་་སྦྱོལྗོ་་་་་་་་་་ཞུཚོ་། །་ཟིརྭ་ ་་ཁུ་་་ཁུཀྱུ་་་་སྤྱ་། 1་་་་་་ཕྱུ་་་་་ཙྭ་་ཁུ་ 1་ཟི་་་ཆི་་་་་་ཁུ་སྒྲོསྦྱོ་་་་1 1་་་་་ཟློ་་་་་་་་་་་་1གོ་ཟི་་་་་སྒྲི་་་་༦་་་་ཁྱེ་1 1་་་་སྤྱཟློ་་་་་་་ཁྱི་་་་་་་། 1་ཟི་་་་་་་་སྟུ་ ་་་་་། 1་་་་་་་་་་་་་ཛི་་་་་་1 1་ཟི་སྟུ་་ཁྱེ་་་་སྤྲི་མོ་་་་1 1ཇོ་་་་་རྙུ་་་་་ཁུ་་སྨོ་་ཁུ1 ་་ཁུ་་་་་་ཉུ་་་་་་་་་།་1་་་་ཁྱེ་བྱུཁྱེ་་་་་་་་་། ་་་་་་་ཙྭ་ཙུཁུ་་གློ་་་་ཟི་1 1་་་་་ཁུ་ཁུ་་་་་་་་་་སྨྲ 1་་་་་་་གོ་་་་་་་་་་་་་་རློ་1 1ཚོ་་་་ཛིསྨོ་་ཅུ་་་་་་ཁུ1 1་ཟི་་་་་༷་་ཁྱི་་་ཁུ་སྐྱི་་་་ཅོ། །སྐྱི་་་་ཚོ་་་ཛི་་་་་ཙོ1 1རྙུ་་་་་་ཁྲུ་་་་སྦྱིསྤྲི1 1རྨྱུ་་་་་ཟོ་་་ཚོ་གློ་་ཁྱི་་་ཁུ1 1༷ཁུ་་་་་་ཁྱེ་་་་་་ཁུ་་་་་གོ 1ཁྱི་་་་་་་1

1་་་་སྨི་་་་་ཁྱི་་་་1 །་༷༷རྩོ་རྨྱི་4་ཁུ་་རྙི་ ་་་་རྨྱི༷༷་་་1 །ཁྱི༔རྨྱི༷་་ཅྭ་ ་་ཅྭ་ ་་། 1་་ཁུ་་4་་ཁྱི་་་་་གློ་་་་་་ཁུ1 14་་་་་་་གློ་་ཁུ་་་་༦་་ཁུ1 1་་་་་་་ཁུ་་་་རྨྱུ་་་་་ཁྱི1 ་ཟི་་་་་ཁྱེ་་ཁུ་་་༷་་་་ཉུ་1 ་༷ ཇོརྩོ་་་་ཅུ་་་་རྨྱི་་ཟློ་་་།་1སྐྱི་་་་་སྟུ་་་4་་་ཁྱི་་སྐྲུ་་་་1 1་་་་་་་་ཛི་་ཁུ་་་་་་་1 1་་་་་ཆོ་ཉུསྤོ་་སྨོཛི་་་་་་་ཁྱི་1 ་1་་སློ་གློ་་་་་་་་་་་ཁྱི་་་༦་། 1ཙྭ་་་་་་མི་་་་་སྨྱོ་་ཛི་་1 །་་་་་་རྙུ་་་་་་་་་ཁྱི། 1རློ་ཇོ་་་་་་་་་་་་་་གོ 1་སློ་་་་་་་་་ཁྱེ་་་་་ཆི1 1ཅུ་་ཙོ་་་ཇོ་སྦྱོཁྱི་་་་ཁུ་་1 1ཁྱི་་་་་་་་་་་་་སྣྲོ་་་་་ཁྱི་1 1་་་་་་་་་་ཛི་་་་་་་་རྩོ1 །་་་་ཚི་་་་་་་་་ཛི་་སྣྲོ་་1 །་་་་་་་ཅོ་་ཁྱིབྱུ་ ་་་་སྐྲུ་་྅1 ་་་་་ཁྱི་་་་་་་་ཁུ་་ཙོ་་་1 1ཉུ་་་་་་་་་་་་་ཁྱེཅྭ་ ་གློ།་1་ཁུ་་་་་4་༷་ ་་ཚོ་་་་་བྱུ། 1ཉུ་་་་་་་་་་་་་་ཉུ་་་་ །ཁུ་༷་་་་་་་་་་་་་ཁྱི་་ཁུ་་1 1ཁྱི་་་་ཁུ་་་་་་་་་་་་་ཁྱི1 1་་་་་་་་་་་་་་་་་་སྟུ་ ་1་1་ཁུ་་་༷་ སྟུ་༔་་ཉུ་སྣྲིསྣི1 1་་་་་་སྦྱི་ཉུ་་་ཅི་་སྤྱོ་་སྨྲོ། 1་་ཁྱི་་་་་བྱུ་་སྟུ་ཟི།1 ་།1་་་་་ཁྱེ་་སྦྱིརྟོ་ ་་་ཛི་་ཁྱི་་་ཛི་་་་སྤྲི་་་་་་་་་ཚོ་གློ 1་་་་་ཁྱི1 
1་་་་་་་་1 1་་་་་་་་་་་ཉུ་་ཟི་་1 1་་་་་་་་་་྅་་་་་ཇོ་1 1་་་་་་་་་་་་་་་ཁུ་་སྔོ་་་་། 1་་་་་་ཁུཛི་་་་་་་4་་ཁུ1 1་་་་་་་་4་་ཁྲི་་་་་་ཆི་། །་་་་་་་་་་་་་་་་་ ་་1 1་ཁྱེ་་་་་་་སྐྲུབྱུ་ ་་སྙོ་་་་་་་་། 1ཟློཆི་་་་་་་མོ་་་་་་་་༦་་ཏོ1 1རྨྱུ་་་་་་་་་་་་་་་ ་་་་རྩོ། ་།ཁྱི་་་་་ཁུ་་་་ཞུརྗི་་ཀེཇེ་་་་། 1་་་་་་ཉུ་་་་༦་ལྗོ་་ །སྦྱི་རྙི་་སྦྱི །རྨྱི་རྙོ་རྨྱི 1རྨྱི་རྨྱི་རྨྱི 1སྤྲོ་་་་་་་་ཁུ1་1་་་་་་་་་་་་་་་་སྟུ་་༷་1 །་་་་་་་་བློ་་་་་བྱུརྙུ་ ་1གོ་་་་་ཁྱེ་་་་ཁྱེ་་་་་་༦་༷་སྦྱི1 1་་་་ཉུ་་་ཐོ་ཁྱི་་ཅི་་་་་་་་1 1ཁྱི༷་༷་་༷་༷་་༷་བྱུ༷བྱུ༷ ་༷ སྨོ་་་1 1་་་་་་ཁྱི་་་་་་་་་་། 1ཁྱི་་་་ཉུཕྱུ་་་་ཁྱི་་། །་ཁུ་་་་་ཁུ་་་་ཁུཁྱི་་་་་་ཁུ་1 ་རྭ་་་་་་༷་་་་ཁུ་་་་ཅྭ་1 ་།་་་་་་་་་་་་་་་་ ་་་་ ་་་་1 1་སྟུ་་་་་་་་་་་༷་་སྟུ་་་་་་་་1 1་་་་༷་ གློ་སྟུ་་་་་་ཁུ་་༑ 1ཁྱི་་་་་་་4་་་་་4་་་་་་་་་། །་་་་་་་་་ཆི་་་་་་ལྗོ་་་་1 1སྤྲོ་་ཉུ་ ་་་་་ཁྱི་་ཁྱིསྦྱི༷ཀེ 1སྣོ་ཁྲུཙྭཁྱི་་་ཟི་ཛི་་སྤྲོ1 1ཛི་་་སྤྲི་་་ཁྱེ་་་་་་་་་་་ཕྲི། །་་་་་་སྦྱོ་ཁྱེ་ཅྭ་ ་གློ་་སྐྱི་་ཁུ 1་༷༷རྨྱུ་སྟོ༷་4་ཁྱི་་་་ཟོ་་སྦྱི༷༷༷་་་་།༷༷་་སྐྲུ་སྦྱི་་སྦྱི་རྨྱོཅི་ཧཱུཾ་་
