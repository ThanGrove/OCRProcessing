ས་མ་ཡ་ཙྭ༄༅། །ཨ་ནུ་རྭ་ག་ཡུ་མི། ཨ་ནུ་རྭ་ག་ཡ་ཧུ༷། ཛ༷༷ཧུ༷་བ༷་ཧོ༷ ཧུ༷་ཧུ༷་ཧུ༷། གཉིས་སུ་མེད་པར་མཉམ་སྦྱར་ནས། །བདེ་བ་ཆེན་པོའི་བྱིན་བརླབས་ལས། །ཐམས་ཅད་དབང་བསྐུར་སྒྲིབ་སྦྱངས་ནས། །ཉོན་མོངས་སྦྱངས་ཤིང་བྱང་སེམས་འཕྲོ། །དྲག་པོ་ལ་སོགས་རྣམ་བཞི་ཡིས། །ཇི་ལྟར་འདུལ་བ་བཞིན་སྤྲོས་ན། །དཀར་དམར་གནག་དང་སེར་བ་དང་། །དུས་རྣམས་ཀུན་ཏུ་རབ་སྦྱར་ནས། །ཡང་དག་ཉིད་དུ་རྣམ་པར་སྤྲོ། །འདི་ནི་སངས་རྒྱས་བྱིན་བརླབས་ཡིན། །ཐབས་དང་ཤེས་རབ་གཉིས་མེད་པར། །དུས་གསུམ་སྔ་ཕྱི་མེད་པར་སྦྱོར། ཧུ༷་ཧུ༷་ཧུ༷། ཤཱརྙིང་ཀུ་རུ་སྣ་ཧྭ། བུཁོ་ཀུ་རུ་གྲྭ་ཧྭ། བ་ཤང་ཀུ་རུ་སྣ་ཧྭ། མྱ་ར་ཡ་ས་མ་ཡ་ཧུ༷་ཕཊ།ཨོ་བཛྲ་ཀི་ལི་ཀི་ལ་ཡ་ཧུ༷་ཧུ༷་ཧུ༷། རྣམ་བཞིས་འདུལ་བ་གང་ཡིན་པ། །དེ་ནི་སངས་རྒྱས་མཛད་སྤྱོད་ཡིན། །རྔམ་པ་ཆེན་པོའི་ལྷ་ཚོགས་ནི། །འཇིགས་བྱེད་ཆེན་པོ་འདི་ལྟ་བུ། །དྲག་པོ་རྔམ་པའི་ཧུ༷་སྒྲ་ཅན། །དེ་ཡིས་སྟོང་གསུམ་རབ་ཏུ་གཡོས། །ཡབ་ཡུམ་སྲས་དང་བཅས་པ་རྣམས། །རྔམ་པའི་ང་རོ་དྲག་པོར་སྒྲོགས། །ཧུ༷་བཛྲ་ཀི་ལི་ཀི་ལ་ཡ་ཧུ༷་ཧུ༷་ཧཱུཾ། ཞེས་རྔམ་ཞིང་ངོ་མཚར་བའི་ཁྲོ་བོ་རབ་ཏུ་འཕྲོས་ནས། ཐམས་ཅད་རབ་ཏུ་འདུས་ནས་སྣང་ལ་རང་བཞིན་མེད་པར་གྱུར་ནས། མཉམ་པའི་ངང་གིས་བཞུགས་སོ། །རྡོ་རྗེཕུརཔཆེནཔོགཏོརམའི་རྒྱུད་ལས། གླེང་བསླང་བའི་ལེའུ་སྟེ་གཉིས་པའོ།།

ཨོ་བཛྲ་ཀི་ལི་ཀི་ལ་ཡ་ཛ༷༷ཧུ༷་༄༅། །བ༷་ཧོ༷༷ སརྦ་བིསྣ༷་བཛྲ་ཀི་ལི་ཀི་ལ་ཡ། ཧུ༷་ཧུ༷་ཕཊ་ཕཊ། ཨོ་བཛྲ་ཀྲོ་དྷ་ཧུ༷་ཀྭ་ར་ཧུ༷་ཕཊ། ཧྭ༷་ཀྭ་ར་ཧུ༷་ཨ་ཨ། དངོས་གྲུབ་བར་ཆད་བྱེད་པའི་བགེགས། །གདུག་ཅིང་སྡང་སེམས་ལྡན་པ་རྣམས། །ཟུང་ཞིག་ཆིངས་ཤིག་རྣམ་པར་སྡོམས། །གཏོགས་ཤིག་གྲོགས་དང་བྲལ་བར་གྱིས། །རིང་པར་ཁུག་ལ་སྨྲར་ཆུག་ཅིག །ཕོབ་ཅིག་རྣམ་པར་གཟིར་བར་གྱིས། །ང་ཡི་དབང་དུ་གྱུར་ནས་ཀྱང་། །བསྒོ་བའི་བཀའ་རྣམས་ཉན་པར་གྱིས། །པ་ར་བིབྱུན་ཧུ༷་ཕཊ། ཚིན་དྷ་ཚིན་དྷ་ཧུ༷་ཕཊ། བིན་དྷ་བིན་དྷ་ཧུ༷་ཕཊ། གྲི་ཎྜ་གྲི་ཎྜ་ཧུ༷་ཕཊ། ཧ་ན་ཧ་ན་ཧུ༷་ཕཊ། བན་དྷ་བན་དྷ་ཧུ༷་ཕཊ། ད་ཧ་ད་ཧ་བཛྲ། པ་ཙ་པ་ཙ་ཧུ༷་ཕཊ། མ་ཐ་མ་ཐ་བཛྲ། སུ་རུ་སུ་རུ་བཛྲ། མུ་རུ་མུ་རུ་བཛྲ། ཛ་ལ་ཛ་ལ་བཛྲ། ལས་ཀྱི་སྔགས་རྣམས་དྲིལ་དྲིལ་ནས། །སྙིང་དང་དབང་པོ་རྣམ་ཤེས་ལ། །རིམ་པ་བཞིན་དུ་ཕུར་པ་ཐོབ། །སྤྱི་མཐུན་ཀུན་ཏུ་མ་ལུས་གདབ། །སྟོང་པ་ཁོ་ནར་ཡང་དག་སྦྱོར། །ཆོས་ཉིད་ཀྱི་བདེན་པའོ། །གསང་སྔགས་ཀྱི་བྱིན་གྱིས་བརླབས་སོ། །སངས་རྒྱས་ཀྱི་མཐུའོ། །ཨ་སྟི་ཙེརུའི་དུས་ལ་བབ། །གནག་པ་དེ་ཁོ་ན་ལའོ། །དཀར་ནག་མཚམས་སུའོ། །ནག་པོ་གྲུ་གསུམ་གཏོར་མ་ལ། །མཚོན་རྩེ་གཉིས་དང་དབྱིག་དུག་ཟངས། །གཞན་གྱི་དངོསཔོམཚོནརྣམསལ། །བསྒྲུབ་པཚུལབཞིནབྱསནསནི། །མཚན་མ་བརྟན་པོ་བྱུང་བ་དང་། །དགྲ་བགེགས་དབང་པོ་མངོན་སུམ་བསྒྲལ།
