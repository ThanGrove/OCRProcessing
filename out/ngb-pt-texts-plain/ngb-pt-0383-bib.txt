རྒྱ་གར་སྐད་དུ། མ་ཧྭ་ཀྭ་ལ་ཏནྟྲ་མ་ག་ལ་ཀྭ་ལ་ནཱ་མ། བོད་སྐད་དུ། དཔལ་ནག་པོ་ཆེན་པོའི་རྒྱུད་དུར་ཁྲོད་ནག་པོ་ཞེས་བྱ་བ། དཔལ་ཀུན་བཟང་ཧེ་རུ་ཀ་ལ་ཕྱག་འཚལ་ལོ། །འདི་སྐད་བདག་གིས་ཐོས་པའི་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་དུར་ཁྲོད་བསིལ་བའི་ཚལ་ན། ནག་པོ་ཆེན་པོ་འཁོར་མ་མོའི་ཚོགས་དང་བཅས་པ་ལ་ཆོས་སྟོན་ཏེ་བཞུགས་སོ། །དེའི་ཚེ་བཅོམ་ལྡན་འདས་ལ་ནག་པོས་གསོལ་པ། བདག་གི་སྐྱེ་རྒྱུད་ཇི་ལྟར་ལགས། །ཁས་བླངས་དོན་ནི་ཇི་ལྟར་ལགས། །རྟགས་དང་མཚན་མ་ཇི་ལྟར་ལགས། །སྒྲུབ་པའི་རྫས་ནི་ཅི་ལ་བྱ། །སྒྲུབ་པའི་དཀྱིལ་འཁོར་ཇི་ལྟར་ལགས། །དེ་སྐད་ཅེས་སྨྲས་པ་དང་།

འབུམ་དུ་བྱས་ནས་སྤྱན་ཕྱེ་༄༅། །ལ། །སྔར་བཤད་པ་ཡི་བྲིས་སྐུ་རྟེན། །དེ་ལ་གྲི་གུག་ཐོད་པ་དང་། །ཐང་ཀས་དབང་བསྐུར་ཚོགས་ལ་བཤད། ནག་པོ་ཆེན་པོའི་རྒྱུད་དུར་ཁྲོད་ནག་པོ་ལས་དབང་བསྐུར་བའི་ལེའུ་སྟེ་གསུམ་པའོ།། །།ནག་པོའི་ལྷ་རྒྱུད་ཇི་ལྟར་ལགས། །བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། །སྐུ་དང་གསུང་དང་ཐུགས་ཀྱི་ནི། །ལྷ་རྒྱུད་བསྟན་གྱིས་མཉན་པར་གྱིས། །ཧུ༷་སྲིན་པོའི་གཟུགས་ཀྱི་གདུག་པ་ཅན། །ཚར་གཅོད་བྱང་ཆུབ་སེམས་དཔའ་སྟེ། །གྲོང་ཁྱེར་སུམ་བརྩེགས་འཇིགས་བྱེད་བདག །ག་་པོ་ཁྱོད་ལ་བདག་བསྟོད་དོ། །ཧུ༷་མདོག་ནག་གཟུགས་ཐུང་གཟི་བརྗིད་ཆེ། །གདུག་པ་ཚར་གཅོད་གྲི་གུག་གཡས། །གཡོན་ན་ཐོད་པ་ཁྲག་བཀང་སྣོད། །ཐོད་ཕྲེང་ཅན་ལ་བདག་བསྟོད་དོ། །ཧུ༷་ཁྲོས་པའིཞབས་བརྡབས་ས་གཞི་གཡོས། །འཇིགས་པའི་གད་མོས་རི་རབ་བསྤྱིལ། །དམ་ཚིག་ཉམས་པའི་སྙིང་འབྱིང་པ། །དམ་ཚིག་ཅན་ལ་བདག་བསྟོད་དོ། །ནག་པོ་ཆེན་པོའི་རྒྱུད་དུར་ཁྲོད་ནག་པོ་ལས། ལྷ་རྒྱུད་བསྟན་པའི་ལེའུ་སྟེ་བཞི་པའོ།། །།གཏོར་མའི་ཆོ་ག་ཇི་ལྟར་ལགས། །བཅོམ་ལྡན་འདས་ཀྱི་བཤད་དུ་གསོལ། །བཅོམ་ལྡན་འད་ཀྱིས་བཀའ་སྩལ་པ། །ལྷ་ནི་མཉེས་པར་བྱེད་པ་དང་། །བར་ཆད་ཞི་བར་བྱ་བའི་ཕྱིར། །གཏོར་མ་བཤད་ཀྱིས་མཉན་པར་གྱིས། །ཨ་ལས་བྱུང་བའི་ཐོད་པ་ཡི། །ནང་དུ་སྔར་བཤད་རྫས་རྣམས་བསམ། 
།འོག་ཏུ་ཡ༷་ལས་རླུད་བསམ་ལ། །དེ་སྟེང་ར༷་ལས་མེ་འབར་བས། །གཏོར་མའི་སྟེང་དུ་ཉི་མ་བསམ། །དེ་སྟེང་ཧུ༷་ལས་གྲི་གུག་བསམ། །དེ་ལས་འོད་ཟེར་གཉིས་སུ་འཕྲོས། །འོག་མས་གཏོར་མ་བདུད་རྩིར་བསྒྱུར། །རླུད་གིས་མེ་སྤར་ཐོད་པ་བསྲོས། །སྔོ་དང་དམར་དང་དཀར་པོར་བསྒྱུར། །དེ་ནས་སྟེང་གི་འོད་ཟེར་གྱིས། །རང་བཞིན་གྲུབ་པའི་བདུད་རྩིར་གྱུར། །གྲི་གུག་ཉི་བཅས་འོད་དུ་ཞུ། །དེ་ནས་སྔགས་འདི་བརྗོད་པར་བྱ། །ཨོ་མ་ཧྭ་ཀྭ་ལ་ཡུ་ཙྭ་ན་ཨ་ཕྲི་ཏ་ཨོ་ཨུཿཧུ༷་ཧུ༷་མ། དེ་ནས་སྔག་ཀྱི་དབུལ་བར་བྱ། །ནག་པོ་ཆེན་པོའི་རྒྱུད་དུར་ཁྲོད་ནག་པོ་ལས། །གཏོར་མ་ཆོ་ག་བསྟན་པའི་ལེའུ་སྟེ་ལྔ་པའོ།། །།སྔག་རྣམ་ཐམས་ཅད་བཤད་དུ་གསོལ། །བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། །ལས་རྣམས་ཐམས་ཅད་འགྲུབ་པར་བྱེད་པ། དྲག་བྱེད་ཐམས་ཅད་གནོན་པར་བྱེད་པ། བགེགས་ཐམས་ཅད་རླག་པར་བྱེད་པ། གདོན་ཐམས་ཅད་སྒྲོལ་བར་བྱེད་པ། སེར་བ་ཐམས་ཅད་གཅོད་པར་བྱེད་པ་དང་། ཆར་པ་དབབ་པ་དང་། བཅད་པ་དང་། ཡི་དྭགས་སྐྲོད་པ་དང་། མ་མོ་དང་འབྱུང་པོ་ཐམས་ཅད་དབང་དུ་སྡུད་པ་དང་། རིག་སྔགས་རྦོད་གཏོང་གཅོད་པ་དང་། ནག་པོ་ཆེན་པོའི་སྙིང་པོ་དང་འབོད་རྦད་བསད་གསུམ་གྱི་སྙིང་པོ་དང་། འཁོར་ལོའི་སྔགས་བརྗོད་པར་བྱ་ཡིས་ཉོན་ཅིག །ཨོ་ཤི་མ་ཧྭ་ཀྭ་ལ་ཧུ༷་ཕཊ། ཨོ་མ་ཧྭ་ཀྭ་ལ་ཤཱ་ས་ནཨུ་པ་ཀྭ་རི་ནི། 
ཨེ་ཥའ་པསྩི་མ་མ་ཀྭ་ལོ་྅་ཡ་མི་རྡ་རཏྣ་ཏྲ་ལྭ་ཡཨུ་བ་ཧ་རི་ཁི་ཡ་༄༅། །ཏི་པྲཙྭ། སྨུ་ར་སི་ཏ་དྭ། ཨི་ད༷་དུཛི༷་ཁ་ཁ་ཁུ་ཧི་ཁུ་ཧི། །མུ་ར་བཱ་ར། གྲིཎྜ་གྲིཎྜ། བྷནྟྲ་བྷརྙ། ད་ཧ་ད་ཧ། པ་ཙ་པ་ཙ། དི་ན་མེ་ཀེ་ན་ཧུ༷་ཕཊ། ཨོ་ཡ་མ་རྭ་ཛ་ས་དྭ་མེ་ཡ། ཡ་མེ་དོ་སྲ ཎ་ཡོ་ད་ཡ། ཡ་ད་ཡོ་ནི། ར་ཡསྟེ་ཡ་ཡཀྟ་ཡསྣྲོ་ནི་ར་མ་ཡ་ཧུ༷་ཕཊ། ཨོ་ནི༷སྐྱི༷༷ཝ་ཀྲི་ཏྭ་ནན་ཧུ༷་ཕཊ། ཨོ་ཡ་མྱརྙ་ཀ་ཧ་ན་ཧུ༷་ཕཊ། ནག་པོ་ཆེན་པོ་རྒྱུད་དུར་ཁྲོད་ནག་པོ་ལས། སྔག་ཀྱི་ལེའུ་སྟེ་དྲུག་པའོ།། །།ཐུག་དམ་བསྐུལ་བ་ཇི་ལྟར་ལགས། །བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ༄་པ། །ནག་པོ་ཆེན་པོ་ལས་ལ་བསྐུལ་བ་ནི། །ང་ཡིས་བཤད་ཀྱི་ཁྱོད་ཀྱིས་ཉོན། །འདས་དང་མ་བྱོན་ད་ལྟར་གྱི། །བསྟན་པ་སྲུང་བར་སྐྱེད་བྱེད་པ། །ནག་པོ་ཆེན་པོ་ཁྱོད་ནི་ཡིན་པ་སྟེ། །གདུག་པ་ཚར་གཅོད་འཇིགས་པའི་གཟུགས། །རལ་པ་ནག་པོ་གྱེན་དུ་གཡེང་། །སྟེང་གི་ལྟ་ཚངས་འཇིགས་པར་བྱེད། །སྤྱན་བགྲད་ལྗགས་འདྲིལ་བར་གྱི་ལྷག་དོན་བསྡིག །གཡས་པ་གྲི་གུག་གནོད་སེམས་ཅན་ལ་སྡང་། །གཡོན་པ་ཐོད་ཁྲག་གནོད་སེམས་ཁྲག་གང་གསོལ། །ཞབས་བརྡབ་གནོད་སེམས་ཅན་གྱི་ལས་ལ་ཉེད། །གང་གི་དཀོན་མཆོག་གསུམ་ཁྲོ་དང་། །བླ་མ་བརྙས་བྱེད་སེམས་ཅན་འཚེ། །བདག་གིས་ད་སྒོས་ཀུན་ཁུག་ལ། །ལིསྐོ་འདི་ལ་མྱུར་དུ་ཆུག །འདི་ཡི་ཤ་ཟོ་ཁྲག་འཐུང་ལ། །རུས་པ་ཀླད་པ་ནང་རོལ་ཟོ། །རིགས་རྒྱུད་རྩད་ནས་ཆོད་ལ་ཤོག །མཛའ་བོ་ཀུན་དང་ཕྲོལ་ལ་ཐོང་།

གསུམ་པའི་བཞི་པའི་མཐའ་མ་གཉིས།ལྔ་པའི་དང་པོ་དང་། མཐར་གནས་པ་ཡི་བཞི་པ་དང་། །མ་ནིང་བཞི་ཡི་དང་པོ་དང་། །བཞི་པའི་མཐའ་མ་ལ་དབྱངས་ཡིག་བཅུ་གཅིག་པ་སྦྱར་བ་དང་། དབྱངས་ཡིག་བཅུ་གཅིག་མཐའ་མ་ལ། །མཐར་གནས་པ་ཡི་གཉིས་པ་ལ། །གསུམ་པའི་དང་པོ་འོག་ཏུ་གདགས། །དབྱངས་ཡིག་གསུམ་པས་མནན་པ་དང་། །ཡང་ན་ཡི་གེ་བཞི་པའི་སྒྲ། ་་ག་པ་དང་པོ་དང་། ལྔ་པའི་ཐ་མ་དང་། དབྱངས་ཡིག་གསུམ་པས་མནན་པ་དང་། བཞི་པའི་གཉིས་པ་ལ། བཞི་པའི་དང་པོ་གདགས་པ་དང་། བཞི་པའི་དང་པོ་ལ། འོག་ཏུ་མེའི་ས་བོན་སྤྱིན། དེ་ནས་བཞི་པའི་དང་པོ་ལ། །བཞི་པའི་མཐའ་མ་གཞག་པར་བྱ། །ཡང་ན་བཞི་པའི་དང་པོ་ལ། །འོག་ཏུ་མེ་ཡིས་ས་བོན་གདགས། །ཨརྙསྦའི་དང་པོ་དང་། ལྔ་པའོ་དང་པོ་དང་། མཐར་གནས་པ་ཡི་བཞི་པ་དང་། མ་ནིང་བཞི་པའི་གཉིས་པ་དང་། གསུམ་པའི་མཐའ་མ་ལ། ལྔ་པའི་མཐའ་མ་བཞག །ཨརྙསྦའི་དང་པོ་དང་། །བཞི་པའི་དང་པོ་ལ། །དབྱངས་ཡིག་གསུམ་པ་བཞག་པའོ། །ལྔ་པའི་དང་པོ་ལ་འོག་ཏུ་མེའི་ས་བོན་གདགས། བཞི་པའི་དང་པོ་ལ། དབྱངས་ཡིག་གསུམ་པ་བཞག་པ་དང་། །དེ་ནས་གཉིས་པའི་གསུམ་པ་བརྗོད། །མཐར་གནས་པ་ཡི་གསུམ་པ་ལ། །ལྔ་པའི་ཐ་མ་འོག་ཏུ་གདགས་པ་དང་། །ཨརྙསྦའི་གཉིས་པ་དང་། །མཐར་གནས་པ་ཡི་གསུམ་པ་ལ། །དབྱངས་ཡིག་གསུམ་པ་སྦྱར་བ་དང་། གསུམ་པའི་དང་པོ་ལ། 
དེ་ཡི་མཐའ་མར་ལྡན་པ་དང་།༄༅། །དེ་ནས་དབྱངས་ཡིག་གསུམ་པ་དང་། གསུམ་པའི་གསུམ་པ་ལ། ལྔ་པའི་ལྔ་པས་མཚན་པ་དང་། བཞི་པའི་གསུམ་པ་ལ། དབྱངས་ཡིག་ལྔ་པ་སྦྱར་བ་དང་། མཐར་གནས་པའི་དང་པོ་ལ། དང་པོའི་གཉིས་པ་ལན་གཉིས་བརྗོད་པ་དང་། དེ་ནས་དང་པོའི་གཉིས་པ་ལན་གཉིས་བརྗོད་པ་དང་། མཐའ་གནས་པའི་བཞི་པ་ལ། དབྱངས་ཡིག་གསུམ་པས་མཚན་པ་གཉིས་དང་སྤེལ་བར་བྱའོ། །སྡེ་ཚན་ལྔ་པའི་ལྔ་པ་དང་། དྲུག་པའི་གཉིས་པ་དང་། དབྱངས་ཡིག་གཉིས་པའི་ཐ་མ་དང་། སྡེ་ཚན་དང་པོའི་གསུམ་པ་ལ། མེ་ཡི་ས་བོན་འོག་ཏུ་སྤྱིན། །དབྱངས་ཡིག་གསུམ་པས་བརྒྱན་པར་བྱ། །གསུམ་པའི་ཐ་མ་རིང་པོར་བྱ། དེ་ནས་ལྔ་པའི་བཞི་པ་དང་། གསུམ་པའི་གསུམ་པ་རིང་པོར་བྱ། བཞི་པའི་གསུམ་པ་དང་། མཐར་གནས་པ་ཡི་བཞི་པ་དང་། ལྔ་པའི་དང་པོ་དང་། གཉིས་པའི་དང་པོ་དང་། དེ་དག་གཉིས་གཉིས་སྤེལ་ཏེ་བརྗོད། བཞི་པའི་གསུམ་པ་ལ།དབྱངས་ཡིག་གསུམ་པ་སྦྱར་བ་དང་། གསུམ་པའི་ཐ་མ་དང་། ལྔ་པའི་ཐ་མ་དང་། དབྱངས་ཡིག་བཅུ་གཅིག་པ་སྦྱར་བ་དང་། དང་པོའི་དང་པོ་ལ། དབྱངས་ཡིག་བཅུ་གཅིག་པ་དང་བཅུ་གཉིས་པ་སྦྱར་བ་དང་། དེ་བཞིན་བཞི་པའི་ཐ་མ་བཞག །སེམས་ཀྱི་ས་བོན་སྒྲོད་བ་མཐར། །འབོད་རྦད་བསད་གསུམ་སྔགས་བཏུ་པའོ། །རང་བྱུང་སྔགས་ལ་བཏུ་བ་མེད། ནགཔོ་ཆེན་པོའི་རྒྱུད་དུར་ཁྲོད་ནག་པོ་ལ། 
སྔགས་བཏུ་བའི་ལེའུསྟེབཅུགཅིགཔའོ།། །།དེ་ན་ནག་པོ་ཆེན་པོ་གསོལ་པ། ནག་པོ་ཆེན་པོའི་དམ་ཚིག་ཇི་ལྟར་ལགས། །བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། །སྒྲུབ་པ་པོ་ལ་ཕན་གདགས་ཕྱིར། །དམ་ཚིག་བཤད་ཀྱི་ཉན་པར་གྱིས། །ནག་པོ་ཆེན་པོའི་དཀྱིལ་འཁོར་དུ། །དབང་མ་བསྐུར་བས་མི་སྒྲུབ་ཅིང་། །གྲི་གུག་དང་ནི་ཐོད་པ་དང་། །ཕུར་བུ་དང་ནི་ཐང་ཀ་དང་། །སུ་ལ་བསྟན་པར་མི་བྱ་སྟེ། །རྡོ་རྗེ་སློབ་དཔོན་མི་སྨད་ཅིང་། །གསང་བ་ལས་ནི་བརྙས་མི་བྱ། །ནག་པོའི་རྒྱུད་དང་དམ་རྫས་ནི། །སྣོད་མིན་པ་ལ་བསྟན་མི་བྱ། །ནག་པོའི་གྲུབ་རྟགས་བྱུང་གྱུར་ན། །ལས་ལ་སྦྱར་བ་མ་བྱས་ན། །ནག་པོའི་དམ་ཚིག་ཉམས་གྱུར་ཏེ། །སྔགས་པ་དེ་ནི་འཆི་བར་འགྱུར། །བསྒྲལ་བའི་ཞིང་དུ་མ་གྱུར་པའི། །སྐྱེ་བོ་རྣམས་ལ་བྱས་གྱུར་ན། །ནག་པོའི་དམ་ཚིག་ཉམས་གྱུར་ཏེ། །སྒྲུབ་པ་པོ་རང་གནོད་པར་འགྱུར། །བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། །ནག་པོ་ཆེན་པོའི་རྒྱུད་དུར་ཁྲོད་ནག་པོ་ལས། །དམ་ཚིག་བསྲུང་པའི་ལེའུ་སྟེ་བཅུ་གཉིས་པའོ།། །།བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་སྩལ་པ་དང་། ནག་པོ་ཆེན་པོ་འཁོར་དང་རྗེ་དང་བཅ་པ་རྗེས་སུ་ཡི་རང་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལ་མངོན་པར་བསྟོད་དོ།། །།ནག་པོ་ཆེན་པོའི་རྒྱུད་རྟོགས་པ་བཅུ་གཉིས་པ་ཞེས་བྱ་བ་རྫོགས་སོ།། །།བླ་མ་ནག་པོ་རོ་འཛིན་དང་། 
གྱི་ལྗང་བུ་དཀར་གྱིས་བསྒྱུར་༄༅། །ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ༄༅། །རྒྱ་གར་སྐད་དུ། མནྟྲ་ཨསྣ་ཀ་ཧ་ཛ་ཏནྟྲ། བོད་སྐད་དུ། མན་ངག་བརྩེགས་པ་གྲི་གུག་གི་རྒྱུད། བཅོམ་ལྡན་འདས་རྡོ་རྗེ་སེམས་དཔའ་ལ་ཕྱག་འ་་ལོ། །འདི་སྐད་བདག་གིས་ཐོས་པའི་དུས་གཅིག་ན། བཅོམ་
